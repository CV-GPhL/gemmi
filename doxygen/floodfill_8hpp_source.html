126  void for_each_islands(Func func) {
127  size_t idx = 0;
128  for (int w = 0; w != mask.nw; ++w)
129  for (int v = 0; v != mask.nv; ++v)

173  Line line_from_point(int u, int v, int w, T* ptr) const {
174  int len = 1;
175  while (u + len < mask.nu && ptr[len] == Land)
176  ++len;
177  if (u + len == mask.nu)
178  while (len < mask.nu && ptr[len - mask.nu] == Land)
179  ++len;
180  for (int i = 0; i > -u; --i)
181  if (ptr[i-1] != Land)
182  return {u + i, v, w, len - i, ptr + i};
183  if (ptr[mask.nu-1-u] != Land)
184  return {0, v, w, len + u, ptr - u};
185  for (int i = mask.nu - 1 - u; i > 1; --i)
186  if (ptr[i-1] != Land)
187  return {u + i, v, w, len + mask.nu - 2 - i, ptr + i};
188  return {u, v, w, mask.nu, ptr};
189  }

219 inline Grid<std::int8_t> flood_fill_above(const Grid<float>& grid,
220  const std::vector<Position>& seeds,
221  double threshold,
222  bool negate=false) {
223  grid.check_not_empty();
224  Grid<std::int8_t> mask;
225  mask_nodes_above_threshold(mask, grid, threshold, negate);
226  FloodFill<std::int8_t, 1> flood_fill{mask};
227  for (const Position& pos : seeds) {
228  auto point = mask.get_nearest_point(pos);
229  if (*point.value == 1)
230  flood_fill.find_all_connected_points(point.u, point.v, point.w);
231  }
232  for (std::int8_t& d : mask.data)
233  d = std::int8_t(d == flood_fill.this_island());
234  return mask;
