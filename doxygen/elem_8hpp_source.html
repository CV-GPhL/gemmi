391 inline El find_element(const char* symbol) {
392  if (symbol == nullptr || symbol[0] == '\0')
393  return El::X;
394  char first = symbol[0] & ~0x20; // lower -> upper, space -> NUL
395  char second = symbol[1] & ~0x20;
396  if (first == '\0')
397  return impl::find_single_letter_element(second);
398  // To handle symbol being "X\n" we have the condition below.
399  // In addition to \t, \v, \r and \n it catches also !"#$%&'()*+,- and
400  // some control characters - inconsistent but not necessarily bad.
401  if (second < 14)
402  return impl::find_single_letter_element(first);
403  elname_t* names = &element_uppercase_name(El::X);
404  for (int i = 0; i != 120; ++i) {
405  if (names[i][0] == first && names[i][1] == second)
406  return static_cast<El>(i);
407  }
408  return El::X;
409 }
