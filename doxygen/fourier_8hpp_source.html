66 template<typename DataProxy>
67 std::array<int, 3> get_size_for_hkl(const DataProxy& data,
68  std::array<int, 3> min_size,
69  double sample_rate) {
70  // adjust min_size by checking Miller indices in the data
71  for (size_t i = 0; i < data.size(); i += data.stride()) {
72  Miller hkl = data.get_hkl(i);
73  for (int j = 0; j != 3; ++j) {
74  int v = 2 * std::abs(hkl[j]) + 1;
75  if (v > min_size[j])
76  min_size[j] = v;
77  }
78  }
79  std::array<double, 3> dsize{{(double)min_size[0],
80  (double)min_size[1],
81  (double)min_size[2]}};
82  if (sample_rate > 0) {
83  const UnitCell& cell = data.unit_cell();
84  double max_1_d2 = 0;
85  for (size_t i = 0; i < data.size(); i += data.stride())
86  max_1_d2 = std::max(max_1_d2, cell.calculate_1_d2(data.get_hkl(i)));
87  double inv_d_min = std::sqrt(max_1_d2);
88  std::array<double, 3> cellr = {{cell.ar, cell.br, cell.cr}};
89  for (int i = 0; i < 3; ++i)
90  dsize[i] = std::max(dsize[i], sample_rate * inv_d_min / cellr[i]);
91  }
92  return good_grid_size(dsize, GridSizeRounding::Up, data.spacegroup());
93 }

118 void add_friedel_mates(ReciprocalGrid<T>& grid) {
119  const T default_val = T(); // initialized to 0 or 0+0i
120  if (grid.axis_order == AxisOrder::XYZ) {
121  for (int w = 0; w != (grid.half_l ? 1 : grid.nw); ++w) {
122  int w_ = w == 0 ? 0 : grid.nw - w;
123  for (int v = 0; v != grid.nv; ++v) {
124  int v_ = v == 0 ? 0 : grid.nv - v;
125  for (int u = 0; u != grid.nu; ++u) {
126  size_t idx = grid.index_q(u, v, w);
127  if (grid.data[idx] == default_val) {
128  int u_ = u == 0 ? 0 : grid.nu - u;
129  size_t inv_idx = grid.index_q(u_, v_, w_);
130  grid.data[idx] = friedel_mate_value(grid.data[inv_idx]);
131  }
132  }
133  }
134  }
135  } else { // grid.axis_order == AxisOrder::ZYX
136  for (int w = 0; w != grid.nw; ++w) {
137  int w_ = w == 0 ? 0 : grid.nw - w;
138  for (int v = 0; v != grid.nv; ++v) {
139  int v_ = v == 0 ? 0 : grid.nv - v;
140  if (grid.half_l) {
141  size_t idx = grid.index_q(0, v, w);

224 template<typename T, typename FPhi>
225 FPhiGrid<T> get_f_phi_on_grid(const FPhi& fphi,
226  std::array<int, 3> size, bool half_l,
227  AxisOrder axis_order=AxisOrder::XYZ) {
228  FPhiGrid<T> grid;
229  initialize_hkl_grid(grid, fphi, size, half_l, axis_order);
230  const std::complex<T> default_val; // initialized to 0+0i
231  GroupOps ops = grid.spacegroup->operations();
232  for (size_t i = 0; i < fphi.size(); i += fphi.stride()) {
233  Miller hkl = fphi.get_hkl(i);
234  T f = (T) fphi.get_f(i);
235  if (f == 0.f) // is there enough of F=0 to justify this 'if'?
236  continue;
237  double phi = fphi.get_phi(i);
238  for (const Op& op : ops.sym_ops) {
239  auto hklp = op.apply_to_hkl(hkl);
240  int lp = hklp[2];
241  if (axis_order == AxisOrder::ZYX)
242  std::swap(hklp[0], hklp[2]);
243  if (!grid.has_index(hklp[0], hklp[1], hklp[2]))
244  continue;
245  int sign = (!half_l || lp >= 0 ? 1 : -1);
246  size_t idx = grid.index_n(sign * hklp[0], sign * hklp[1], sign * hklp[2]);
247  if (grid.data[idx] == default_val) {
248  T theta = sign * T(phi + op.phase_shift(hkl));
249  grid.data[idx] = std::complex<T>(f * std::cos(theta), f * std::sin(theta));
250  }
251  }
252  }
253  if (!ops.is_centrosymmetric())
254  add_friedel_mates(grid);
255  return grid;
256 }

268 ReciprocalGrid<T> get_value_on_grid(const DataProxy& data, size_t column,
269  std::array<int, 3> size, bool half_l,
270  AxisOrder axis_order=AxisOrder::XYZ) {

272  initialize_hkl_grid(grid, data, size, half_l, axis_order);
273 
274  if (column >= data.stride())
275  fail("Map coefficients not found.");
276  GroupOps ops = grid.spacegroup->operations();
277  for (size_t i = 0; i < data.size(); i += data.stride()) {
278  Miller hkl = data.get_hkl(i);
279  T val = (T) data.get_num(i + column);
280  if (val != 0.) {
281  for (const Op& op : ops.sym_ops) {
282  auto hklp = op.apply_to_hkl(hkl);

292  }
293  }
294  }

308 template<typename T>
309 void transform_f_phi_grid_to_map_(FPhiGrid<T>&& hkl, Grid<T>& map) {
310  // NaNs are not good for FFT, so we change them to 0.
311  // x -> conj(x) is equivalent to changing axis direction before FFT.
312  for (std::complex<T>& x : hkl.data)
313  if (std::isnan(x.imag()))
314  x = 0;
315  else
316  x.imag(-x.imag());
317  map.spacegroup = hkl.spacegroup;
318  map.unit_cell = hkl.unit_cell;
319  map.axis_order = hkl.axis_order;
320  if (hkl.axis_order == AxisOrder::XYZ) {
321  int nw = hkl.half_l ? 2 * (hkl.nw - 1) : hkl.nw;
322  map.set_size(hkl.nu, hkl.nv, nw);
323  } else { // hkl.axis_order == AxisOrder::ZYX
324  int nu = hkl.half_l ? 2 * (hkl.nu - 1) : hkl.nu;
325  check_grid_factors(map.spacegroup, {{hkl.nw, hkl.nv, nu}});
326  map.set_size_without_checking(nu, hkl.nv, hkl.nw);
327  }
328  // FIXME set_size_without_checking is changing axis_order - bad
329  map.axis_order = hkl.axis_order;
330  pocketfft::shape_t shape{(size_t)hkl.nw, (size_t)hkl.nv, (size_t)hkl.nu};
331  std::ptrdiff_t s = sizeof(T);
332  pocketfft::stride_t stride{2*s * hkl.nv * hkl.nu, 2*s * hkl.nu, 2*s};
333  pocketfft::shape_t axes{2, 1, 0};
334  if (hkl.axis_order == AxisOrder::ZYX)
335  std::swap(axes[0], axes[2]);
336  T norm = T(1.0 / hkl.unit_cell.volume);
337  if (hkl.half_l) {
338  size_t last_axis = axes.back();
339  axes.pop_back();
340  pocketfft::c2c<T>(shape, stride, stride, axes, pocketfft::BACKWARD,
341  &hkl.data[0], &hkl.data[0], norm);
342  pocketfft::stride_t stride_out{s * map.nu * map.nv, s * map.nu, s};
343  shape[0] = (size_t) map.nw;
344  shape[2] = (size_t) map.nu;
345  pocketfft::c2r<T>(shape, stride, stride_out, last_axis, pocketfft::BACKWARD,
346  &hkl.data[0], &map.data[0], 1.0f);
347  } else {
348  pocketfft::c2c<T>(shape, stride, stride, axes, pocketfft::BACKWARD,
349  &hkl.data[0], &hkl.data[0], norm);
350  assert(map.data.size() == hkl.data.size());
351  for (size_t i = 0; i != map.data.size(); ++i)
352  map.data[i] = hkl.data[i].real();
353  }
354 }

417 template<typename T>
418 FPhiGrid<T> transform_map_to_f_phi(const Grid<T>& map, bool half_l, bool use_scale=true) {
419  if (half_l && map.axis_order == AxisOrder::ZYX)
420  fail("transform_map_to_f_phi(): half_l + ZYX order are not supported yet");
421  FPhiGrid<T> hkl;
422  hkl.unit_cell = map.unit_cell;
423  hkl.spacegroup = map.spacegroup;
424  hkl.axis_order = map.axis_order;
425  hkl.half_l = half_l;
426  int half_nw = map.nw / 2 + 1;
427  hkl.set_size_without_checking(map.nu, map.nv, half_l ? half_nw : map.nw);
428  T norm = use_scale ? T(map.unit_cell.volume / map.point_count()) : 1;
429  pocketfft::shape_t shape{(size_t)map.nw, (size_t)map.nv, (size_t)map.nu};
430  std::ptrdiff_t s = sizeof(T);
431  pocketfft::stride_t stride_in{s * hkl.nv * hkl.nu, s * hkl.nu, s};
432  pocketfft::stride_t stride{2*s * hkl.nv * hkl.nu, 2*s * hkl.nu, 2*s};
433  pocketfft::r2c<T>(shape, stride_in, stride, /*axis=*/0, pocketfft::FORWARD,
434  &map.data[0], &hkl.data[0], norm);
435  shape[0] = half_nw;
436  pocketfft::c2c<T>(shape, stride, stride, {1, 2}, pocketfft::FORWARD,
437  &hkl.data[0], &hkl.data[0], 1.0f);
438  if (!half_l) // add Friedel pairs
439  for (int w = half_nw; w != hkl.nw; ++w) {
440  int w_ = hkl.nw - w;
441  for (int v = 0; v != hkl.nv; ++v) {
442  int v_ = v == 0 ? 0 : hkl.nv - v;
443  for (int u = 0; u != hkl.nu; ++u) {
444  int u_ = u == 0 ? 0 : hkl.nu - u;
445  size_t idx = hkl.index_q(u, v, w);
446  size_t inv_idx = hkl.index_q(u_, v_, w_);
447  hkl.data[idx] = hkl.data[inv_idx]; // conj() is called later
448  }
449  }
450  }
451  for (int i = 0; i != hkl.nu * hkl.nv * half_nw; ++i)
452  hkl.data[i].imag(-hkl.data[i].imag());
453  return hkl;
454 }
