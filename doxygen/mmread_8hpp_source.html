42 inline CoorFormat coor_format_from_content(const char* buf, const char* end) {
43  while (buf < end - 8) {
44  if (std::isspace(*buf)) {
45  ++buf;
46  } else if (*buf == '#') {
47  while (buf < end - 8 && *buf != '\n')
48  ++buf;
49  } else if (*buf == '{') {
50  return CoorFormat::Mmjson;
51  } else if (ialpha4_id(buf) == ialpha4_id("data") && buf[4] == '_') {
52  return CoorFormat::Mmcif;
53  } else {
54  return CoorFormat::Pdb;
55  }
56  }
57  return CoorFormat::Unknown;
58 }

131 template<typename T>
132 Structure read_structure(T&& input, CoorFormat format=CoorFormat::Unknown,
133  cif::Document* save_doc=nullptr) {
134  if (format == CoorFormat::Detect) {
135  CharArray mem = read_into_buffer(input);
136  return read_structure_from_memory(mem.data(), mem.size(), input.path(), format, save_doc);
137  }
138  if (save_doc)
139  save_doc->clear();
140  if (format == CoorFormat::Unknown)
141  format = coor_format_from_ext(input.basepath());
142  switch (format) {
143  case CoorFormat::Pdb:
144  return read_pdb(input);
145  case CoorFormat::Mmcif:
146  return make_structure(cif::read(input), save_doc);
147  case CoorFormat::Mmjson: {
148  Structure st = make_structure(cif::read_mmjson(input), save_doc);
149  st.input_format = CoorFormat::Mmjson;
150  return st;
151  }
152  case CoorFormat::ChemComp:
153  return make_structure_from_chemcomp_doc(cif::read(input), save_doc);
154  case CoorFormat::Unknown:
155  case CoorFormat::Detect:
156  fail("Unknown format of " +
157  (input.path().empty() ? "coordinate file" : input.path()) + ".");
158  }
159  unreachable();
160 }
