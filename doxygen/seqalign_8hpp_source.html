143  void backtrack_to_cigar(const std::uint8_t *p, int i, int j) {
144  i--;
145  int j0 = j--;
146  int state = 0;
147  while (i >= 0 && j >= 0) {
148  // at the beginning of the loop, _state_ tells us which state to check
149  // if requesting the H state, find state one maximizes it.
150  uint32_t tmp = p[(std::size_t)i * j0 + j];
151  if (state == 0 || (tmp & (1 << (state + 2))) == 0)
152  state = tmp & 7;
153  if (state == 0) { // match
154  push_cigar(0, 1);
155  --i;
156  --j;
157  } else if (state == 1) { // deletion
158  push_cigar(2, 1);
159  --i;
160  } else { // insertion
161  push_cigar(1, 1);
162  --j;
163  }
164  }
165  if (i >= 0)
166  push_cigar(2, i + 1); // first deletion
167  else if (j >= 0)
168  push_cigar(1, j + 1); // first insertion
169  std::reverse(cigar.begin(), cigar.end());
170  }

177  void count_matches(const std::vector<std::uint8_t>& query,
178  const std::vector<std::uint8_t>& target) {
179  match_count = 0;
180  size_t pos1 = 0, pos2 = 0;
181  for (Item item : cigar)
182  if (item.op() == 'M') {
183  for (uint32_t i = 0; i < item.len(); ++i)
184  if (query[pos1++] == target[pos2++]) {
185  ++match_count;
186  match_string += '|';
187  } else {
188  match_string += '.';
189  }
190  } else if (item.op() == 'I') {
191  pos1 += item.len();
192  match_string.append(item.len(), ' ');
193  } else /*item.op() == 'D'*/ {
194  pos2 += item.len();
195  match_string.append(item.len(), ' ');
196  }
197  }

255 AlignmentResult align_sequences(const std::vector<std::uint8_t>& query,
256  const std::vector<std::uint8_t>& target,
257  const std::vector<int>& target_gapo,
258  std::uint8_t m,
259  const AlignmentScoring& scoring) {
260  // generate the query profile
261  std::int16_t *query_profile = new std::int16_t[query.size() * m];
262  {
263  std::uint32_t mat_size = (std::uint32_t) scoring.matrix_encoding.size();
264  if (mat_size * mat_size != scoring.score_matrix.size())
265  fail("align_sequences: internal error (wrong score_matrix)");
266  std::int32_t i = 0;
267  for (std::uint8_t k = 0; k < m; ++k)
268  for (std::uint8_t q : query)
269  if (k < mat_size && q < mat_size)
270  query_profile[i++] = scoring.score_matrix[k * mat_size + q];
271  else
272  query_profile[i++] = (k == q ? scoring.match : scoring.mismatch);
273  }
274 
275  struct eh_t { std::int32_t h, e; };
276  eh_t *eh = new eh_t[query.size() + 1];
277  std::int32_t gape = scoring.gape;
278  std::int32_t gapoe = scoring.gapo + gape;
279 
280  // fill the first row
281  {
282  std::int32_t gap0 = !target_gapo.empty() ? target_gapo[0] + gape : gapoe;
283  eh[0].h = 0;
284  eh[0].e = gap0 + gapoe;
285  for (std::int32_t j = 1; j <= (std::int32_t)query.size(); ++j) {
286  eh[j].h = gap0 + gape * (j - 1);
287  eh[j].e = gap0 + gapoe + gape * j;
288  }
289  }
290 
291  // backtrack matrix; in each cell: f<<4|e<<2|h
292  std::uint8_t *z = new std::uint8_t[query.size() * target.size()];
293  // DP loop
294  for (std::int32_t i = 0; i < (std::int32_t)target.size(); ++i) {
295  std::uint8_t target_item = target[i];
296  std::int16_t *scores = &query_profile[target_item * query.size()];
297  std::uint8_t *zi = &z[i * query.size()];
298  std::int32_t h1 = gapoe + gape * i;
299  std::int32_t f = gapoe + gapoe + gape * i;
300  std::int32_t gapx = i+1 < (std::int32_t)target_gapo.size()
301  ? target_gapo[i+1] + gape : gapoe;
302  for (std::size_t j = 0; j < query.size(); ++j) {
303  // At the beginning of the loop:
304  // eh[j] = { H(i-1,j-1), E(i,j) }, f = F(i,j) and h1 = H(i,j-1)
305  // Cells are computed in the following order:
306  // H(i,j) = max{H(i-1,j-1) + S(i,j), E(i,j), F(i,j)}
307  // E(i+1,j) = max{H(i,j)+gapo, E(i,j)} + gape
308  // F(i,j+1) = max{H(i,j)+gapo, F(i,j)} + gape
309  eh_t *p = &eh[j];
310  std::int32_t h = p->h;
311  std::int32_t e = p->e;
312  p->h = h1;
313  h += scores[j];
314  std::uint8_t direction = 0;
315  if (h < e) {
316  direction = 1; // deletion
317  h = e;
318  }
319  if (h <= f) {
320  direction = 2; // insertion
321  h = f;
322  }
323  h1 = h;
324 
325  h += gapoe;
326  e += gape;
327  if (e > h)
328  direction |= 0x08;
329  else
330  e = h;
331 
332  h = h1 + gapx;
333  p->e = e;
334  f += gape;
335  if (f > h)
336  direction |= 0x10;
337  else
338  f = h;
339 
340  // z[i,j] keeps h for the current cell and e/f for the next cell
341  zi[j] = direction;
342  }
343  eh[query.size()].h = h1;
344  eh[query.size()].e = -0x40000000; // -infinity
345  }
346 
347  AlignmentResult result;
348  result.score = eh[query.size()].h;
349  delete [] query_profile;
350  delete [] eh;
351  result.backtrack_to_cigar(z, (int)target.size(), (int)query.size());
352  delete [] z;
353  result.count_matches(query, target);
354  return result;
355 }

367 AlignmentResult align_string_sequences(const std::vector<std::string>& query,
368  const std::vector<std::string>& target,
369  const std::vector<int>& target_gapo,
370  const AlignmentScoring* scoring) {
371  if (scoring == nullptr)
372  scoring = AlignmentScoring::simple();
373  std::map<std::string, std::uint8_t> encoding;
374  for (const std::string& res_name : scoring->matrix_encoding)
375  encoding.emplace(res_name, (std::uint8_t)encoding.size());
376  for (const std::string& s : query)
377  encoding.emplace(s, (std::uint8_t)encoding.size());
378  for (const std::string& s : target)
379  encoding.emplace(s, (std::uint8_t)encoding.size());
380  if (encoding.size() > 255)
381  return AlignmentResult();
382  std::vector<std::uint8_t> encoded_query(query.size());
383  for (size_t i = 0; i != query.size(); ++i)
384  encoded_query[i] = encoding.at(query[i]);
385  std::vector<std::uint8_t> encoded_target(target.size());
386  for (size_t i = 0; i != target.size(); ++i)
387  encoded_target[i] = encoding.at(target[i]);
388  return align_sequences(encoded_query, encoded_target,
389  target_gapo, (std::uint8_t)encoding.size(), *scoring);
390 }
