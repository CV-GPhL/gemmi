24 inline std::complex<double> calculate_sf_part(const Fractional& fpos,
25  const Miller& hkl) {
26  double arg = 2 * pi() * (hkl[0]*fpos.x + hkl[1]*fpos.y + hkl[2]*fpos.z);
27  return std::complex<double>{std::cos(arg), std::sin(arg)};

64  double get_scattering_factor(Element element, signed char charge) {
65  double& sfactor = scattering_factors_[element.ordinal()];
66  if (sfactor == 0.) {
67  if (!Table::has(element.elem))
68  fail("Missing scattering factor for ", element.name());
69  sfactor = Table::get(element.elem, charge).calculate_sf(stol2_) + addends.get(element);

116  template<typename Site>
117  std::complex<double> calculate_sf_from_atom_sf(const Fractional& fract,
118  const Site& site,
119  const Miller& hkl,
120  double sf) {
121  double oc_sf = site.occ * sf;
122  std::complex<double> sum = calculate_sf_part(fract, hkl);
123  if (!site.aniso.nonzero()) {
124  for (const FTransform& image : cell_.images)
125  sum += calculate_sf_part(image.apply(fract), hkl);
126  return oc_sf * dwf_iso(site) * sum;
127  }
128  Vec3 vhkl(hkl[0], hkl[1], hkl[2]);
129  sum *= dwf_aniso(site, vhkl);
130  for (const FTransform& image : cell_.images)
131  sum += calculate_sf_part(image.apply(fract), hkl) *
132  dwf_aniso(site, image.mat.left_multiply(vhkl));
133  return oc_sf * sum;
134  }

155  std::complex<double> calculate_sf_from_model(const Model& model, const Miller& hkl) {
156  std::complex<double> sf = 0.;
157  set_stol2_and_scattering_factors(hkl);
158  for (const Chain& chain : model.chains)
159  for (const Residue& res : chain.residues)
160  for (const Atom& site : res.atoms)
161  sf += calculate_sf_from_atom(cell_.fractionalize(site.pos), site, hkl);
162  return sf;
163  }

171  std::complex<double> calculate_mb_z(const Model& model, const Miller& hkl, bool only_h) {
172  std::complex<double> sf = 0.;
173  stol2_ = (coef_type) cell_.calculate_stol_sq(hkl);
174  for (const Chain& chain : model.chains)
175  for (const Residue& res : chain.residues)
176  for (const Atom& site : res.atoms)

178  sf += calculate_sf_from_atom_sf(cell_.fractionalize(site.pos), site, hkl,
