124 inline std::pair<double, double> calculate_b_aniso_range(const Model& model) {
125  std::pair<double, double> range{INFINITY, -INFINITY};
126  for (const Chain& chain : model.chains)
127  for (const Residue& residue : chain.residues)
128  for (const Atom& atom : residue.atoms) {
129  if (atom.occ == 0)
130  continue;
131  if (atom.aniso.nonzero()) {
132  std::array<double,3> eig = atom.aniso.calculate_eigenvalues();
133  auto u = std::minmax({eig[0], eig[1], eig[2]});
134  range.first = std::min(range.first, u.first * u_to_b());
135  range.second = std::max(range.second, u.second * u_to_b());
136  } else {
137  range.first = std::min(range.first, (double) atom.b_iso);
138  range.second = std::max(range.second, (double) atom.b_iso);
139  }
140  }
141  return range;
142 }

219 inline double calculate_dihedral(const Position& p0, const Position& p1,
220  const Position& p2, const Position& p3) {
221  Vec3 b0 = p1 - p0;
222  Vec3 b1 = p2 - p1;
223  Vec3 b2 = p3 - p2;
224  Vec3 u = b1.cross(b0);
225  Vec3 w = b2.cross(b1);
226  double y = u.cross(w).dot(b1);
227  double x = u.dot(w) * b1.length();
228  return std::atan2(y, x);
229 }
