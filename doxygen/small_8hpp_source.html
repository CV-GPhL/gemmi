115  void determine_and_set_spacegroup(const char* order) {
116  spacegroup = nullptr;
117  if (order)
118  for (const char* c = order; *c != '\0' && spacegroup == nullptr; ++c) {
119  try {
120  GroupOps gops;
121  spacegroup = determine_spacegroup_from(*c, gops);
122  if (!spacegroup && *(c+1) == '.') {
123  // If symops don't correspond to tabulated settings,
124  // we can't set spacegroup, but we can set UnitCell::images.
125  if (gops.order() == (int) symops.size() && is_complete(gops)) {
126  cell.set_cell_images_from_groupops(gops);
127  return;
128  }
129  ++c;
130  }
131  } catch (std::exception&) {}
132  }
133  setup_cell_images();
134  }

141  const SpaceGroup* determine_spacegroup_from(char c, GroupOps& gops) const {
142  switch (lower(c)) {
143  case 's':
144  if (symops.empty())
145  return nullptr;
146  gops = split_centering_vectors(triplets_to_ops(symops));
147  return find_spacegroup_by_ops(gops);
148  case 'h':
149  if (spacegroup_hall.empty())
150  return nullptr;
151  return find_spacegroup_by_ops(symops_from_hall(spacegroup_hall.c_str()));
152  case '1':
153  case '2': {
154  if (spacegroup_hm.empty())
155  return nullptr;
156  char prefer[] = {c, '\0'};
157  return find_spacegroup_by_name(spacegroup_hm, cell.alpha, cell.gamma, prefer);
158  }
159  case 'n':
160  if (spacegroup_number == 0)
161  return nullptr;
162  return find_spacegroup_by_number(spacegroup_number);
163  default:
164  throw std::invalid_argument("determine_and_set_spacegroup(): wrong character in 'order'");
165  }

172  std::string check_spacegroup() const {
173  std::string err;
174  if (!symops.empty())
175  try {
176  std::vector<Op> ops = triplets_to_ops(symops);
177  for (Op& op : ops)
178  op.wrap();
179  std::sort(ops.begin(), ops.end());
180  GroupOps gops = split_centering_vectors(ops);
181  if (!is_complete(gops))
182  cat_to(err, "symops list is incomplete or incorrect\n");
183  else if (gops.all_ops_sorted() != ops)
184  cat_to(err, "symops list is incorrect or incomplete or redundant\n");
185  const SpaceGroup* sg = find_spacegroup_by_ops(gops);
186  if (!sg)
187  cat_to(err, "space group from symops not found in the table\n");
188  else if (sg != spacegroup)
189  cat_to(err, "space group from symops differs: ", sg->xhm(), '\n');
190  } catch (std::exception& e) {
191  cat_to(err, "error while processing symops: ", e.what(), '\n');
192  }
193  if (!spacegroup_hall.empty())
194  try {
195  const SpaceGroup* sg = find_spacegroup_by_ops(symops_from_hall(spacegroup_hall.c_str()));
196  if (!sg)
197  cat_to(err, "space group from Hall symbol (", spacegroup_hall,
198  ") not found in the table\n");
199  else if (spacegroup != sg)
200  cat_to(err, "space group from Hall symbol (", spacegroup_hall,
201  ") differs: ", sg->xhm(), '\n');
202  } catch (std::exception& e) {
203  cat_to(err, "error while processing Hall symbol: ", e.what(), '\n');
204  }
205  if (!spacegroup_hm.empty()) {
206  const SpaceGroup* sg = find_spacegroup_by_name(spacegroup_hm, cell.alpha, cell.gamma, "2");
207  if (!sg)
208  cat_to(err, "H-M symbol (", spacegroup_hm, ") not found in the table\n");
209  else if (!spacegroup || strcmp(spacegroup->hm, sg->hm) != 0)
210  cat_to(err, "space group from H-M symbol (", spacegroup_hm,
211  ") differs: ", sg->hm, '\n');
212  }
213  if (spacegroup_number != 0 && spacegroup && spacegroup->number != spacegroup_number)
214  cat_to(err, "space group number (", spacegroup_number, ") differs\n");
215  return err;
216  }

273 template<typename T>
274 inline void split_element_and_charge(const std::string& label, T* dest) {
275  int len = label.size() > 1 && std::isalpha(label[1]) ? 2 : 1;
276  dest->element = len == 1 ? impl::find_single_letter_element(label[0] & ~0x20)
277  : find_element(label.c_str());
278  if (dest->element != El::X && (label.back() == '+' || label.back() == '-')) {
279  int sign = label.back() == '+' ? 1 : -1;
280  if (label.size() - len == 1)
281  dest->charge = sign;
282  else if (label.size() - len == 2 && label[len] >= '0' && label[len] <= '9')
283  dest->charge = sign * (label[len] - '0');
284  }
285 }

292 inline std::vector<SmallStructure::Site>
293 SmallStructure::get_all_unit_cell_sites() const {
294  const double SPECIAL_POS_TOL = 0.4;
295  std::vector<Site> all;
296  for (const Site& site : sites) {
297  size_t start = all.size();
298  all.push_back(site);
299  for (const FTransform& image : cell.images) {
300  Fractional fpos = image.apply(site.fract);
301  if (std::any_of(all.begin() + start, all.end(), [&](const Site& other) {
302  return cell.distance_sq(fpos, other.fract) < sq(SPECIAL_POS_TOL);
303  }))
304  continue;
305  all.push_back(site);
306  all.back().fract = fpos;
307  }
308  }
309  return all;
310 }
