109 inline bool are_connected2(const Residue& r1, const Residue& r2, PolymerType ptype) {
110  auto this_or_first = [](const Atom* a, const Residue& r, El el) -> const Atom* {
111  if (a || r.atoms.empty())
112  return a;
113  if (const Atom* b = r.find_by_element(el))
114  return b;
115  return &r.atoms.front();
116  };
117  if (is_polypeptide(ptype)) {
118  const Atom* a1 = this_or_first(r1.get_ca(), r1, El::C);
119  const Atom* a2 = this_or_first(r2.get_ca(), r2, El::C);
120  return a1 && a2 && a1->pos.dist_sq(a2->pos) < sq(5.0);
121  }
122  if (is_polynucleotide(ptype)) {
123  const Atom* a1 = this_or_first(r1.get_p(), r1, El::P);
124  const Atom* a2 = this_or_first(r2.get_p(), r2, El::P);
125  return a1 && a2 && a1->pos.dist_sq(a2->pos) < sq(7.5);
126  }
127  return false;
128 }

137 inline bool are_connected3(const Residue& r1, const Residue& r2, PolymerType ptype) {
138  if (is_polypeptide(ptype)) {
139  if (const Atom* a1 = r1.get_c())
140  if (const Atom* a2 = r2.get_n())
141  return a1->pos.dist_sq(a2->pos) < sq(1.341 * 1.5);
142  if (const Atom* a1 = r1.get_ca())
143  if (const Atom* a2 = r2.get_ca())
144  return a1->pos.dist_sq(a2->pos) < sq(5.0);
145  } else if (is_polynucleotide(ptype)) {
146  if (const Atom* a1 = r1.get_o3prim())
147  if (const Atom* a2 = r2.get_p())
148  return a1->pos.dist_sq(a2->pos) < sq(1.6 * 1.5);
149  if (const Atom* a1 = r1.get_p())
150  if (const Atom* a2 = r2.get_p())
151  return a1->pos.dist_sq(a2->pos) < sq(7.5);
152  }
153  return false;
