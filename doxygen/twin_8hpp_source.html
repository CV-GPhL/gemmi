141 inline double calculate_cos_obliquity(const UnitCell& reduced_cell,
142  const Vec3& d_axis, const Vec3& r_axis) {
143  // From the Le Page paper:
144  // tan(delta) = |t x tau| / |t . tau| (i.e. delta is angle 0...90deg)
145  // t = L U where U is direct-space axis
146  // tau = (L^T)^-1 h where h is reciprocal-space axis
147  // L is orthogonalization matrix, although in a different order
148  Vec3 t = reduced_cell.orth.mat.multiply(d_axis);
149  Vec3 tau = reduced_cell.frac.mat.left_multiply(r_axis);
150  // it's faster to calculate cos(delta) than tan(delta)
151  return std::min(1.0, std::fabs(t.cos_angle(tau)));
152 }

158 inline std::vector<OpObliquity> find_lattice_2fold_ops(const UnitCell& reduced_cell,
159  double max_obliq) {
160  std::vector<OpObliquity> ret;
161  const double cos_max_obliq = std::cos(rad(max_obliq));
162  for (const impl::TwoFoldData& row : impl::TwoFold::table) {
163  Vec3 d_axis(row.ds_axis[0], row.ds_axis[1], row.ds_axis[2]);
164  Vec3 r_axis(row.rs_axis[0], row.rs_axis[1], row.rs_axis[2]);
165  double cos_delta = calculate_cos_obliquity(reduced_cell, d_axis, r_axis);
166  if (cos_delta > cos_max_obliq) {
167  constexpr int D = Op::DEN;
168  Op op{Op::Rot{{{{row.matrix[0] * D, row.matrix[1] * D, row.matrix[2] * D}},
169  {{row.matrix[3] * D, row.matrix[4] * D, row.matrix[5] * D}},
170  {{row.matrix[6] * D, row.matrix[7] * D, row.matrix[8] * D}}}},
171  Op::Tran{{0, 0, 0}}, 'x'};
172  ret.emplace_back(op, deg(std::acos(cos_delta)));
173  }
174  }
175  std::sort(ret.begin(), ret.end(),
176  [](const OpObliquity& a, const OpObliquity& b) { return a.second < b.second; });
177  return ret;
178 }

184 inline GroupOps find_lattice_symmetry_r(const UnitCell& reduced_cell, double max_obliq) {
185  std::vector<OpObliquity> gen = find_lattice_2fold_ops(reduced_cell, max_obliq);
186  std::vector<Op> genops;
187  genops.reserve(gen.size());
188  for (const OpObliquity& op_obl : gen)
189  genops.push_back(op_obl.first);
190  GroupOps go;
191  go.sym_ops.push_back(Op::identity());
192  go.cen_ops.push_back({0,0,0});
193  // cf. GroupOps::add_missing_elements()
194  if (!gen.empty())
195  go.sym_ops.push_back(gen[0].first);
196  // no need to try operator^2, we know it must be identity
197  go.add_missing_elements_part2(genops, 24, true);
198  return go;
199 }

206 inline GroupOps find_lattice_symmetry(const UnitCell& cell, char centring,
207  double max_obliq) {

231 inline std::vector<Op> find_twin_laws(const UnitCell& cell,
232  const SpaceGroup* sg,
233  double max_obliq,
234  bool all_ops) {
235  if (sg == nullptr)
236  sg = &get_spacegroup_p1();
237  GroupOps go = sg->operations();
238  GroupOps lat_go = find_lattice_symmetry(cell, sg->centring_type(), max_obliq);
239  if (!go.has_same_centring(lat_go))
240  fail("find_twin_laws(): internal error");
241  std::vector<Op> ops;
242  size_t sg_symop_count = go.sym_ops.size();
243  for (const Op& op : lat_go.sym_ops)
244  if (!go.find_by_rotation(op.rot) &&
245  !go.find_by_rotation(op.negated_rot())) {
246  ops.push_back(op);
247  if (!all_ops)
248  for (size_t i = 1; i < sg_symop_count; ++i)
249  go.sym_ops.push_back(op * go.sym_ops[i]);
250  }
251  return ops;
252 }
