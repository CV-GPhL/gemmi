34 template<int N, typename Real>
35 Real determine_cutoff_radius(Real x1, const ExpSum<N, Real>& precal, Real cutoff_level) {
36  Real y1, dy;
37  std::tie(y1, dy) = precal.calculate_with_derivative(x1);
38  // Generally, density is supposed to decrease with radius.
39  // But if we have addends (in particular -Z for Mott-Bothe),
40  // it can first rise, then decrease. We want to be after the maximum.
41  while (std::copysign(dy, y1 * dy) > 0) { // unlikely
42  x1 += 1.0f;
43  std::tie(y1, dy) = precal.calculate_with_derivative(x1);
44  }
45  Real x2 = x1;
46  Real y2 = y1;
47  if (std::fabs(y1) < cutoff_level) {
48  while (std::fabs(y1) < cutoff_level) {
49  x2 = x1;
50  y2 = y1;
51  x1 -= 0.5f;
52  std::tie(y1, dy) = precal.calculate_with_derivative(x1);
53  // with addends it's possible to land on the left side of the maximum
54  if (std::copysign(dy, y1 * dy) > 0) { // unlikely
55  while (std::copysign(dy, y1 * dy) > 0 && x1 + 0.1f < x2) {
56  x1 += 0.1f;
57  std::tie(y1, dy) = precal.calculate_with_derivative(x1);
58  }
59  if (std::fabs(y1) < cutoff_level)
60  return x1;
61  break;
62  }
63  if (x1 < 0) { // unlikely
64  x1 = 0;
65  y1 = precal.calculate(x1 * x1);
66  break;
67  }
68  }
69  } else {
70  while (std::fabs(y2) > cutoff_level) {
71  x1 = x2;
72  y1 = y2;
73  x2 += 0.5f;
74  y2 = precal.calculate(x2 * x2);
75  }
76  }
77 
78  return x1 + (x1 - x2) / (y1 - y2) * (cutoff_level - y1);
79 }

200  template<typename Coef>
201  void do_add_atom_density_to_grid(const Atom& atom, const Coef& coef, float addend) {
202 #if GEMMI_COUNT_DC
203  ++atoms_added;
204 #endif
205  Fractional fpos = grid.unit_cell.fractionalize(atom.pos);
206  if (!atom.aniso.nonzero()) {
207  // isotropic
208  CReal b = static_cast<CReal>(atom.b_iso + blur);
209  auto precal = coef.precalculate_density_iso(b, addend);
210  CReal radius = estimate_radius(precal, b);
211  grid.template use_points_around<true>(fpos, radius, [&](GReal& point, double r2) {
212  point += GReal(atom.occ * precal.calculate((CReal)r2));
213 #if GEMMI_COUNT_DC
214  ++density_computations;
215 #endif
216  }, /*fail_on_too_large_radius=*/false);
217  } else {
218  // anisotropic
219  auto aniso_b = atom.aniso.scaled(CReal(u_to_b())).added_kI(CReal(blur));
220  // rough estimate, so we don't calculate eigenvalues
221  CReal b_max = std::max(std::max(aniso_b.u11, aniso_b.u22), aniso_b.u33);
222  auto precal_iso = coef.precalculate_density_iso(b_max, addend);
223  double radius = estimate_radius(precal_iso, b_max);
224  auto precal = coef.precalculate_density_aniso_b(aniso_b, addend);
225  int du = (int) std::ceil(radius / grid.spacing[0]);
226  int dv = (int) std::ceil(radius / grid.spacing[1]);
227  int dw = (int) std::ceil(radius / grid.spacing[2]);
228  grid.template use_points_in_box<true>(
229  fpos, du, dv, dw,
230  [&](GReal& point, double, const Position& delta, int, int, int) {
231  point += GReal(atom.occ * precal.calculate(delta));
232 #if GEMMI_COUNT_DC
233  ++density_computations;
234 #endif
235  },
236  false, radius);
237  }
238  }
