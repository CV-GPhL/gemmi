26 inline Mtz xds_to_mtz(XdsAscii& xds) {
27  if (xds.is_merged()) {
28  Intensities intensities;
29  intensities.import_xds(xds);
30  return intensities.prepare_merged_mtz(/*with_nobs=*/false);
31  }
32  Mtz mtz;
33  mtz.cell.set_from_array(xds.cell_constants);
34  mtz.spacegroup = find_spacegroup_by_number(xds.spacegroup_number);
35  mtz.add_base();
36  const char* pxd[3] = {"XDSproject", "XDScrystal", "XDSdataset"};
37  if (xds.isets.empty()) {
38  mtz.datasets.push_back({1, pxd[0], pxd[1], pxd[2], mtz.cell, xds.wavelength});
39  } else {
40  for (XdsAscii::Iset& iset : xds.isets) {
41  double wavelength = iset.wavelength != 0 ? iset.wavelength : xds.wavelength;
42  UnitCell cell;
43  cell.set_from_array(iset.cell_constants[0] != 0 ? iset.cell_constants
44  : xds.cell_constants);
45  mtz.datasets.push_back({iset.id, pxd[0], pxd[1], pxd[2], cell, wavelength});
46  }
47  }
48  mtz.add_column("M/ISYM", 'Y', 0, -1, false);
49  mtz.add_column("BATCH", 'B', 0, -1, false);
50  mtz.add_column("I", 'J', 0, -1, false);
51  mtz.add_column("SIGI", 'Q', 0, -1, false);
52  mtz.add_column("XDET", 'R', 0, -1, false);
53  mtz.add_column("YDET", 'R', 0, -1, false);
54  mtz.add_column("ROT", 'R', 0, -1, false);
55  if (xds.read_columns >= 11) {
56  mtz.add_column("FRACTIONCALC", 'R', 0, -1, false);
57  mtz.add_column("LP", 'R', 0, -1, false);
58  mtz.add_column("CORR", 'R', 0, -1, false);
59  if (xds.read_columns > 11)
60  mtz.add_column("MAXC", 'I', 0, -1, false);
61  }
62  mtz.add_column("FLAG", 'I', 0, -1, false);
63  mtz.nreflections = (int) xds.data.size();
64  mtz.data.resize(mtz.columns.size() * xds.data.size());
65  UnmergedHklMover hkl_mover(mtz.spacegroup);
66  int max_frame = 0;
67  for (const XdsAscii::Refl& refl : xds.data)
68  max_frame = std::max(max_frame, refl.frame());
69  int iset_offset = (max_frame + 11000) / 10000 * 10000;
70  // iset,frame -> batch
71  std::map<std::pair<int,int>, int> frames;
72  size_t k = 0;
73  for (const XdsAscii::Refl& refl : xds.data) {
74  auto hkl = refl.hkl;
75  int isym = hkl_mover.move_to_asu(hkl);
76  for (size_t j = 0; j != 3; ++j)
77  mtz.data[k++] = (float) hkl[j];
78  mtz.data[k++] = (float) isym;
79  int frame = refl.frame();
80  int batch = frame + iset_offset * std::max(refl.iset - 1, 0);
81  frames.emplace(std::make_pair(refl.iset, frame), batch);
82  mtz.data[k++] = (float) batch;
83  mtz.data[k++] = (float) refl.iobs; // I
84  mtz.data[k++] = (float) std::fabs(refl.sigma); // SIGI
85  mtz.data[k++] = (float) refl.xd;
86  mtz.data[k++] = (float) refl.yd;
87  mtz.data[k++] = (float) xds.rot_angle(refl); // ROT
88  if (xds.read_columns >= 11) {
89  mtz.data[k++] = float(0.01 * refl.peak); // FRACTIONCALC
90  mtz.data[k++] = (float) refl.rlp;
91  mtz.data[k++] = float(0.01 * refl.corr);
92  if (xds.read_columns > 11)
93  mtz.data[k++] = (float) refl.maxc;
94  }
95  mtz.data[k++] = refl.sigma < 0 ? 64.f : 0.f; // FLAG
96  }
97  // Prepare a similar batch header as Pointless.
98  Mtz::Batch batch;
99  batch.set_dataset_id(1);
100 
101  // We don't set lbcell (refinement flags for unit cell),
102  // because it's probably not used by any program anyway.
103  // batch.ints[4] to [9] = left unset
104 
105  // We also skip jumpax, which is defined as:
106  // reciprocal axis closest to principle goniostat axis E1
107  // batch.ints[11] = left unset
108 
109  // We assume one crystal, one goniostat axis, one detector.
110  batch.ints[12] = 1; // ncryst
111  batch.ints[14] = 2; // ldtype 3D
112  batch.ints[15] = 1; // jsaxs - goniostat scan axis number
113  batch.ints[17] = 1; // ngonax - number of goniostat axes
114  batch.ints[19] = 1; // ndet
115 
116  batch.set_cell(mtz.cell); // batch.floats[0] to [5]
117  Vec3 s0(-1, 0, 0); // will be re-set if we have geometry info
118  try {
119  Mat33 Q = xds.calculate_conversion_from_cambridge().inverse();
120  s0 = -Q.multiply(xds.get_s0_direction());
121  // Orientation matrix U. It is calculated differently in Pointless,
122  // so the results are slightly different (due to limited precision
123  // of numbers in XDS file).
124  Mat33 U = Q.multiply(xds.get_orientation());
125  for (int i = 0; i < 3; ++i)
126  for (int j = 0; j < 3; ++j)
127  batch.floats[6 + 3*i + j] = (float) U[j][i];
128  } catch (std::runtime_error& e) {
129  // if some of the headers are absent, U is not set
130  // and s0 is set to "idealised" s0
131  std::fprintf(stderr, "Note: %s, orientation matrix U not set.\n", e.what());
132  }
133  batch.floats[21] = float(xds.reflecting_range_esd); // crydat(0)
134  // In the so-called "Cambridge" frame (as used by Mosflm),
135  // the principal rotation axis is along z
136  // and the incident beam S0 is along x.
137  // Therefore, we set the rotation axis phi (scanax) to:
138  batch.floats[38+2] = 1.f; // scanax = [0, 0, 1]
139  batch.floats[47] = float(xds.oscillation_range); // phi range
140  // E1,E2,E3 vectors are the goniostat axes in Cambridge laboratory frame.
141  // E1 is set to rotation axis. E2 and E3 are not set for ngonax==1.
142  batch.floats[59+2] = 1.f; // e1 = scanax
143 
144  // Idealised source vector is -x ([-1 0 0]), antiparallel to beam.
145  batch.floats[80+0] = -1.f; // source[0]
146 
147  // s0 source vector (including tilts), CMtz::MTZBAT::so in libccp4
148  batch.floats[83] = (float) s0.x;
149  batch.floats[84] = (float) s0.y;
150  batch.floats[85] = (float) s0.z;
151 
152  batch.set_wavelength((float)xds.wavelength); // batch.floats[86]
153  // or batch.set_wavelength(iset.wavelength);
154  // Detector geometry.
155  batch.floats[111] = (float) xds.detector_distance; // dx[0]
156  batch.floats[113] = 1.f; // detlm[0][0][0]
157  batch.floats[114] = (float) xds.nx;
158  batch.floats[115] = 1.f;
159  batch.floats[116] = (float) xds.ny;
160  batch.axes.push_back("PHI"); // gonlab[0]
161 
162  for (auto& t : frames) {
163  batch.set_dataset_id(t.first.first);
164  batch.number = t.second;
165  int frame = t.first.second;
166  double phistt = xds.starting_angle +
167  xds.oscillation_range * (frame - xds.starting_frame);
168  batch.floats[36] = float(phistt);
169  batch.floats[37] = float(phistt + xds.oscillation_range); // phiend
170  mtz.batches.push_back(batch);
171  }
172  mtz.sort(5);
173  return mtz;
174 }
