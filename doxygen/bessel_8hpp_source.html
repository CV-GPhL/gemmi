27 inline double evaluate_polynomial(const double(&poly)[N], double x) {
28  static_assert(N > 1, "");
29  double result = poly[N-1];
30  for (int i = N-2; i >= 0; --i)
31  result = result * x + poly[i];

97 inline double bessel_i1_over_i0(double x) {
98  using B = BesselTables_<void>;
99  if (x < 0)
100  return -bessel_i1_over_i0(-x);
101  if (x < 7.75) {
102  double a = x * x / 4;
103  double bessel0 = a * evaluate_polynomial(B::Q1, a) + 1;
104  double R[3] = { 1, 0.5f, evaluate_polynomial(B::P1, a) };
105  double bessel1 = x * evaluate_polynomial(R, a) / 2;
106  return bessel1 / bessel0;
107  }
108  double p = evaluate_polynomial(B::P2, 1 / x);
109  double q = x < 50 ? evaluate_polynomial(B::Q2, 1 / x)
110  : evaluate_polynomial(B::Q3, 1 / x);
111  return p / q;
112 }

119 inline double bessel_i0(double x) {
120  using B = BesselTables_<void>;
121  x = std::fabs(x);
122  if (x < 7.75) {
123  double a = x * x / 4;
124  return a * evaluate_polynomial(B::Q1, a) + 1;
125  }
126  if (x < 50)
127  return std::exp(x) * evaluate_polynomial(B::Q2, 1 / x) / std::sqrt(x);
128  double ex = std::exp(x / 2);
129  return ex * evaluate_polynomial(B::Q3, 1 / x) / std::sqrt(x) * ex;
130 }
