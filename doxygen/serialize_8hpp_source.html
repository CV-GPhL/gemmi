191 template <typename Archive>
195 void serialize(Archive& archive, Item& o) {
196  archive(o.type, o.line_number);
197  switch (o.type) {
198  case ItemType::Pair:
199  case ItemType::Comment: new(&o.pair) cif::Pair; archive(o.pair); break;
200  case ItemType::Loop: new(&o.loop) cif::Loop; archive(o.loop); break;
201  case ItemType::Frame: new(&o.frame) cif::Block; archive(o.frame); break;
202  case ItemType::Erased: break;
203  }
204 }
