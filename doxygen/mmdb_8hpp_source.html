108 inline void transfer_links_to_mmdb(const Structure& st, mmdb::Manager* mol) {
109  // based on code provided by Paul Emsley
110  for (const Connection& con : st.connections) {
111  if (!con.partner1.res_id.seqid.num || !con.partner2.res_id.seqid.num)
112  continue;
113  mmdb::Link link{};
114  // partner1
115  strcpy_to_mmdb(link.atName1, con.partner1.atom_name);
116  link.aloc1[0] = con.partner1.altloc;
117  set_seqid_in_mmdb(&link.seqNum1, link.insCode1, con.partner1.res_id.seqid);
118  strcpy_to_mmdb(link.resName1, con.partner1.res_id.name);
119  strcpy_to_mmdb(link.chainID1, con.partner1.chain_name);
120  // partner2
121  strcpy_to_mmdb(link.atName2, con.partner2.atom_name);
122  link.aloc2[0] = con.partner2.altloc;
123  set_seqid_in_mmdb(&link.seqNum2, link.insCode2, con.partner2.res_id.seqid);
124  strcpy_to_mmdb(link.resName2, con.partner2.res_id.name);
125  strcpy_to_mmdb(link.chainID2, con.partner2.chain_name);
126  if (con.reported_distance > 0)
127  link.dist = con.reported_distance;
128  if (con.asu == Asu::Different) {
129  link.s2 = con.reported_sym[0];
130  link.i2 = link.i1 + con.reported_sym[1];
131  link.j2 = link.j1 + con.reported_sym[2];
132  link.k2 = link.k1 + con.reported_sym[3];
133  }
134  // add links to models
135  for (int imod = 1; imod <= mol->GetNumberOfModels(); imod++)
136  if (mmdb::Model* model_p = mol->GetModel(imod))
137  model_p->AddLink(new mmdb::Link(link));
