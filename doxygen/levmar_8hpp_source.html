36 inline void jordan_solve(double* a, double* b, int n) {
37  for (int i = 0; i < n; i++) {
38  // looking for a pivot element
39  int maxnr = -1;
40  double amax = 0;
41  for (int j = i; j < n; j++) {
42  double aji = std::fabs(a[n * j + i]);
43  if (aji > amax) {
44  maxnr = j;
45  amax = aji;
46  }
47  }
48  // handle singular matrix
49  if (maxnr == -1) {
50  // i-th column has only zeros.
51  // If it's the same about i-th row, and b[i]==0, let x[i]==0.
52  for (int j = i; j < n; j++)
53  if (a[n * i + j] != 0. || b[i] != 0.)
54  fail("Trying to reverse singular matrix. Column ", std::to_string(i), " is zeroed.");
55  continue; // x[i]=b[i], b[i]==0
56  }
57  // interchanging rows
58  if (maxnr != i) {
59  for (int j = i; j < n; j++)
60  std::swap(a[n * maxnr + j], a[n * i + j]);
61  std::swap(b[i], b[maxnr]);
62  }
63  // divide by a_ii -- to get a_ii=1
64  double c = 1.0 / a[i * n + i];
65  for (int j = i; j < n; j++)
66  a[i * n + j] *= c;
67  b[i] *= c;
68  // subtract -- to zero all remaining elements of this row
69  for (int k = 0; k < n; k++)
70  if (k != i) {
71  double d = a[k * n + i];
72  for (int j = i; j < n; j++)
73  a[k * n + j] -= a[i * n + j] * d;
74  b[k] -= b[i] * d;
75  }
76  }
77 }

105 double compute_wssr(const Target& target) {
106  long double wssr = 0; // long double here notably increases the accuracy
107  for (const auto& p : target.points)
108  wssr += sq(p.get_weight() * (p.get_y() - target.compute_value(p)));
109  return (double) wssr;

173 template<typename Target>
174 double compute_lm_matrices(const Target& target,
175  std::vector<double>& alpha,
176  std::vector<double>& beta) {
177  assert(!beta.empty());
178  assert(alpha.size() == beta.size() * beta.size());
179  long double wssr = 0; // long double here notably increases the accuracy
180  size_t na = beta.size();
181  std::fill(alpha.begin(), alpha.end(), 0.0);
182  std::fill(beta.begin(), beta.end(), 0.0);
183  std::vector<double> dy_da(na);
184  for (const auto& p : target.points) {
185  double y = target.compute_value_and_derivatives(p, dy_da);
186  double weight = p.get_weight();
187  double dy_sig = weight * (p.get_y() - y);
188  for (size_t j = 0; j != na; ++j) {
189  if (dy_da[j] != 0) {
190  dy_da[j] *= weight;
191  for (size_t k = j+1; k-- != 0;)
192  alpha[na * j + k] += dy_da[j] * dy_da[k];
193  beta[j] += dy_sig * dy_da[j];
194  }
195  }
196  wssr += sq(dy_sig);
197  }
198 
199  // Only half of the alpha matrix was filled above. Fill the rest.
200  for (size_t j = 1; j < na; j++)
201  for (size_t k = 0; k < j; k++)
202  alpha[na * k + j] = alpha[na * j + k];
203  return (double) wssr;
204 }

261  template<typename Target>
262  double fit(Target& target) {
263  std::vector<double> initial_a = target.get_parameters();
264 #ifdef GEMMI_DEBUG_LEVMAR
265  print_parameters("ini", initial_a);
266 #endif
267  std::vector<double> best_a = initial_a;
268  size_t na = initial_a.size();
269 
270  double lambda = lambda_start;
271  alpha.resize(na * na);
272  beta.resize(na);
273 
274  initial_wssr = compute_lm_matrices(target, alpha, beta);
275  double wssr = initial_wssr;
276 
277  int small_change_counter = 0;
278  eval_count = 1; // number of function evaluations so far
279  for (int iter = 0; ; iter++) {
280  if (eval_limit > 0 && eval_count >= eval_limit)
281  break;
282 
283  // prepare next parameters -> temp_beta
284  temp_alpha = alpha;
285  // Using '*=' not '+=' below applies the dampling factor as:
286  // J^T J + lambda * diag(J^T J); not ... + lambda * I.
287  for (size_t j = 0; j < na; j++)
288  temp_alpha[na * j + j] *= (1.0 + lambda);
289  temp_beta = beta;
290 
291  // Matrix solution (Ax=b) temp_alpha * da == temp_beta
292  jordan_solve(temp_alpha, temp_beta);
293 
294  for (size_t i = 0; i < na; i++)
295  // put new a[] into temp_beta[]
296  temp_beta[i] += best_a[i];
297 
298  target.set_parameters(temp_beta);
299  double new_wssr = compute_wssr(target);
300  ++eval_count;
301 #ifdef GEMMI_DEBUG_LEVMAR
302  fprintf(stderr, " #%d WSSR=%.8g %+g%% (%+.4g%%) lambda=%g\n",
303  iter, new_wssr, 100. * (new_wssr / initial_wssr - 1.),
304  100. * (new_wssr / wssr - 1.), lambda);
305  if (new_wssr < wssr)
306  print_parameters("", temp_beta);
307 #else
308  (void) iter;
309 #endif
310  if (new_wssr < wssr) {
311  double rel_change = (wssr - new_wssr) / wssr;
312  wssr = new_wssr;
313  best_a = temp_beta;
314 
315  if (wssr == 0)
316  break;
317  // termination criterion: negligible change of wssr
318  if (rel_change < stop_rel_change) {
319  if (++small_change_counter >= 2)
320  break;
321  } else {
322  small_change_counter = 0;
323  }
324  compute_lm_matrices(target, alpha, beta);
325  ++eval_count;
326  lambda *= lambda_down_factor;
327  } else { // worse fitting
328  if (lambda > lambda_limit) // termination criterion: large lambda
329  break;
330  lambda *= lambda_up_factor;
331  }
332  }
333 
334  target.set_parameters(wssr < initial_wssr ? best_a : initial_a);
335  return wssr;
336  }
