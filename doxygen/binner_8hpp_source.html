34  void setup_from_1_d2(int nbins, Method method, std::vector<double>&& inv_d2,
35  const UnitCell* cell_) {
36  if (nbins < 1)
37  fail("Binner: nbins argument must be positive");
38  if (inv_d2.empty())
39  fail("Binner: no data");
40  if (cell_)
41  cell = *cell_;
42  if (!cell.is_crystal())
43  fail("Binner: unknown unit cell");
44  // first setup 2N bins to get both bin limits and middle points
45  limits.resize(2 * nbins);
46  if (method == Method::EqualCount) {
47  std::sort(inv_d2.begin(), inv_d2.end());
48  min_1_d2 = inv_d2.front();
49  max_1_d2 = inv_d2.back();
50  } else {
51  min_1_d2 = max_1_d2 = inv_d2.front();
52  for (double x : inv_d2) {
53  if (x < min_1_d2)
54  min_1_d2 = x;
55  if (x > max_1_d2)
56  max_1_d2 = x;
57  }
58  }
59  switch (method) {
60  case Method::EqualCount: {
61  double avg_count = double(inv_d2.size()) / limits.size();
62  for (size_t i = 1; i < limits.size(); ++i)
63  limits[i-1] = inv_d2[int(avg_count * i)];
64  break;
65  }
66  case Method::Dstar2: {
67  double step = (max_1_d2 - min_1_d2) / limits.size();
68  for (size_t i = 1; i < limits.size(); ++i)
69  limits[i-1] = min_1_d2 + i * step;
70  break;
71  }
72  case Method::Dstar: {
73  double min_1_d = std::sqrt(min_1_d2);
74  double max_1_d = std::sqrt(max_1_d2);
75  double step = (max_1_d - min_1_d) / limits.size();
76  for (size_t i = 1; i < limits.size(); ++i)
77  limits[i-1] = sq(min_1_d + i * step);
78  break;
79  }
80  case Method::Dstar3: {
81  double min_1_d3 = min_1_d2 * std::sqrt(min_1_d2);
82  double max_1_d3 = max_1_d2 * std::sqrt(max_1_d2);
83  double step = (max_1_d3 - min_1_d3) / limits.size();
84  for (size_t i = 1; i < limits.size(); ++i)
85  limits[i-1] = sq(std::cbrt(min_1_d3 + i * step));
86  break;
87  }
88  }
89  limits.back() = std::numeric_limits<double>::infinity();
90 
91  mids.resize(nbins);
92  for (int i = 0; i < nbins; ++i) {
93  mids[i] = limits[2*i];
94  limits[i] = limits[2*i+1];
95  }
96  limits.resize(nbins);
97  }
