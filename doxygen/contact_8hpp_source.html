99 template<typename Func>
100 void ContactSearch::for_each_contact(NeighborSearch& ns, const Func& func) {
101  if (!ns.model)
102  fail(ns.small_structure ? "ContactSearch does not work with SmallStructure"
103  : "NeighborSearch not initialized");
104  for (int n_ch = 0; n_ch != (int) ns.model->chains.size(); ++n_ch) {
105  Chain& chain = ns.model->chains[n_ch];
106  PolymerType pt = PolymerType::Unknown;
107  if (ignore == Ignore::AdjacentResidues)
108  pt = check_polymer_type(chain.get_polymer());
109  for (int n_res = 0; n_res != (int) chain.residues.size(); ++n_res) {
110  Residue& res = chain.residues[n_res];
111  for (int n_atom = 0; n_atom != (int) res.atoms.size(); ++n_atom) {
112  Atom& atom = res.atoms[n_atom];
113  if (!ns.include_h && is_hydrogen(atom.element))
114  continue;
115  if (atom.occ < min_occupancy)
116  continue;
117  ns.for_each(atom.pos, atom.altloc, search_radius,
118  [&](NeighborSearch::Mark& m, double dist_sq) {
119  // do not consider connections inside a residue
120  if (ignore != Ignore::Nothing && m.image_idx == 0 &&
121  m.chain_idx == n_ch && m.residue_idx == n_res)
122  return;
123  switch (ignore) {
124  case Ignore::Nothing:
125  break;
126  case Ignore::SameResidue:
127  if (m.image_idx == 0 && m.chain_idx == n_ch)
128  if (m.residue_idx == n_res)
129  return;
130  break;
131  case Ignore::AdjacentResidues:
132  if (m.image_idx == 0 && m.chain_idx == n_ch)
133  if (m.residue_idx == n_res ||
134  are_connected(res, chain.residues[m.residue_idx], pt) ||
135  are_connected(chain.residues[m.residue_idx], res, pt))
136  return;
137  break;
138  case Ignore::SameChain:
139  if (m.image_idx == 0 && m.chain_idx == n_ch)
140  return;
141  break;
142  case Ignore::SameAsu:
143  if (m.image_idx == 0)
144  return;
145  break;
146  }
147  // additionally, we may have per-element distances
148  if (!radii.empty()) {
149  double d = radii[atom.element.ordinal()] + radii[m.element.ordinal()];
150  if (d < 0 || dist_sq > d * d)
151  return;
152  }
153  // avoid reporting connections twice (A-B and B-A)
154  if (!twice)
155  if (m.chain_idx < n_ch || (m.chain_idx == n_ch &&
156  (m.residue_idx < n_res || (m.residue_idx == n_res &&
157  m.atom_idx < n_atom))))
158  return;
159  // atom can be linked with its image, but if the image
160  // is too close the atom is likely on special position.
161  if (m.chain_idx == n_ch && m.residue_idx == n_res &&
162  m.atom_idx == n_atom && dist_sq < special_pos_cutoff_sq)
163  return;
164  CRA cra2 = m.to_cra(*ns.model);
165  // ignore atoms with occupancy below the specified value
166  if (cra2.atom->occ < min_occupancy)
167  return;
168  func(CRA{&chain, &res, &atom}, cra2, m.image_idx, dist_sq);
169  }, ns.sufficient_k(search_radius));
170  }
171  }
172  }
173 }
