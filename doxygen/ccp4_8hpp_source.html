38 // Minimal IEEE 754 binary16 type for reading CCP4 mode-12 maps.
39 struct float16_type {
40  std::uint16_t bits;
41  operator float() const {
42  std::uint32_t sign = (bits >> 15) & 1;
43  std::uint32_t exp = (bits >> 10) & 0x1F;
44  std::uint32_t mant = bits & 0x3FF;
45  float result;
46  if (exp == 0) { // subnormal or zero
47  result = std::ldexp(static_cast<float>(mant), -24);
48  } else if (exp == 31) { // inf or nan
49  std::uint32_t f = (sign << 31) | 0x7F800000u | (mant << 13);
50  std::memcpy(&result, &f, 4);
51  return result;
52  } else {
53  result = std::ldexp(static_cast<float>(mant + 1024), static_cast<int>(exp) - 25);
54  }
55  return sign ? -result : result;
56  }
57  bool operator!=(float16_type o) const { return bits != o.bits; }
58  bool operator==(float16_type o) const { return bits == o.bits; }
59 };
60 #endif

294  void read_ccp4_header_(GridMeta* grid, AnyStream& f, const std::string& path) {
295  const size_t hsize = 256;
296  ccp4_header.resize(hsize);
297  if (!f.read(ccp4_header.data(), 4 * hsize))
298  fail("Failed to read map header: " + path);
299  if (header_str(53, 4) != "MAP ")
300  fail("Not a CCP4 map: " + path);
301  std::string machst = header_str(54, 4);
302  if (machst[0] != 0x44 && machst[0] != 0x11)
303  fail("Unsupported machine stamp (endianness) in the file?");
304  same_byte_order = machst[0] == (is_little_endian() ? 0x44 : 0x11);
305  size_t ext_w = header_i32(24) / 4;
306  if (ext_w != 0) {
307  if (ext_w > 1000000)
308  fail("Unexpectedly long extended header: " + path);
309  ccp4_header.resize(hsize + ext_w);
310  if (!f.read(ccp4_header.data() + hsize, 4 * ext_w))
311  fail("Failed to read extended header: " + path);
312  }
313  for (int i = 0; i < 3; ++i) {
314  int axis = header_i32(17 + i);
315  if (axis < 1 || axis > 3)
316  fail("Unexpected axis value in word " + std::to_string(17 + i)
317  + ": " + std::to_string(axis));
318  }
319  hstats.dmin = header_float(20);
320  hstats.dmax = header_float(21);
321  hstats.dmean = header_float(22);
322  hstats.rms = header_float(55);
323  if (grid) {
324  grid->unit_cell.set(header_rfloat(11), header_rfloat(12), header_rfloat(13),
325  header_rfloat(14), header_rfloat(15), header_rfloat(16));
326  grid->nu = header_i32(1);
327  grid->nv = header_i32(2);
328  grid->nw = header_i32(3);
329  grid->spacegroup = find_spacegroup_by_number(header_i32(23));
330  auto pos = axis_positions();
331  grid->axis_order = AxisOrder::Unknown;
332  if (pos[0] == 0 && pos[1] == 1 && pos[2] == 2 && full_cell_(*grid))
333  grid->axis_order = AxisOrder::XYZ;
334  }

461 void read_data(AnyStream& f, std::vector<TMem>& content) {
462  if (std::is_same<TFile, TMem>::value) {
463  size_t len = content.size();
464  if (!f.read(content.data(), sizeof(TMem) * len))
465  fail("Failed to read all the data from the map file.");
466  } else {
467  constexpr size_t chunk_size = 64 * 1024;
468  std::vector<TFile> work(chunk_size);
469  for (size_t i = 0; i < content.size(); i += chunk_size) {
470  size_t len = std::min(chunk_size, content.size() - i);
471  if (!f.read(work.data(), sizeof(TFile) * len))
472  fail("Failed to read all the data from the map file.");
473  for (size_t j = 0; j < len; ++j)
474  content[i+j] = translate_map_point<TFile,TMem>(work[j]);
475  }
476  }
477 }

504 void Ccp4<T>::read_ccp4_stream(AnyStream& f, const std::string& path) {
505  read_ccp4_header(f, path);
506  grid.data.resize(grid.point_count());
507  int mode = header_i32(4);
508  if (mode == 0)
509  impl::read_data<std::int8_t>(f, grid.data);
510  else if (mode == 1)
511  impl::read_data<std::int16_t>(f, grid.data);
512  else if (mode == 2)
513  impl::read_data<float>(f, grid.data);
514  else if (mode == 6)

516  else if (mode == 12)
517  impl::read_data<float16_type>(f, grid.data);

522  if (!same_byte_order) {
523  if (sizeof(T) == 2)
524  for (T& value : grid.data)
525  swap_two_bytes(&value);
526  else if (sizeof(T) == 4)
527  for (T& value : grid.data)
528  swap_four_bytes(&value);

533 template<typename T>
534 void Ccp4<T>::setup(T default_value, MapSetup mode) {
535  if (grid.axis_order == AxisOrder::XYZ || ccp4_header.empty())
536  return;
537  // cell sampling does not change
538  const std::array<int, 3> sampl = header_3i32(8);
539  // get old metadata
540  const std::array<int, 3> pos = axis_positions();
541  std::array<int, 3> start = header_3i32(5);
542  int end[3] = { start[0] + grid.nu, start[1] + grid.nv, start[2] + grid.nw };
543  // set new metadata
544  if (mode == MapSetup::ReorderOnly) {
545  set_header_3i32(5, start[pos[0]], start[pos[1]], start[pos[2]]);
546  for (int i = 0; i < 3; ++i) {
547  end[i] -= start[i];
548  start[i] = 0;
549  }
550  int crs[3] = { grid.nu, grid.nv, grid.nw };
551  grid.nu = crs[pos[0]];
552  grid.nv = crs[pos[1]];
553  grid.nw = crs[pos[2]];
554  } else {
555  grid.nu = sampl[0];
556  grid.nv = sampl[1];
557  grid.nw = sampl[2];
558  set_header_3i32(5, 0, 0, 0);
559  }
560  set_header_3i32(1, grid.nu, grid.nv, grid.nw);
561  set_header_3i32(17, 1, 2, 3);
562  grid.axis_order = full_cell() ? AxisOrder::XYZ : AxisOrder::Unknown;
563  if (grid.axis_order == AxisOrder::XYZ)
564  grid.calculate_spacing();
565 
566  // now set the data
567  {
568  std::vector<T> full(grid.point_count(), default_value);
569  int it[3];
570  int idx = 0;
571  for (it[2] = start[2]; it[2] < end[2]; it[2]++)
572  for (it[1] = start[1]; it[1] < end[1]; it[1]++)
573  for (it[0] = start[0]; it[0] < end[0]; it[0]++) {
574  T val = grid.data[idx++];
575  size_t new_index = grid.index_s(it[pos[0]], it[pos[1]], it[pos[2]]);
576  full[new_index] = val;
577  }
578  grid.data = std::move(full);
579  }
580 
581  if (mode == MapSetup::Full &&
582  (end[pos[0]] - start[pos[0]] < sampl[0] ||
583  end[pos[1]] - start[pos[1]] < sampl[1] ||
584  end[pos[2]] - start[pos[2]] < sampl[2]))
585  grid.symmetrize_nondefault(default_value);
586 }

589 template<typename T>
590 void Ccp4<T>::set_extent(const Box<Fractional>& box) {
591  if (ccp4_header.empty())
592  fail("set_extent(): no header in the map. Call update_ccp4_header() first");
593  if (!full_cell())
594  fail("Ccp4::set_extent() works only after setup()");
595  if (grid.axis_order != AxisOrder::XYZ)
596  fail("Ccp4::set_extent() works only with XYZ order");
597  int u0 = (int)std::ceil(box.minimum.x * grid.nu);
598  int v0 = (int)std::ceil(box.minimum.y * grid.nv);
599  int w0 = (int)std::ceil(box.minimum.z * grid.nw);
600  int nu = (int)std::floor(box.maximum.x * grid.nu) - u0 + 1;
601  int nv = (int)std::floor(box.maximum.y * grid.nv) - v0 + 1;
602  int nw = (int)std::floor(box.maximum.z * grid.nw) - w0 + 1;
603  // set the data
604  std::vector<T> new_data((size_t)nu * nv * nw);
605  grid.get_subarray(new_data.data(), {u0, v0, w0}, {nu, nv, nw});
606  grid.data.swap(new_data);
607  // and metadata
608  grid.nu = nu;
609  grid.nv = nv;
610  grid.nw = nw;
611  set_header_3i32(1, grid.nu, grid.nv, grid.nw);
612  set_header_3i32(5, u0, v0, w0);
613  grid.axis_order = AxisOrder::Unknown;
614 }
