97 inline int string_to_int(const char* p, bool checked, size_t length=0) {
98  int mult = -1;
99  int n = 0;
100  size_t i = 0;
101  while ((length == 0 || i < length) && is_space(p[i]))
102  ++i;
103  if (p[i] == '-') {
104  mult = 1;
105  ++i;
106  } else if (p[i] == '+') {
107  ++i;
108  }
109  bool has_digits = false;
110  // use negative numbers because INT_MIN < -INT_MAX
111  for (; (length == 0 || i < length) && is_digit(p[i]); ++i) {
112  n = n * 10 - (p[i] - '0');
113  has_digits = true;
114  }
115  if (checked) {
116  while ((length == 0 || i < length) && is_space(p[i]))
117  ++i;
118  if (!has_digits || p[i] != '\0')
119  throw std::invalid_argument("not an integer: " +
120  std::string(p, length ? length : i+1));
121  }
122  return mult * n;
123 }
