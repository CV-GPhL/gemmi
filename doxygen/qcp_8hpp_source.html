113 inline double qcp_inner_product(Mat33& mat,
114  const Position* pos1, const Position& ctr1,
115  const Position* pos2, const Position& ctr2,
116  size_t len, const double* weight) {
117  double G1 = 0.0, G2 = 0.0;
118  for (size_t i = 0; i < len; ++i) {
119  Position f1 = pos1[i] - ctr1;
120  Position f2 = pos2[i] - ctr2;
121  double w = (weight != nullptr ? weight[i] : 1.);
122  Vec3 v1 = w * f1;
123  G1 += v1.dot(f1);
124  G2 += w * f2.length_sq();
125  mat[0][0] += v1.x * f2.x;
126  mat[0][1] += v1.x * f2.y;
127  mat[0][2] += v1.x * f2.z;
128  mat[1][0] += v1.y * f2.x;
129  mat[1][1] += v1.y * f2.y;
130  mat[1][2] += v1.y * f2.z;
131  mat[2][0] += v1.z * f2.x;
132  mat[2][1] += v1.z * f2.y;
133  mat[2][2] += v1.z * f2.z;
134  }
135  return (G1 + G2) * 0.5;
136 }

153 inline int fast_calc_rmsd_and_rotation(Mat33* rot, const Mat33& A, double *rmsd,
154  double E0, double len, double min_score) {
155  const double evecprec = 1e-6;
156  const double evalprec = 1e-11;
157 
158  double Sxx, Sxy, Sxz, Syx, Syy, Syz, Szx, Szy, Szz;
159  Sxx = A[0][0]; Sxy = A[0][1]; Sxz = A[0][2];
160  Syx = A[1][0]; Syy = A[1][1]; Syz = A[1][2];
161  Szx = A[2][0]; Szy = A[2][1]; Szz = A[2][2];
162 
163  double Sxx2 = Sxx * Sxx;
164  double Syy2 = Syy * Syy;
165  double Szz2 = Szz * Szz;
166 
167  double Sxy2 = Sxy * Sxy;
168  double Syz2 = Syz * Syz;
169  double Sxz2 = Sxz * Sxz;
170 
171  double Syx2 = Syx * Syx;
172  double Szy2 = Szy * Szy;
173  double Szx2 = Szx * Szx;
174 
175  double SyzSzymSyySzz2 = 2.0 * (Syz*Szy - Syy*Szz);
176  double Sxx2Syy2Szz2Syz2Szy2 = Syy2 + Szz2 - Sxx2 + Syz2 + Szy2;
177 
178  double C[4];
179  C[2] = -2.0 * (Sxx2 + Syy2 + Szz2 + Sxy2 + Syx2 + Sxz2 + Szx2 + Syz2 + Szy2);
180  C[1] = 8.0 * (Sxx*Syz*Szy + Syy*Szx*Sxz + Szz*Sxy*Syx - Sxx*Syy*Szz - Syz*Szx*Sxy - Szy*Syx*Sxz);
181 
182  double SxzpSzx = Sxz + Szx;
183  double SyzpSzy = Syz + Szy;
184  double SxypSyx = Sxy + Syx;
185  double SyzmSzy = Syz - Szy;
186  double SxzmSzx = Sxz - Szx;
187  double SxymSyx = Sxy - Syx;
188  double SxxpSyy = Sxx + Syy;
189  double SxxmSyy = Sxx - Syy;
190  double Sxy2Sxz2Syx2Szx2 = Sxy2 + Sxz2 - Syx2 - Szx2;
191 
192  C[0] = Sxy2Sxz2Syx2Szx2 * Sxy2Sxz2Syx2Szx2
193  + (Sxx2Syy2Szz2Syz2Szy2 + SyzSzymSyySzz2) * (Sxx2Syy2Szz2Syz2Szy2 - SyzSzymSyySzz2)
194  + (-(SxzpSzx)*(SyzmSzy)+(SxymSyx)*(SxxmSyy-Szz)) * (-(SxzmSzx)*(SyzpSzy)+(SxymSyx)*(SxxmSyy+Szz))
195  + (-(SxzpSzx)*(SyzpSzy)-(SxypSyx)*(SxxpSyy-Szz)) * (-(SxzmSzx)*(SyzmSzy)-(SxypSyx)*(SxxpSyy+Szz))
196  + (+(SxypSyx)*(SyzpSzy)+(SxzpSzx)*(SxxmSyy+Szz)) * (-(SxymSyx)*(SyzmSzy)+(SxzpSzx)*(SxxpSyy+Szz))
197  + (+(SxypSyx)*(SyzmSzy)+(SxzmSzx)*(SxxmSyy-Szz)) * (-(SxymSyx)*(SyzpSzy)+(SxzmSzx)*(SxxpSyy-Szz));
198 
199  /* Newton-Raphson */
200  double mxEigenV = E0;
201  int i;
202  double oldg = 0.0;
203  for (i = 0; i < 50; ++i) {
204  oldg = mxEigenV;
205  double x2 = mxEigenV * mxEigenV;
206  double b = (x2 + C[2]) * mxEigenV;
207  double a = b + C[1];
208  double delta = (a * mxEigenV + C[0]) / (2.0 * x2 * mxEigenV + b + a);
209  mxEigenV -= delta;
210  // printf("\n diff[%3d]: %16g %16g %16g", i, mxEigenV - oldg, evalprec*mxEigenV, mxEigenV);
211  if (std::fabs(mxEigenV - oldg) < std::fabs(evalprec * mxEigenV))
212  break;
213  }
214  if (i == 50)
215  std::fprintf(stderr,"\nMore than %d iterations needed!\n", i);
216 
217  // the fabs() is to guard against extremely small, but *negative* numbers
218  // due to floating point error
219  double rms = std::sqrt(std::fabs(2.0 * (E0 - mxEigenV) / len));
220  (*rmsd) = rms;
221  // printf("\n\n %16g %16g %16g \n", rms, E0, 2.0 * (E0 - mxEigenV)/len);
222 
223  if (rot == nullptr)
224  return -1;
225  if (min_score > 0)
226  if (rms < min_score)
227  return -1; // Don't bother with rotation.
228 
229  double a11, a12, a13, a14, a21, a22, a23, a24;
230  a11 = SxxpSyy + Szz-mxEigenV; a12 = SyzmSzy; a13 = - SxzmSzx; a14 = SxymSyx;
231  a21 = SyzmSzy; a22 = SxxmSyy - Szz-mxEigenV; a23 = SxypSyx; a24= SxzpSzx;
232  double a31, a32, a33, a34, a41, a42, a43, a44;
233  a31 = a13; a32 = a23; a33 = Syy-Sxx-Szz - mxEigenV; a34 = SyzpSzy;
234  a41 = a14; a42 = a24; a43 = a34; a44 = Szz - SxxpSyy - mxEigenV;
235  double a3344_4334, a3244_4234, a3243_4233, a3143_4133,a3144_4134, a3142_4132;
236  a3344_4334 = a33 * a44 - a43 * a34; a3244_4234 = a32 * a44-a42*a34;
237  a3243_4233 = a32 * a43 - a42 * a33; a3143_4133 = a31 * a43-a41*a33;
238  a3144_4134 = a31 * a44 - a41 * a34; a3142_4132 = a31 * a42-a41*a32;
239  double q1 = a22*a3344_4334-a23*a3244_4234+a24*a3243_4233;
240  double q2 = -a21*a3344_4334+a23*a3144_4134-a24*a3143_4133;
241  double q3 = a21*a3244_4234-a22*a3144_4134+a24*a3142_4132;
242  double q4 = -a21*a3243_4233+a22*a3143_4133-a23*a3142_4132;
243 
244  double qsqr = q1 * q1 + q2 * q2 + q3 * q3 + q4 * q4;
245 
246  /* The following code tries to calculate another column in the adjoint matrix
247  when the norm of the current column is too small.
248  Usually this block will never be activated. To be absolutely safe this should be
249  uncommented, but it is most likely unnecessary.
250  */
251  if (qsqr < evecprec) {
252  q1 = a12*a3344_4334 - a13*a3244_4234 + a14*a3243_4233;
253  q2 = -a11*a3344_4334 + a13*a3144_4134 - a14*a3143_4133;
254  q3 = a11*a3244_4234 - a12*a3144_4134 + a14*a3142_4132;
255  q4 = -a11*a3243_4233 + a12*a3143_4133 - a13*a3142_4132;
256  qsqr = q1*q1 + q2 *q2 + q3*q3+q4*q4;
257 
258  if (qsqr < evecprec) {
259  double a1324_1423 = a13 * a24 - a14 * a23, a1224_1422 = a12 * a24 - a14 * a22;
260  double a1223_1322 = a12 * a23 - a13 * a22, a1124_1421 = a11 * a24 - a14 * a21;
261  double a1123_1321 = a11 * a23 - a13 * a21, a1122_1221 = a11 * a22 - a12 * a21;
262 
263  q1 = a42 * a1324_1423 - a43 * a1224_1422 + a44 * a1223_1322;
264  q2 = -a41 * a1324_1423 + a43 * a1124_1421 - a44 * a1123_1321;
265  q3 = a41 * a1224_1422 - a42 * a1124_1421 + a44 * a1122_1221;
266  q4 = -a41 * a1223_1322 + a42 * a1123_1321 - a43 * a1122_1221;
267  qsqr = q1*q1 + q2 *q2 + q3*q3+q4*q4;
268 
269  if (qsqr < evecprec) {
270  q1 = a32 * a1324_1423 - a33 * a1224_1422 + a34 * a1223_1322;
271  q2 = -a31 * a1324_1423 + a33 * a1124_1421 - a34 * a1123_1321;
272  q3 = a31 * a1224_1422 - a32 * a1124_1421 + a34 * a1122_1221;
273  q4 = -a31 * a1223_1322 + a32 * a1123_1321 - a33 * a1122_1221;
274  qsqr = q1*q1 + q2 *q2 + q3*q3 + q4*q4;
275 
276  if (qsqr < evecprec) {
277  /* if qsqr is still too small, return the identity matrix. */
278  *rot = Mat33();
279  return 0;
280  }
281  }
282  }
283  }
284 
285  double normq = std::sqrt(qsqr);
286  q1 /= normq;
287  q2 /= normq;
288  q3 /= normq;
289  q4 /= normq;
290 
291  double a2 = q1 * q1;
292  double x2 = q2 * q2;
293  double y2 = q3 * q3;
294  double z2 = q4 * q4;
295 
296  double xy = q2 * q3;
297  double az = q1 * q4;
298  double zx = q4 * q2;
299  double ay = q1 * q3;
300  double yz = q3 * q4;
301  double ax = q1 * q2;
302 
303  rot->a[0][0] = a2 + x2 - y2 - z2;
304  rot->a[0][1] = 2 * (xy + az);
305  rot->a[0][2] = 2 * (zx - ay);
306  rot->a[1][0] = 2 * (xy - az);
307  rot->a[1][1] = a2 - x2 + y2 - z2;
308  rot->a[1][2] = 2 * (yz + ax);
309  rot->a[2][0] = 2 * (zx + ay);
310  rot->a[2][1] = 2 * (yz - ax);
311  rot->a[2][2] = a2 - x2 - y2 + z2;
312 
313  return 1;
314 }

321 inline Position qcp_calculate_center(const Position* pos, size_t len, const double *weight) {
322  double wsum = 0.0;
323  Position ctr;
324  for (size_t i = 0; i < len; ++i) {
325  double w = (weight != nullptr ? weight[i] : 1.);
326  ctr += w * pos[i];
327  wsum += w;
328  }
329  return ctr / wsum;
330 }

350 inline SupResult superpose_positions(const Position* pos1, const Position* pos2,
351  size_t len, const double* weight) {
352  SupResult result;
353  result.count = len;
354 
355  /* center the structures -- if precentered you can omit this step */
356  result.center1 = qcp_calculate_center(pos1, len, weight);
357  result.center2 = qcp_calculate_center(pos2, len, weight);
358 
359  if (len == 0) {
360  result.rmsd = NAN;
361  return result;
362  }
363 
364  double wsum = 0.0;
365  if (weight == nullptr)
366  wsum = (double) len;
367  else
368  for (size_t i = 0; i < len; ++i)
369  wsum += weight[i];
370 
371  Mat33 A(0);
372  /* calculate the (weighted) inner product of two structures */
373  double E0 = qcp_inner_product(A, pos1, result.center1, pos2, result.center2, len, weight);
374 
375  /* calculate the RMSD & rotational matrix */
376  fast_calc_rmsd_and_rotation(&result.transform.mat, A, &result.rmsd, E0, wsum, -1);
377  result.transform.vec = Vec3(result.center1) - result.transform.mat.multiply(result.center2);
378 
379  return result;
380 }

392 inline double calculate_rmsd_of_superposed_positions(const Position* pos1,

404  else
405  for (size_t i = 0; i < len; ++i)
406  wsum += weight[i];

409  Mat33 A(0);
410  double E0 = qcp_inner_product(A, pos1, ctr1, pos2, ctr2, len, weight);
411 
412  // calculate the RMSD
