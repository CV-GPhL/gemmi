124  void normalize(double eps=1e-9) {
125  auto step_N1 = [&]() {
126  if (A - B > eps || (A - B >= -eps && std::abs(xi) > std::abs(eta) + eps)) { // N1
127  std::swap(A, B);
128  std::swap(xi, eta);
129  if (change_of_basis)
130  swap_columns_and_negate(0, 1);
131  }
132  };
133  step_N1();
134  if (B - C > eps || (B - C >= -eps && std::abs(eta) > std::abs(zeta) + eps)) { // N2
135  std::swap(B, C);
136  std::swap(eta, zeta);
137  if (change_of_basis)
138  swap_columns_and_negate(1, 2);
139  // To make it faster, instead of "go to the point N1" we repeat N1 once
140  // (which is equivalent - three swaps are sufficient to reorder ABC).
141  step_N1();
142  }
143  // N3
144  // xi * eta * zeta > 0 <=> positive count is 1 or 3 and no zeros
145  int pos_count = (xi > eps) + (eta > eps) + (zeta > eps);
146  int nonneg_count = (xi >= -eps) + (eta >= -eps) + (zeta >= -eps);
147  double sgn = (pos_count == nonneg_count && pos_count % 2 == 1) ? 1 : -1;
148  if (change_of_basis) {
149  if (sgn * xi < -eps) negate_column(0);
150  if (sgn * eta < -eps) negate_column(1);
151  if (sgn * zeta < -eps) negate_column(2);
152  if (pos_count != nonneg_count && pos_count % 2 == 1)
153  negate_column(std::fabs(zeta) <= eps ? 2 :
154  std::fabs(eta) <= eps ? 1 : 0);
155  }
156  xi = std::copysign(xi, sgn);
157  eta = std::copysign(eta, sgn);
158  zeta = std::copysign(zeta, sgn);
159  }

164  bool buerger_step() {
165  if (std::abs(xi) > B) { // B2
166  double j = std::floor(0.5*xi/B + 0.5);
167  C += j * (j*B - xi);
168  xi -= 2 * j * B;
169  eta -= j * zeta;
170  } else if (std::abs(eta) > A) { // B3
171  double j = std::floor(0.5*eta/A + 0.5);
172  C += j * (j*A - eta);
173  xi -= j * zeta;
174  eta -= 2 * j * A;
175  } else if (std::abs(zeta) > A) { // B4
176  double j = std::floor(0.5*zeta/A + 0.5);
177  B += j * (j*A - zeta);
178  xi -= j * eta;
179  zeta -= 2 * j * A;
180  } else if (xi + eta + zeta + A + B < 0) { // B5
181  double j = std::floor(0.5 * (xi + eta) / (A + B + zeta) + 0.5);
182  C += j * (j * (A + B + zeta) - (xi + eta));
183  xi -= j * (2*B + zeta);
184  eta -= j * (2*A + zeta);
185  } else {
186  return true;
187  }
188  return false;
189  }

194  int buerger_reduce() {
195  int n = 0;
196  double prev_sum = -1;
197  int stall_count = 0;
198  for (;;) {
199  normalize();
200  // In rare cases numerical errors push the algorithm into infinite loop,
201  // as described in Grosse-Kunstleve et al, Acta Cryst. (2004) A60, 1.
202  // Ad-hoc solution: stop if a+b+c is stalled for 5 iterations.
203  if (++n > 8) { // don't waste time during the first few iterations
204  double sum = std::sqrt(A) + std::sqrt(B) + std::sqrt(C);
205  if (std::abs(sum - prev_sum) < sum * 1e-6) {
206  if (++stall_count == 5)
207  break;
208  } else {
209  stall_count = 0;
210  }
211  prev_sum = sum;
212  }
213  if (buerger_step())
214  break;
215  }
216  return n;
217  }

224  bool niggli_step(double epsilon=1e-9) {
225  if (std::abs(xi) > B + epsilon || // step 5. from Krivy-Gruber (1976)
226  (xi >= B - epsilon && 2 * eta < zeta - epsilon) ||
227  (xi <= -(B - epsilon) && zeta < -epsilon)) {
228  double sign_xi = xi >= 0 ? 1 : -1;
229  C += B - xi * sign_xi;
230  eta -= zeta * sign_xi;
231  xi -= 2 * B * sign_xi;
232  if (change_of_basis)
233  add_column(1, 2, -int(sign_xi));
234  } else if (std::abs(eta) > A + epsilon || // step 6.
235  (eta >= A - epsilon && 2 * xi < zeta - epsilon) ||
236  (eta <= -(A - epsilon) && zeta < -epsilon)) {
237  double sign_eta = eta >= 0 ? 1 : -1;
238  C += A - eta * sign_eta;
239  xi -= zeta * sign_eta;
240  eta -= 2 * A * sign_eta;
241  if (change_of_basis)
242  add_column(0, 2, -int(sign_eta));
243  } else if (std::abs(zeta) > A + epsilon || // step 7.
244  (zeta >= A - epsilon && 2 * xi < eta - epsilon) ||
245  (zeta <= -(A - epsilon) && eta < -epsilon)) {
246  double sign_zeta = zeta >= 0 ? 1 : -1;
247  B += A - zeta * sign_zeta;
248  xi -= eta * sign_zeta;
249  zeta -= 2 * A * sign_zeta;
250  if (change_of_basis)
251  add_column(0, 1, -int(sign_zeta));
252  } else if (xi + eta + zeta + A + B < -epsilon || // step 8.
253  (xi + eta + zeta + A + B <= epsilon && 2 * (A + eta) + zeta > epsilon)) {
254  C += A + B + xi + eta + zeta;
255  xi += 2 * B + zeta;
256  eta += 2 * A + zeta;
257  if (change_of_basis) {
258  add_column(0, 2, 1);
259  add_column(1, 2, 1);
260  }
261  } else {
262  return true;
263  }
264  return false;
265  }

374  bool reduce_step(double eps=1e-9) {
375  //printf(" s = %g %g %g %g %g %g sum=%g\n",
376  // s[0], s[1], s[2], s[3], s[4], s[5], sum_b_squared());
377  const int table[6][5] = {
378  // When negating s[n] we need to apply operations from table[n]:
379  // 2 x add, subtract, 2 x swap&add
380  {2, 4, 3, 1, 5}, // 0
381  {2, 3, 4, 0, 5}, // 1
382  {1, 3, 5, 0, 4}, // 2
383  {1, 2, 0, 4, 5}, // 3
384  {0, 2, 1, 3, 5}, // 4
385  {0, 1, 2, 3, 4}, // 5
386  };
387 
388  double max_s = eps;
389  int max_s_pos = -1;
390  for (int i = 0; i < 6; ++i)
391  if (s[i] > max_s) {
392  max_s = s[i];
393  max_s_pos = i;
394  }
395  if (max_s_pos < 0)
396  return false;
397  const int (&indices)[5] = table[max_s_pos];
398  s[max_s_pos] = -max_s;
399  s[indices[0]] += max_s;
400  s[indices[1]] += max_s;
401  s[indices[2]] -= max_s;
402  std::swap(s[indices[3]], s[indices[4]]);
403  s[indices[3]] += max_s;
404  s[indices[4]] += max_s;
405  //printf(" s[%d]=%g sum: %g\n", max_s_pos, max_s, sum_b_squared());
406  return true;
407  }

435  void sort(double eps=1e-9) {
436  double abcd_sq_neg[4] = {
437  // -a^2, -b^2, -c^2, -d^2 (negated - to be sorted in descending order)
438  s[1]+s[2]+s[3], s[0]+s[2]+s[4], s[0]+s[1]+s[5], s[3]+s[4]+s[5]
439  };
440  // First, make sure that d >= a,b,c (therefore -d^2 <= -a^2,...).
441  int min_idx = 3;
442  for (int i = 0; i < 3; ++i)
443  if (abcd_sq_neg[i] < abcd_sq_neg[min_idx] - eps)
444  min_idx = i;
445  switch (min_idx) {
446  case 0: // a <-> d
447  std::swap(s[1], s[5]);
448  std::swap(s[2], s[4]);
449  break;
450  case 1: // b <-> d
451  std::swap(s[0], s[5]);
452  std::swap(s[2], s[3]);
453  break;
454  case 2: // c <-> d
455  std::swap(s[0], s[4]);
456  std::swap(s[1], s[3]);
457  break;
458  }
459  // we could stop here and not care about the order of a,b,c.
460  std::swap(abcd_sq_neg[min_idx], abcd_sq_neg[3]);
461  if (abcd_sq_neg[0] < abcd_sq_neg[1] - eps) { // a <-> b
462  std::swap(s[0], s[1]);
463  std::swap(s[3], s[4]);
464  std::swap(abcd_sq_neg[0], abcd_sq_neg[1]);
465  }
466  if (abcd_sq_neg[1] < abcd_sq_neg[2] - eps) { // b <-> c
467  std::swap(s[1], s[2]);
468  std::swap(s[4], s[5]);
469  std::swap(abcd_sq_neg[1], abcd_sq_neg[2]);
470  }
471  if (abcd_sq_neg[0] < abcd_sq_neg[1] - eps) { // a <-> b
472  std::swap(s[0], s[1]);
473  std::swap(s[3], s[4]);
474  //std::swap(abcd_sq_neg[0], abcd_sq_neg[1]);
475  }
476  }
