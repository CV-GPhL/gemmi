38  void index_chem_links(const MonLib& monlib, bool use_alias=true) {
39  std::map<ChemComp::Group, std::map<std::string, std::vector<std::string>>> aliases;
40  if (use_alias)
41  for (const auto& iter : monlib.monomers)
42  for (const ChemComp::Aliasing& a : iter.second.aliases)
43  for (const std::pair<std::string, std::string>& r : a.related) {
44  const ChemComp::Group& gr = ChemComp::is_nucleotide_group(a.group) ? ChemComp::Group::DnaRna : a.group;
45  aliases[gr][r.second].push_back(r.first);
46  }
47 
48  for (const auto& iter : monlib.links) {
49  const ChemLink& link = iter.second;
50  if (link.rt.bonds.empty())
51  continue;
52  if (link.rt.bonds.size() > 1)
53  fprintf(stderr, "Note: considering only the first bond in %s\n",
54  link.id.c_str());
55  if (link.side1.comp.empty() && link.side2.comp.empty())
56  if (link.side1.group == ChemComp::Group::Null ||
57  link.side2.group == ChemComp::Group::Null ||
58  link.id == "SS")
59  continue;
60  const Restraints::Bond& bond = link.rt.bonds[0];
61  if (bond.value > global_max_dist)
62  global_max_dist = bond.value;
63  links.emplace(bond.lexicographic_str(), &link);
64 
65  if (!use_alias || (!link.side1.comp.empty() && !link.side2.comp.empty()))
66  continue;
67  std::vector<std::string> *names1 = nullptr, *names2 = nullptr;
68  if (link.side1.comp.empty()) {
69  auto i = aliases.find(link.side1.group);
70  if (i != aliases.end()) {
71  auto j = i->second.find(bond.id1.atom);
72  if (j != i->second.end())
73  names1 = &j->second;
74  }
75  }
76  if (link.side2.comp.empty()) {
77  auto i = aliases.find(link.side2.group);
78  if (i != aliases.end()) {
79  auto j = i->second.find(bond.id2.atom);
80  if (j != i->second.end())
81  names2 = &j->second;
82  }
83  }
84  if (names1 && names2)
85  for (const std::string& n1 : *names1)
86  for (const std::string& n2 : *names2)
87  links.emplace(Restraints::lexicographic_str(n1, n2), &link);
88  else if (names1 || names2) {
89  const std::string& n1 = names1 ? bond.id2.atom : bond.id1.atom;
90  for (const std::string& n2 : (names1 ? *names1 : *names2))
91  links.emplace(Restraints::lexicographic_str(n1, n2), &link);
92  }
93  }
94  monlib_ptr = &monlib;
95  }

103  std::vector<Match> find_possible_links(Structure& st,
104  double bond_margin,
105  double radius_margin,
106  ContactSearch::Ignore ignore) {
107  std::vector<Match> results;
108  Model& model = st.first_model();
109  double search_radius = std::max(global_max_dist * bond_margin,
110  /*max r1+r2 ~=*/3.0 * radius_margin);
111  NeighborSearch ns(model, st.cell, std::max(5.0, search_radius));
112  ns.populate();
113 
114  ContactSearch contacts((float) search_radius);
115  contacts.ignore = ignore;
116  contacts.for_each_contact(ns, [&](const CRA& cra1, const CRA& cra2,
117  int image_idx, double dist_sq) {
118  Match match;
119 
120  // search for a match in chem_links
121  if (bond_margin > 0) {
122  auto range = links.equal_range(Restraints::lexicographic_str(
123  cra1.atom->name, cra2.atom->name));
124  // similar to MonLib::match_link()
125  for (auto iter = range.first; iter != range.second; ++iter) {
126  const ChemLink& link = *iter->second;
127  const Restraints::Bond& bond = link.rt.bonds[0];
128  if (dist_sq > sq(bond.value * bond_margin))
129  continue;
130  const ChemComp::Aliasing* aliasing1 = nullptr;
131  const ChemComp::Aliasing* aliasing2 = nullptr;
132  bool order1;
133  if (monlib_ptr->link_side_matches_residue(link.side1, cra1.residue->name, &aliasing1) &&
134  monlib_ptr->link_side_matches_residue(link.side2, cra2.residue->name, &aliasing2) &&
135  atom_match_with_alias(bond.id1.atom, cra1.atom->name, aliasing1))
136  order1 = true;
137  else if (monlib_ptr->link_side_matches_residue(link.side2, cra1.residue->name, &aliasing1) &&
138  monlib_ptr->link_side_matches_residue(link.side1, cra2.residue->name, &aliasing2) &&
139  atom_match_with_alias(bond.id2.atom, cra1.atom->name, aliasing1))
140  order1 = false;
141  else
142  continue;
143  int link_score = link.calculate_score(
144  order1 ? *cra1.residue : *cra2.residue,
145  order1 ? cra2.residue : cra1.residue,
146  order1 ? cra1.atom->altloc : cra2.atom->altloc,
147  order1 ? cra2.atom->altloc : cra1.atom->altloc,
148  order1 ? aliasing1 : aliasing2,
149  order1 ? aliasing2 : aliasing1);
150  match.chem_link_count++;
151  if (link_score > match.score) {
152  match.chem_link = &link;
153  match.score = link_score;
154  if (order1) {
155  match.cra1 = cra1;
156  match.cra2 = cra2;
157  } else {
158  match.cra1 = cra2;
159  match.cra2 = cra1;
160  }
161  }
162  }
163  }
164 
165  // potential other links according to covalent radii
166  if (!match.chem_link) {
167  float r1 = cra1.atom->element.covalent_r();
168  float r2 = cra2.atom->element.covalent_r();
169  if (dist_sq > sq((r1 + r2) * radius_margin))
170  return;
171  match.cra1 = cra1;
172  match.cra2 = cra2;
173  }
174 
175  // finalize
176  match.same_image = !image_idx;
177  match.bond_length = std::sqrt(dist_sq);
178  results.push_back(match);
179  });
180 
181  // add references to st.connections
182  for (Match& match : results)
183  match.conn = st.find_connection_by_cra(match.cra1, match.cra2);
184 
185  return results;
186  }
