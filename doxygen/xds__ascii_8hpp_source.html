218  Mat33 calculate_conversion_from_cambridge() const {
219  // Cambridge z direction is along the principal rotation axis
220  Vec3 z = get_rotation_axis();
221  // Cambridge x direction is along beam
222  Vec3 x = get_s0_direction();
223  Vec3 y = z.cross(x).normalized();
224  // beam and rotation axis may not be orthogonal
225  x = y.cross(z).normalized();
226  return Mat33::from_columns(x, y, z);
227  }
228 
232  Mat33 get_orientation() const {
233  if (!has_cell_axes())
234  fail("unknown unit cell axes");
235  Vec3 a = cell_axes.row_copy(0);
236  Vec3 b = cell_axes.row_copy(1);
237  Vec3 c = cell_axes.row_copy(2);
238  Vec3 ar = b.cross(c).normalized();
239  Vec3 br = c.cross(a);
240  Vec3 cr = ar.cross(br).normalized();
241  br = cr.cross(ar);
242  return Mat33::from_columns(ar, br, cr);
243  }
