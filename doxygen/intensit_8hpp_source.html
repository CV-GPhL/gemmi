319 template<typename DataProxy>
320 std::pair<DataType, size_t> check_data_type_under_symmetry(const DataProxy& proxy) {
321  const SpaceGroup* sg = proxy.spacegroup();
322  if (!sg)
323  return {DataType::Unknown, 0};
324  std::unordered_map<Op::Miller, int, MillerHash> seen;
325  ReciprocalAsu asu(sg);
326  GroupOps gops = sg->operations();
327  bool centric = gops.is_centrosymmetric();
328  DataType data_type = DataType::Mean;
329  for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
330  auto hkl_sign = asu.to_asu_sign(proxy.get_hkl(i), gops);
331  int sign = hkl_sign.second ? 2 : 1; // 2=positive, 1=negative
332  auto r = seen.emplace(hkl_sign.first, sign);
333  if (data_type != DataType::Unmerged && !r.second) {
334  if ((r.first->second & sign) != 0 || centric) {
335  data_type = DataType::Unmerged;
336  } else {
337  r.first->second |= sign;
338  data_type = DataType::Anomalous;
339  }
340  }
341  }
342  return {data_type, seen.size()};
343 }
