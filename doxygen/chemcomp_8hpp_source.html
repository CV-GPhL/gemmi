279  std::vector<AtomId> find_shortest_path(const AtomId& a, const AtomId& b,
280  std::vector<AtomId> visited,
281  int min_length=1) const {
282  int start = (int) visited.size();
283  int end = -1;
284  visited.push_back(b);
285  std::vector<int> parent(visited.size(), -1);
286  std::vector<int> depth(visited.size(), -1);
287  depth[start] = 0;
288  for (int n = start; end == -1 && n != (int) visited.size(); ++n) {
289  for (const Bond& bond : bonds)
290  if (const AtomId* id = bond.other(visited[n])) {
291  if (*id == a) {
292  int cand_len = depth[n] + 1;
293  if (cand_len >= min_length) {
294  end = (int) visited.size();
295  if (!in_vector(*id, visited)) {
296  visited.push_back(*id);
297  parent.push_back(n);
298  depth.push_back(cand_len);
299  }
300  }
301  } else if (!in_vector(*id, visited)) {
302  visited.push_back(*id);
303  parent.push_back(n);
304  depth.push_back(depth[n] + 1);
305  }
306  }
307  }
308  std::vector<AtomId> path;
309  for (int n = end; n != -1; n = parent[n])
310  path.push_back(visited[n]);
311  return path;
312  }

421  void rename_atom(const AtomId& atom_id, const std::string& new_name) {
422  auto rename_atom = [&](AtomId& id) {
423  if (id == atom_id)

473 inline double chiral_abs_volume(double bond1, double bond2, double bond3,
474  double angle1, double angle2, double angle3) {
475  double mult = bond1 * bond2 * bond3;
476  double x = 1;
477  double y = 2;
478  for (double a : {angle1, angle2, angle3}) {
479  double cosine = a == 90. ? 0. : std::cos(rad(a));
480  x -= cosine * cosine;
481  y *= cosine;
482  }
483  return mult * std::sqrt(std::max(0., x + y));
484 }

844 inline ChemComp make_chemcomp_from_block(const cif::Block& block_) {
845  ChemComp cc;
846  cc.name = block_.name.substr(starts_with(block_.name, "comp_") ? 5 : 0);
847  cif::Block& block = const_cast<cif::Block&>(block_);
848  // CCD uses _chem_comp.type, monomer libraries use .group in a separate block
849  // named comp_list, but it'd be a better idea to have _chem_comp.group in the
850  // same block, so we try read it.
851  if (cif::Column group_col = block.find_values("_chem_comp.group"))
852  cc.set_group(group_col.str(0));
853  else if (cif::Column type_col = block.find_values("_chem_comp.type"))
854  cc.type_or_group = type_col.str(0);
855  for (auto row : block.find("_chem_comp_atom.",
856  {"atom_id", "type_symbol", "?type_energy",
857  "?charge", "?partial_charge", "?alt_atom_id",
858  "?atom_type",
859  "?model_Cartn_x",
860  "?model_Cartn_y",
861  "?model_Cartn_z",
862  "?x",
863  "?y",
864  "?z",
865  "?pdbx_model_Cartn_x_ideal",
866  "?pdbx_model_Cartn_y_ideal",
867  "?pdbx_model_Cartn_z_ideal"})) {
868  ChemComp::Atom atom;
869  atom.id = row.str(0);
870  atom.old_id = row.has(5) ? row.str(5) : "";
871  atom.el = Element(row.str(1));
872  atom.charge = (float) cif::as_number(row.one_of(3, 4), 0.0);
873  atom.chem_type = row.has(2) ? row.str(2) : "";
874  atom.acedrg_type = row.has(6) ? row.str(6) : "";
875  auto set_xyz_if_finite = [&](int ix, int iy, int iz) {
876  if (!row.has(ix) || !row.has(iy) || !row.has(iz))
877  return false;
878  double x = cif::as_number(row[ix]);
879  double y = cif::as_number(row[iy]);
880  double z = cif::as_number(row[iz]);
881  if (!std::isfinite(x) || !std::isfinite(y) || !std::isfinite(z))
882  return false;
883  atom.xyz = Position(x, y, z);
884  return true;
885  };
886  if (!set_xyz_if_finite(7, 8, 9) &&
887  !set_xyz_if_finite(10, 11, 12))
888  set_xyz_if_finite(13, 14, 15);
889  cc.atoms.push_back(std::move(atom));
890  }
891  // Also check _chem_comp_acedrg table for atom types (used by acedrg output)
892  for (auto row : block.find("_chem_comp_acedrg.", {"atom_id", "atom_type"})) {
893  std::string atom_id = row.str(0);
894  std::string atom_type = row.str(1);
895  for (auto& atom : cc.atoms)
896  if (atom.id == atom_id && atom.acedrg_type.empty()) {
897  atom.acedrg_type = atom_type;
898  break;
899  }
900  }
901  for (auto row : block.find("_chem_comp_bond.",
902  {"atom_id_1", "atom_id_2", // 0, 1
903  "?type", "?value_order", // 2, 3
904  "?aromatic", "?pdbx_aromatic_flag", // 4, 5
905  "?value_dist", "?value_dist_esd", // 6, 7
906  "?value_dist_nucleus", "?value_dist_nucleus_esd"})) { // 8, 9
907  bool aromatic_flag = (row.one_of(4, 5)[0] | 0x20) == 'y';
908  double dist = row.has(6) ? cif::as_number(row[6]) : NAN;
909  double esd = row.has(7) ? cif::as_number(row[7]) : NAN;
910  double dist_nucl = row.has(8) ? cif::as_number(row[8]) : NAN;
911  double esd_nucl = row.has(9) ? cif::as_number(row[9]) : NAN;
912  BondType bt = bond_type_from_string(row.one_of(2, 3));
913  cc.rt.bonds.push_back({{1, row.str(0)}, {1, row.str(1)},
914  bt,
915  aromatic_flag, dist, esd, dist_nucl, esd_nucl});
916  }
917  for (auto row : block.find("_chem_comp_angle.",
918  {"atom_id_1", "atom_id_2", "atom_id_3",
919  "value_angle", "value_angle_esd"}))
920  cc.rt.angles.push_back({{1, row.str(0)}, {1, row.str(1)}, {1, row.str(2)},
921  cif::as_number(row[3]), cif::as_number(row[4])});
922  for (auto row : block.find("_chem_comp_tor.",
923  {"id",
924  "atom_id_1", "atom_id_2",
925  "atom_id_3", "atom_id_4",
926  "value_angle", "value_angle_esd",
927  "period"}))
928  cc.rt.torsions.push_back({row.str(0),
929  {1, row.str(1)}, {1, row.str(2)},
930  {1, row.str(3)}, {1, row.str(4)},
931  cif::as_number(row[5]), cif::as_number(row[6]),
932  cif::as_int(row[7])});
933  for (auto row : block.find("_chem_comp_chir.",
934  {"atom_id_centre",
935  "atom_id_1", "atom_id_2", "atom_id_3",
936  "volume_sign"}))
937  if (row[4][0] != 'c') // ignore crossN
938  cc.rt.chirs.push_back({{1, row.str(0)},
939  {1, row.str(1)}, {1, row.str(2)}, {1, row.str(3)},
940  chirality_from_string(row[4])});
941  // mmCIF compliant
942  cif::Table chir_tab = block.find("_chem_comp_chir_atom.",
943  {"chir_id", "atom_id"});
944  if (chir_tab.ok()) {
945  std::map<std::string, std::vector<std::string>> chir_atoms;
946  for (auto chir : chir_tab)
947  chir_atoms[chir[0]].push_back(chir[1]);
948  for (auto row : block.find("_chem_comp_chir.",
949  {"id", "atom_id", "volume_flag", "volume_three"})) {
950  auto atoms = chir_atoms.find(row.str(0));
951  if (atoms != chir_atoms.end() && atoms->second.size() == 3)
952  cc.rt.chirs.push_back({{1, row.str(1)},
953  {1, atoms->second[0]}, {1, atoms->second[1]},
954  {1, atoms->second[2]},
955  chirality_from_flag_and_volume(row[2],
956  cif::as_number(row[3]))});
957  }
958  }
959  for (auto row : block.find("_chem_comp_plane_atom.",
960  {"plane_id", "atom_id" , "dist_esd"})) {
961  Restraints::Plane& plane = cc.rt.get_or_add_plane(row.str(0));
962  if (plane.esd == 0.0)
963  plane.esd = cif::as_number(row[2]);
964  plane.ids.push_back({1, row.str(1)});
965  }
966  for (auto row : block.find("_chem_comp_alias.",
967  {"group", "atom_id", "atom_id_standard"})) {
968  ChemComp::Group group = ChemComp::read_group(row.str(0));
969  if (cc.aliases.empty() || cc.aliases.back().group != group) {
970  cc.aliases.emplace_back();
971  cc.aliases.back().group = group;
972  }
973  cc.aliases.back().related.emplace_back(row.str(1), row.str(2));
974  }
975 
976  return cc;
977 }
