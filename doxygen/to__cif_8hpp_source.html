121  void write(const char* s, size_t len) {
122  constexpr int margin = sizeof(buf) - 512;
123  if (ptr - buf + len > margin) {
124  flush();
125  if (len > margin) {
126  os.write(s, len);

168  for (size_t pos = 0, end = 0; end != std::string::npos; pos = end + 1) {

194 inline void write_out_loop(BufOstream& os, const Loop& loop, WriteOptions options) {
195  if (loop.values.empty())
196  return;
197  if (options.prefer_pairs && loop.length() == 1) {
198  for (size_t i = 0; i != loop.tags.size(); ++i)
199  write_out_pair(os, loop.tags[i], loop.values[i], options);
200  return;
201  }
202  // tags
203  os.write("loop_", 5);
204  for (const std::string& tag : loop.tags) {
205  os.put('\n');
206  os << tag;
207  }
208  // values
209  size_t ncol = loop.tags.size();
210 
211  std::vector<size_t> col_width(ncol, 0);
212  if (options.align_loops > 0) {
213  size_t col = 0;
214  for (const std::string& val : loop.values) {
215  if (!is_text_field(val))
216  col_width[col] = std::max(col_width[col], val.size());
217  if (++col == ncol)
218  col = 0;
219  }
220  for (size_t& w : col_width)
221  w = std::min(w, (size_t)options.align_loops);
222  }
223 
224  size_t col = 0;
225  bool need_new_line = true;
226  for (const std::string& val : loop.values) {
227  bool text_field = is_text_field(val);
228  os.put(need_new_line || text_field ? '\n' : ' ');
229  need_new_line = text_field;
230  if (text_field)
231  write_text_field(os, val);
232  else
233  os << val;
234  if (col != ncol - 1) {
235  if (val.size() < col_width[col])
236  os.pad(col_width[col] - val.size());
237  ++col;
238  } else {
239  col = 0;
240  need_new_line = true;
241  }
242  }
243  os.put('\n');
244 }

246 inline void write_out_item(BufOstream& os, const Item& item, WriteOptions options) {
247  switch (item.type) {
248  case ItemType::Pair:
249  write_out_pair(os, item.pair[0], item.pair[1], options);
250  break;
251  case ItemType::Loop:
252  write_out_loop(os, item.loop, options);
253  break;
254  case ItemType::Frame:
255  os.write("save_", 5);
256  os << item.frame.name;
257  os.put('\n');
258  for (const Item& inner_item : item.frame.items)
259  write_out_item(os, inner_item, options);
260  os.write("save_\n", 6);
261  break;
262  case ItemType::Comment:
263  os << item.pair[1];
264  os.put('\n');
265  break;
266  case ItemType::Erased:
267  break;
268  }

271 inline bool should_be_separated_(const Item& a, const Item& b) {
272  if (a.type == ItemType::Comment || b.type == ItemType::Comment)
273  return false;
274  if (a.type != ItemType::Pair || b.type != ItemType::Pair)
275  return true;
276  // check if we have mmcif-like tags from different categories
277  auto adot = a.pair[0].find('.');
278  if (adot == std::string::npos)
279  return false;
280  auto bdot = b.pair[0].find('.');
281  return adot != bdot || a.pair[0].compare(0, adot, b.pair[0], 0, adot) != 0;
282 }

291 inline void write_cif_block_to_stream(std::ostream& os_, const Block& block,
292  WriteOptions options=WriteOptions()) {
293  BufOstream os(os_);

295  os << block.name;
296  os.put('\n');

304  if (options.misuse_hash)

306  os.put('\n');
307  }

309  prev = &item;
310  }
311  if (options.misuse_hash)
312  os.write("#\n", 2);
313 }
