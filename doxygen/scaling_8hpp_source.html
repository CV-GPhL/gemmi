36 inline std::vector<Vec6> adp_symmetry_constraints(const SpaceGroup* sg) {
37  auto constraints = [](const std::initializer_list<int>& l) {
38  constexpr double K2 = 0.70710678118654752440; // sqrt(1/2)
39  constexpr double K3 = 0.57735026918962576451; // sqrt(1/3)
40  static const Vec6 vv[10] = {
41  {1, 0, 0, 0, 0, 0}, // 0
42  {0, 1, 0, 0, 0, 0}, // 1
43  {0, 0, 1, 0, 0, 0}, // 2
44  {0, 0, 0, 1, 0, 0}, // 3
45  {0, 0, 0, 0, 1, 0}, // 4
46  {0, 0, 0, 0, 0, 1}, // 5
47  {K2, K2, 0, 0, 0, 0}, // 6
48  {2/3., 2/3., 0, 1/3., 0, 0}, // 7
49  {K3, K3, K3, 0, 0, 0}, // 8
50  {0, 0, 0, K3, K3, K3} // 9
51  };
52  std::vector<Vec6> c;
53  c.reserve(l.size());
54  for (int i : l)
55  c.push_back(vv[i]);
56  return c;
57  };
58  CrystalSystem cr_system = sg ? sg->crystal_system() : CrystalSystem::Triclinic;
59  switch (cr_system) {
60  case CrystalSystem::Triclinic:
61  return constraints({0, 1, 2, 3, 4, 5});
62  case CrystalSystem::Monoclinic:
63  // the last index is: a->5, b->4, c->3
64  return constraints({0, 1, 2, 3 + 'c' - sg->monoclinic_unique_axis()});
65  case CrystalSystem::Orthorhombic:
66  return constraints({0, 1, 2});
67  case CrystalSystem::Tetragonal:
68  return constraints({6, 2});
69  case CrystalSystem::Hexagonal:
70  case CrystalSystem::Trigonal:
71  if (sg->ext == 'R')
72  return constraints({8, 9});
73  return constraints({7, 2});
74  case CrystalSystem::Cubic:
75  return constraints({8});
76  }
77  unreachable();
78 }

225  void prepare_points(const AsuData<std::complex<Real>>& calc,
226  const AsuData<ValueSigma<Real>>& obs,

228  if (use_solvent && !(mask_data && mask_data->size() == calc.size()))

231  points.reserve(std::min(calc.size(), obs.size()));
232  auto c = calc.v.begin();

234  if (c->hkl != o.hkl) {

236  ++c;
237  if (c == calc.v.end())

240  if (c->hkl != o.hkl)
241  continue;
242  }
243  if (use_solvent) {
244  const HklValue<std::complex<Real>>& m = mask_data->v[c - calc.v.begin()];
245  if (m.hkl != c->hkl)
246  fail("prepare_points(): unexpected data");
247  fmask = m.value;
248  }

250  if (!std::isnan(o.value.value) && !std::isnan(o.value.sigma))
251  points.push_back({o.hkl, stol2, c->value, fmask, o.value.value, o.value.sigma});
252  ++c;
253  if (c == calc.v.end())
254  break;
255  }

285  void fit_isotropic_b_approximately() {
286  double sx = 0, sy = 0, sxx = 0, sxy = 0;
287  int n = 0;
288  for (const Point& p : points) {
289  if (p.fobs < 1 || p.fobs < p.sigma) // skip weak reflections
290  continue;
291  double fcalc = std::abs(get_fcalc(p));
292  double x = p.stol2;
293  double y = std::log(static_cast<float>(p.fobs / fcalc));
294  sx += x;
295  sy += y;
296  sxx += x * x;
297  sxy += x * y;
298  n += 1;
299  }
300  if (n <= 5) // this is not expected to happen
301  return;
302  double slope = (n * sxy - sx * sy) / (n * sxx - sx * sx);
303  double intercept = (sy - slope * sx) / n;
304  double b_iso = -slope;
305  k_overall = std::exp(intercept);
306  set_b_overall({b_iso, b_iso, b_iso, 0, 0, 0});
307  }

311  double lsq_k_overall() const {
312  double sxx = 0, sxy = 0;
313  for (const Point& p : points) {
314  if (p.fobs < 1 || p.fobs < p.sigma) // skip weak reflections
315  continue;
316  double x = std::abs(get_fcalc(p));
317  double y = p.fobs;
318  sxx += x * x;
319  sxy += x * y;
320  }
321  return sxx != 0. ? sxy / sxx : 1.;
322  }

327  void fit_b_star_approximately() {
328  double b_iso = 1/3. * get_b_overall().trace();
329  //size_t nc = constraint_matrix.size();
330  double M[36] = {};
331  double b[6] = {};
332  //std::vector<double> Vc(nc);
333  for (const Point& p : points) {
334  double fcalc = std::abs(get_fcalc(p));
335  // the factor 1 / 2 pi^2 will be taken into account later
336  double Z = std::log(p.fobs / (k_overall * fcalc)) + b_iso * p.stol2;
337  Vec3 h(p.hkl);
338  double V[6] = {h.x * h.x, h.y * h.y, h.z * h.z,
339  2 * h.x * h.y, 2 * h.x * h.z, 2 * h.y * h.z};
340  //for (size_t i = 0; i < nc; ++i)
341  // Vc[i] = vec6_dot(constraint_matrix[i], V);
342  for (size_t i = 0; i < 6; ++i) {
343  for (size_t j = 0; j < 6; ++j)
344  M[6*i+j] += V[i] * V[j];
345  b[i] -= Z * V[i];
346  }
347  }
348  jordan_solve(M, b, 6);
349  double b_star_iso = 1/3. * b_star.trace();
350  SMat33<double> u_star{b[0], b[1], b[2], b[3], b[4], b[5]};
351  // u_to_b() / (2 pi^2) = 8 pi^2 / 2 pi^2 = 4
352  b_star = u_star.scaled(4.0).added_kI(b_star_iso);
353  //auto e = get_b_overall().elements_pdb();
354  //printf("fitted B = {%g %g %g %g %g %g}\n", e[0], e[1], e[2], e[3], e[4], e[5]);
355  }

387  double compute_value_and_derivatives(const Point& p, std::vector<double>& dy_da) const {
388  Vec3 h(p.hkl);
389  double kaniso = std::exp(-0.25 * b_star.r_u_r(h));
390  double fcalc_abs;
391  int n = 1;
392  if (use_solvent) {
393  double solv_b = std::exp(-b_sol * p.stol2);
394  double solv_scale = k_sol * solv_b;
395  auto fcalc = p.fcmol + (Real)solv_scale * p.fmask;
396  fcalc_abs = std::abs(fcalc);
397  double dy_dsol = (fcalc.real() * p.fmask.real() +
398  fcalc.imag() * p.fmask.imag()) / fcalc_abs * k_overall * kaniso;
399  if (!fix_k_sol)
400  dy_da[n++] = solv_b * dy_dsol;
401  if (!fix_b_sol)
402  dy_da[n++] = -p.stol2 * solv_scale * dy_dsol;
403  } else {
404  fcalc_abs = std::abs(p.fcmol);
405  }
406  double fe = fcalc_abs * kaniso;
407  double y = k_overall * fe;
408  dy_da[0] = fe; // dy/d k_overall
409  SMat33<double> du = {
410  -0.25 * y * (h.x * h.x),
411  -0.25 * y * (h.y * h.y),
412  -0.25 * y * (h.z * h.z),
413  -0.5 * y * (h.x * h.y),
414  -0.5 * y * (h.x * h.z),
415  -0.5 * y * (h.y * h.z),
416  };
417  for (size_t j = 0; j < constraint_matrix.size(); ++j)
418  dy_da[n+j] = vec6_dot(constraint_matrix[j], du);
419  return y;
420  }
