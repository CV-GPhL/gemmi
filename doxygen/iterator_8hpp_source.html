27 template <typename Policy>
28 struct BidirIterator : Policy {
29  using value_type = typename std::remove_cv<typename Policy::value_type>::type;
30  using difference_type = std::ptrdiff_t;
31  using pointer = typename Policy::value_type*;
32  using reference = typename Policy::reference;
33  using iterator_category = std::bidirectional_iterator_tag;
34 
36  BidirIterator() = default;
39  BidirIterator(Policy&& p) : Policy(p) {}
40 
42  BidirIterator& operator++() { Policy::increment(); return *this; }
44  BidirIterator operator++(int) { BidirIterator x = *this; ++*this; return x; }
46  BidirIterator& operator--() { Policy::decrement(); return *this; }
48  BidirIterator operator--(int) { BidirIterator x = *this; --*this; return x; }
50  bool operator==(const BidirIterator &o) const { return Policy::equal(o); }
52  bool operator!=(const BidirIterator &o) const { return !Policy::equal(o); }
54  reference operator*() { return Policy::dereference(); }
56  pointer operator->() { return &Policy::dereference(); }
57  using const_variant = BidirIterator<typename Policy::const_policy>;
59  operator const_variant() const {
60  return const_variant(static_cast<const Policy&>(*this));
61  }
62 };

134 template<typename Vector, typename Value>
135 class UniqIterPolicy {
136 public:
137  using value_type = Value;
138  using reference = Value&;
140  UniqIterPolicy() : vec_(nullptr), pos_(0) {}
144  UniqIterPolicy(Vector* vec, std::size_t pos) : vec_(vec), pos_(pos) {}
146  void increment() {
147  // move to the first element of the next group
148  const auto& key = (*vec_)[pos_].group_key();
149  ++pos_;
150  while (pos_ != vec_->size() && (*vec_)[pos_].group_key() == key)
151  ++pos_;
152  }
154  void decrement() {
155  --pos_; // now we are at the last element of the previous group
156  const auto& key = (*vec_)[pos_].group_key();
157  while (pos_ != 0 && (*vec_)[pos_-1].group_key() == key)
158  --pos_; // move to the group beginning
159  }
161  bool equal(const UniqIterPolicy& o) const { return pos_ == o.pos_; }
163  Value& dereference() { return (*vec_)[pos_]; }
164  using const_policy = UniqIterPolicy<Vector const, Value const>;
166  operator const_policy() const { return const_policy(vec_, pos_); }
167 private:
168  Vector* vec_;
169  std::size_t pos_;
170 };

200 template<typename Vector, typename Value>
201 class GroupingIterPolicy {
202 public:
203  using value_type = Value;
204  using reference = Value&;
206  GroupingIterPolicy() = default;
209  GroupingIterPolicy(const Value& span) : span_(span) {}
211  void increment() {
212  span_.set_begin(span_.end());
213  span_.set_size(0);
214  while (!span_.is_ending() &&
215  span_.begin()->group_key() == span_.end()->group_key())
216  span_.set_size(span_.size() + 1);
217  }
219  void decrement() {
220  span_.set_begin(span_.begin() - 1);
221  span_.set_size(1);
222  while (!span_.is_beginning() &&
223  span_.begin()->group_key() == (span_.begin() - 1)->group_key()) {
224  span_.set_begin(span_.begin() - 1);
225  span_.set_size(span_.size() + 1);
226  }
227  }
229  bool equal(const GroupingIterPolicy& o) const {
230  return span_.begin() == o.span_.begin();
231  }
233  Value& dereference() { return span_; }
234  using const_policy = GroupingIterPolicy<Vector const, Value const>;
236  operator const_policy() const { return const_policy(span_); }
