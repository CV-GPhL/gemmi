182  AsuData<R> prepare_asu_data(double dmin=0, double unblur=0,
183  bool with_000=false, bool with_sys_abs=false,
184  bool mott_bethe=false) {
185  AsuData<R> asu_data;
186  if (this->axis_order == AxisOrder::ZYX)
187  fail("get_asu_values(): ZYX order is not supported yet");
188  int max_h = (this->nu - 1) / 2;
189  int max_k = (this->nv - 1) / 2;
190  int max_l = half_l ? this->nw - 1 : (this->nw - 1) / 2;
191  double max_1_d2 = 0.;
192  if (dmin != 0.) {
193  max_1_d2 = 1. / (dmin * dmin);
194  Miller lim = this->unit_cell.get_hkl_limits(dmin);
195  max_h = std::min(max_h, lim[0]);
196  max_k = std::min(max_k, lim[1]);
197  max_l = std::min(max_l, lim[2]);
198  }
199  gemmi::ReciprocalAsu asu(this->spacegroup);
200  std::unique_ptr<GroupOps> gops;
201  if (!with_sys_abs && this->spacegroup)
202  gops.reset(new GroupOps(this->spacegroup->operations()));
203  Miller hkl;
204  for (hkl[0] = -max_h; hkl[0] <= max_h; ++hkl[0]) {
205  int hi = hkl[0] >= 0 ? hkl[0] : hkl[0] + this->nu;
206  int hi_ = -hkl[0] >= 0 ? -hkl[0] : -hkl[0] + this->nu;
207  for (hkl[1] = -max_k; hkl[1] <= max_k; ++hkl[1]) {
208  hkl[2] = -max_l;
209  // (hkl)s with l<0 might be needed to get complete asu.
210  // If they are absent in the data (Hermitian FFT), use Friedel's pairs.
211  if (half_l) {
212  int ki_ = -hkl[1] >= 0 ? -hkl[1] : -hkl[1] + this->nv;
213  for (; hkl[2] < 0; ++hkl[2])
214  if (asu.is_in(hkl) &&
215  (max_1_d2 == 0. || this->unit_cell.calculate_1_d2(hkl) < max_1_d2) &&
216  (with_sys_abs || !gops->is_systematically_absent(hkl)))
217  asu_data.v.push_back({hkl,
218  friedel_mate_value(this->get_value_q(hi_, ki_, -hkl[2]))});
219  }
220  int ki = hkl[1] >= 0 ? hkl[1] : hkl[1] + this->nv;
221  for (; hkl[2] <= max_l; ++hkl[2])
222  if (asu.is_in(hkl) &&
223  (max_1_d2 == 0. || this->unit_cell.calculate_1_d2(hkl) < max_1_d2) &&
224  (with_sys_abs || !gops->is_systematically_absent(hkl)) &&
225  (with_000 || !(hkl[0] == 0 && hkl[1] == 0 && hkl[2] == 0))) {
226  int li = hkl[2] >= 0 ? hkl[2] : hkl[2] + this->nw;
227  asu_data.v.push_back({hkl, this->get_value_q(hi, ki, li)});
228  }
229  }
230  }
231  if (unblur != 0. || mott_bethe)
232  for (HklValue<R>& hv : asu_data.v) {
233  double inv_d2 = this->unit_cell.calculate_1_d2(hv.hkl);
234  double mult = 1;
235  if (unblur != 0)
236  // cf. reciprocal_space_multiplier()
237  mult = std::exp(unblur * 0.25 * inv_d2);
238  if (mott_bethe)
239  // cf. mott_bethe_factor
240  mult *= -mott_bethe_const() / inv_d2;
241  hv.value *= static_cast<decltype(std::abs(hv.value))>(mult);
242  }
243  asu_data.unit_cell_ = this->unit_cell;
244  asu_data.spacegroup_ = this->spacegroup;
245  return asu_data;
246  }
