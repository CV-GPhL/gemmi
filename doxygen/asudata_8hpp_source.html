20 struct ComplexCorrelation {
21  int n = 0;
22  double sum_xx = 0.;
23  double sum_yy = 0.;
24  std::complex<double> sum_xy = 0.;
25  std::complex<double> mean_x = 0.;
26  std::complex<double> mean_y = 0.;
30  void add_point(std::complex<double> x, std::complex<double> y) {
31  ++n;
32  double inv_n = 1.0 / n;
33  double weight = (n - 1.0) * inv_n;
34  std::complex<double> dx = x - mean_x;
35  std::complex<double> dy = y - mean_y;
36  sum_xx += weight * std::norm(dx);
37  sum_yy += weight * std::norm(dy);
38  sum_xy += weight * (dx * std::conj(dy));
39  mean_x += dx * inv_n;
40  mean_y += dy * inv_n;
41  }
45  void add_point(std::complex<float> x, std::complex<float> y) {
46  add_point(std::complex<double>(x), std::complex<double>(y));
47  }
49  std::complex<double> coefficient() const { return sum_xy / std::sqrt(sum_xx * sum_yy); }
51  double mean_ratio() const { return std::abs(mean_y) / std::abs(mean_x); }
52 };

64 void for_matching_reflections(const std::vector<T>& a,
65  const std::vector<T>& b,
66  const Func& func) {
67  auto r1 = a.begin();
68  auto r2 = b.begin();
69  while (r1 != a.end() && r2 != b.end()) {
70  if (r1->hkl == r2->hkl) {
71  func(*r1, *r2);
72  ++r1;
73  ++r2;
74  } else if (r1->hkl < r2->hkl) {
75  ++r1;
76  } else {
77  ++r2;
78  }

168 template<typename R>
169 void move_to_asu(const GroupOps& gops, const Miller& hkl, int isym,
170  HklValue<std::complex<R>>& v) {
171  v.hkl = hkl;
172  // cf. Mtz::ensure_asu()
173  const Op& op = gops.sym_ops[(isym - 1) / 2];
174  double shift = op.phase_shift(hkl);
175  if (shift != 0) {
176  double phase = std::arg(v.value) + shift;
177  v.value = std::polar(std::abs(v.value), (R)phase);
178  }
179  if (isym % 2 == 0)
180  v.value.imag(-v.value.imag());
181 }

213  void ensure_asu(bool tnt_asu=false) {
214  if (!spacegroup_)
215  fail("AsuData::ensure_asu(): space group not set");
216  GroupOps gops = spacegroup_->operations();
217  ReciprocalAsu asu(spacegroup_, tnt_asu);
218  for (HklValue<T>& hkl_value : v) {
219  const Miller& hkl = hkl_value.hkl;
220  if (asu.is_in(hkl))
221  continue;
222  auto result = asu.to_asu(hkl, gops);
223  impl::move_to_asu(gops, result.first, result.second, hkl_value);
224  }

235  void load_values(const DataProxy& proxy, const std::string& label,
236  bool as_is=false) {
237  std::size_t col = proxy.column_index(label);

260  template<int N, typename DataProxy>
261  void load_values(const DataProxy& proxy, const std::array<std::string,N>& labels,
262  bool as_is=false) {
263  std::array<std::size_t, N> cols;
264  for (int i = 0; i < N; ++i)
265  cols[i] = proxy.column_index(labels[i]);
266  unit_cell_ = proxy.unit_cell();
267  spacegroup_ = proxy.spacegroup();
268  using Val = typename T::value_type;
269  for (size_t i = 0; i < proxy.size(); i += proxy.stride()) {
270  std::array<Val, N> nums;
271  for (int j = 0; j < N; ++j)
272  nums[j] = (Val) proxy.get_num(i + cols[j]);
273  if (!std::any_of(nums.begin(), nums.end(), [](Val f) { return std::isnan(f); })) {
274  v.emplace_back();
275  v.back().hkl = proxy.get_hkl(i);
276  set_value_from_array(v.back().value, nums);
277  }
278  }
279  if (!as_is) {
280  ensure_asu();
281  ensure_sorted();
282  }
283  }
