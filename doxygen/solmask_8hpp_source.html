131 template<typename T>
132 void mask_points_in_radius(Grid<T>& mask, const Model& model,
133  AtomicRadiiSet atomic_radii_set,
134  double r_probe, T value,
135  bool ignore_hydrogen,
136  bool ignore_zero_occupancy_atoms) {
137  for (const Chain& chain : model.chains)
138  for (const Residue& res : chain.residues)
139  for (const Atom& atom : res.atoms) {
140  if ((ignore_hydrogen && atom.is_hydrogen()) ||
141  (ignore_zero_occupancy_atoms && atom.occ <= 0))
142  continue;
143  El elem = atom.element.elem;
144  double r = r_probe;
145  switch (atomic_radii_set) {
146  case AtomicRadiiSet::VanDerWaals: r += vdw_radius(elem); break;
147  case AtomicRadiiSet::Cctbx: r += cctbx_vdw_radius(elem); break;
148  case AtomicRadiiSet::Refmac: r += refmac_radius_for_bulk_solvent(elem); break;
149  case AtomicRadiiSet::Constant: /* r is included in r_probe */ break;
150  }
151  mask.set_points_around(atom.pos, r, value);
152  }
153 }

194 inline
195 void mask_points_using_occupancy(Grid<float>& mask, const Model& model,
196  AtomicRadiiSet atomic_radii_set, double r_probe,
197  bool ignore_hydrogen,
198  bool ignore_zero_occupancy_atoms) {
199 
200  std::string altlocs = distinct_altlocs(model);
201  altlocs += '\0'; // no altloc
202 
203  gemmi::Grid<float> m;
204  m.copy_metadata_from(mask);
205  for (char& altloc : altlocs) {
206  m.fill(0.0);
207  for (const Chain& chain : model.chains) {
208  for (const Residue& res : chain.residues) {
209  for (const Atom& atom : res.atoms) {
210  if ((ignore_hydrogen && atom.is_hydrogen()) ||
211  (ignore_zero_occupancy_atoms && atom.occ <= 0))
212  continue;
213  if (atom.altloc == altloc) {
214  El elem = atom.element.elem;
215  double r = r_probe;
216  switch (atomic_radii_set) {
217  case AtomicRadiiSet::VanDerWaals: r += vdw_radius(elem); break;
218  case AtomicRadiiSet::Cctbx: r += cctbx_vdw_radius(elem); break;
219  case AtomicRadiiSet::Refmac: r += refmac_radius_for_bulk_solvent(elem); break;
220  case AtomicRadiiSet::Constant: /* r is included in r_probe */ break;
221  }
222  Fractional fpos = m.unit_cell.fractionalize(atom.pos);
223  m.use_points_around<true>(fpos, r, [&](float& ref, double) {
224  ref = std::min(ref, -atom.occ);
225  }, false);
226  }
227  }
228  }
229  }
230 
231  // reduce starting mask
232  for (size_t i = 0; i < m.data.size(); ++i) {
233  if (m.data[i] < 0.0)
234  mask.data[i] += m.data[i];
235  }
236  }
237 
238  // When we reduced the solvent mask by the atom occupancy each time
239  // we hit a grid point, we might end up with a value below zero (due
240  // to overlapping atoms) and need to limit it here.
241  for (float& d : mask.data)
242  d = std::max(d, 0.f);
243 }

254 template<typename T>
255 void set_margin_around(Grid<T>& mask, double r, T value, T margin_value) {
256  int du = (int) std::floor(r / mask.spacing[0]);
257  int dv = (int) std::floor(r / mask.spacing[1]);
258  int dw = (int) std::floor(r / mask.spacing[2]);
259  double max_spacing2 = sq(std::max(std::max(mask.unit_cell.a / mask.nu,
260  mask.unit_cell.b / mask.nv),
261  mask.unit_cell.c / mask.nw)) + 1e-6;
262  if (2 * du >= mask.nu || 2 * dv >= mask.nv || 2 * dw >= mask.nw)
263  fail("grid operation failed: radius bigger than half the unit cell?");
264  std::vector<std::array<int,3>> stencil1;
265  std::vector<std::array<int,3>> stencil2;
266  for (int w = -dw; w <= dw; ++w)
267  for (int v = -dv; v <= dv; ++v)
268  for (int u = -du; u <= du; ++u) {
269  Fractional fdelta = mask.get_fractional(u, v, w);
270  double r2 = mask.unit_cell.orthogonalize_difference(fdelta).length_sq();
271  if (r2 <= r * r && r2 != 0.) {
272  std::array<int,3> wvu{{w <= 0 ? w : w - mask.nw,
273  v <= 0 ? v : v - mask.nv,
274  u <= 0 ? u : u - mask.nu}};
275  if (r2 < max_spacing2)
276  stencil1.push_back(wvu);
277  else
278  stencil2.push_back(wvu);
279  }
280  }
281  if (stencil2.empty()) {
282  // r is small; it should be faster to go through masked points and
283  // for each one check if any point in given radius are unmasked.
284  for (typename Grid<T>::Point p : mask)
285  if (*p.value < value) {
286  for (const auto& wvu : stencil1) {
287  size_t idx = mask.index_near_zero(p.u + wvu[2], p.v + wvu[1], p.w + wvu[0]);
288  if (mask.data[idx] >= value) {
289  *p.value = margin_value;
290  break;
291  }
292  }
293  }
294  } else {
295  // r is large; it should be faster to go through unmasked points
296  // and for each one check if it has masked near neighbors (stencil1).
297  // These neighbors get marked as margin. If all near neighbors are
298  // unmasked, we can skip further neighbors (they will be checked
299  // from other points).
300  for (typename Grid<T>::Point p : mask) {
301  if (*p.value >= value) {
302  bool found = false;
303  for (const auto& wvu : stencil1) {
304  size_t idx = mask.index_near_zero(p.u + wvu[2], p.v + wvu[1], p.w + wvu[0]);
305  if (mask.data[idx] < value) {
306  mask.data[idx] = margin_value;
307  found = true;
308  }
309  }
310  if (found)
311  for (const auto& wvu : stencil2) {
312  size_t idx = mask.index_near_zero(p.u + wvu[2], p.v + wvu[1], p.w + wvu[0]);
313  if (mask.data[idx] < value)
314  mask.data[idx] = margin_value;
315  }
316  }
317  }
318  }
319  //printf("stencil sizes: %zu\n", stencil1.size(), stencil2.size());
320  //printf("margin: %zu\n", std::count(mask.data.begin(), mask.data.end(), margin_value));
321 }

552 inline void unmask_symmetry_mates(Grid<NodeInfo>& mask) {
553  std::vector<GridOp> symmetry_ops = mask.get_scaled_ops_except_id();
554  size_t idx = 0;
555  for (int w = 0; w != mask.nw; ++w)
556  for (int v = 0; v != mask.nv; ++v)
557  for (int u = 0; u != mask.nu; ++u, ++idx) {
558  NodeInfo& ni = mask.data[idx];
559  if (ni.found)
560  for (const GridOp& grid_op : symmetry_ops) {
561  std::array<int,3> t = grid_op.apply(u, v, w);
562  NodeInfo& im_ni = mask.data[mask.index_n(t[0], t[1], t[2])];
563  if (im_ni.found && im_ni.dist_sq > ni.dist_sq)
564  im_ni.found = false;
565  }
566  }

580 template<typename T>
581 void interpolate_grid_around_model(Grid<T>& dest, const Grid<T>& src,
582  const Transform& tr,
583  const Model& dest_model, double radius,
584  int order=1) {

588  unmask_symmetry_mates(mask);
589  // Interpolate values for selected nodes.
590  FTransform frac_tr = src.unit_cell.frac.combine(tr.combine(dest.unit_cell.orth));
591  for (size_t idx = 0; idx != mask.data.size(); ++idx) {
592  const NodeInfo& ni = mask.data[idx];
593  if (ni.found) {
594  Fractional dest_fr = dest.get_fractional(ni.u, ni.v, ni.w);
595  Fractional src_fr = frac_tr.apply(dest_fr);
596  dest.data[idx] = src.interpolate_value(src_fr, order);
597  }
598  }
