43 inline Structure make_structure(cif::Document&& doc, cif::Document* save_doc=nullptr) {
44  // mmCIF files for deposition may have more than one block:
45  // coordinates in the first block and restraints in the others.
46  for (size_t i = 1; i < doc.blocks.size(); ++i)
47  if (doc.blocks[i].has_tag("_atom_site.id"))
48  fail("2+ blocks are ok if only the first one has coordinates;\n"
49  "_atom_site in block #" + std::to_string(i+1) + ": " + doc.source);
50  Structure st = make_structure_from_block(doc.blocks.at(0));
51  if (save_doc)
52  *save_doc = std::move(doc);
53  return st;
54 }

95 inline Structure make_structure_from_chemcomp_block(const cif::Block& block, int which=7) {
96  Structure st;
97  st.input_format = CoorFormat::ChemComp;
98  if (const std::string* name = block.find_value("_chem_comp.id"))
99  st.name = *name;
100  auto ok = [which](ChemCompModel x) { return which & static_cast<int>(x); };
101  if (ok(ChemCompModel::Xyz) && block.has_any_value("_chem_comp_atom.x"))
102  st.models.push_back(make_model_from_chemcomp_block(block, ChemCompModel::Xyz));
103  if (ok(ChemCompModel::Example) && block.has_any_value("_chem_comp_atom.model_Cartn_x"))
104  st.models.push_back(make_model_from_chemcomp_block(block, ChemCompModel::Example));
105  if (ok(ChemCompModel::Ideal) && block.has_any_value("_chem_comp_atom.pdbx_model_Cartn_x_ideal"))
106  st.models.push_back(make_model_from_chemcomp_block(block, ChemCompModel::Ideal));
107  st.renumber_models();
108  return st;
109 }

123 inline int check_chemcomp_block_number(const cif::Document& doc) {
124  // monomer library file without global_
125  if (doc.blocks.size() == 2 && doc.blocks[0].name == "comp_list")
126  return 1;
127  // monomer library file with global_
128  if (doc.blocks.size() == 3 && doc.blocks[0].name.empty() &&
129  doc.blocks[1].name == "comp_list")
130  return 2;
131  // CCD file
132  if (doc.blocks.size() == 1 &&
133  !doc.blocks[0].has_tag("_atom_site.id") &&
134  !doc.blocks[0].has_tag("_cell.length_a") &&
135  doc.blocks[0].has_tag("_chem_comp_atom.atom_id"))
136  return 0;
137  return -1;
138 }
