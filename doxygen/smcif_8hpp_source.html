30 SmallStructure make_small_structure_from_block(const cif::Block& block_) {
31  using cif::as_number;
32  cif::Block& block = const_cast<cif::Block&>(block_);
33  SmallStructure st;
34  st.name = block.name;
35 
36  // unit cell and symmetry
37  cif::Table cell = block.find("_cell_",
38  {"length_a", "length_b", "length_c",
39  "angle_alpha", "angle_beta", "angle_gamma"});
40  if (cell.ok()) {
41  auto c = cell.one();
42  if (!cif::is_null(c[0]) && !cif::is_null(c[1]) && !cif::is_null(c[2]))
43  st.cell.set(as_number(c[0]), as_number(c[1]), as_number(c[2]),
44  as_number(c[3]), as_number(c[4]), as_number(c[5]));
45  }
46  for (const char* tag : {"_space_group_name_H-M_alt",
47  "_symmetry_space_group_name_H-M"})
48  if (const std::string* val = block.find_value(tag)) {
49  st.spacegroup_hm = cif::as_string(*val);
50  break;
51  }
52  for (const char* tag : {"_space_group_symop_operation_xyz",
53  "_symmetry_equiv_pos_as_xyz"}) {
54  if (const cif::Column col = block.find_values(tag)) {
55  st.symops.reserve(col.length());
56  for (const std::string& value : col)
57  st.symops.push_back(cif::as_string(value));
58  break;
59  }
60  }
61  for (const char* tag : {"_space_group_name_Hall", "_symmetry_space_group_name_Hall"})
62  if (const std::string* val = block.find_value(tag))
63  st.spacegroup_hall = cif::as_string(*val);
64  for (const char* tag : {"_space_group_IT_number", "_symmetry_Int_Tables_number"})
65  if (const std::string* val = block.find_value(tag)) {
66  st.spacegroup_number = cif::as_int(*val, 0);
67  break;
68  }
69  st.determine_and_set_spacegroup("S.H2");
70 
71  enum { kLabel, kSymbol, kX, kY, kZ, kUiso, kBiso, kOcc, kDisorderGroup };
72  cif::Table atom_table = block.find("_atom_site_",
73  {"label",
74  "?type_symbol",
75  "?fract_x",
76  "?fract_y",
77  "?fract_z",
78  "?U_iso_or_equiv",
79  "?B_iso_or_equiv",
80  "?occupancy",
81  "?disorder_group"});
82  for (auto row : atom_table) {
83  SmallStructure::Site site;
84  site.label = cif::as_string(row[kLabel]);
85  if (row.has(kSymbol))
86  site.type_symbol = cif::as_string(row[kSymbol]);
87  else
88  site.type_symbol = site.label;
89  if (row.has(kX))
90  site.fract.x = as_number(row[kX]);
91  if (row.has(kY))
92  site.fract.y = as_number(row[kY]);
93  if (row.has(kZ))
94  site.fract.z = as_number(row[kZ]);
95  if (row.has(kUiso))
96  site.u_iso = as_number(row[kUiso], 0.0);
97  else if (row.has(kBiso))
98  site.u_iso = as_number(row[kBiso], 0.0) / u_to_b();
99  if (row.has(kOcc))
100  site.occ = as_number(row[kOcc], 1.0);
101  if (row.has2(kDisorderGroup))
102  // ignore non-integer _atom_site_disorder_group
103  site.disorder_group = string_to_int(row[kDisorderGroup], false);
104  split_element_and_charge(site.type_symbol, &site);
105  st.sites.push_back(site);
106  }
107 
108  std::vector<SmallStructure::Site>::iterator aniso_site = st.sites.begin();
109  for (auto row : block.find("_atom_site_aniso_",
110  {"label", "U_11", "U_22", "U_33",
111  "U_12", "U_13", "U_23"})) {
112  std::string aniso_label = row.str(0);
113  if (aniso_site == st.sites.end() || aniso_site->label != aniso_label) {
114  aniso_site = std::find_if(st.sites.begin(), st.sites.end(),
115  [&](SmallStructure::Site& x) { return x.label == aniso_label; });
116  if (aniso_site == st.sites.end())
117  continue;
118  }
119  aniso_site->aniso.u11 = as_number(row[1], 0.0);
120  aniso_site->aniso.u22 = as_number(row[2], 0.0);
121  aniso_site->aniso.u33 = as_number(row[3], 0.0);
122  aniso_site->aniso.u12 = as_number(row[4], 0.0);
123  aniso_site->aniso.u13 = as_number(row[5], 0.0);
124  aniso_site->aniso.u23 = as_number(row[6], 0.0);
125  ++aniso_site;
126  }
127 
128  for (auto row : block.find("_atom_type_",
129  {"symbol",
130  "scat_dispersion_real",
131  "scat_dispersion_imag"})) {
132  SmallStructure::AtomType atom_type;
133  atom_type.symbol = row.str(0);
134  atom_type.dispersion_real = cif::as_number(row[1]);
135  atom_type.dispersion_imag = cif::as_number(row[2]);
136  split_element_and_charge(atom_type.symbol, &atom_type);
137  st.atom_types.push_back(atom_type);
138  }
139  if (cif::Column w_col = block.find_values("_diffrn_radiation_wavelength"))
140  st.wavelength = cif::as_number(w_col.at(0));
141  return st;

151 inline cif::Block make_cif_block_from_small_structure(const SmallStructure& st) {
152  cif::Block block;
153  block.name = st.name;
154  if (st.cell.is_crystal()) {
155  block.set_pair("_cell_length_a", to_str(st.cell.a));
156  block.set_pair("_cell_length_b", to_str(st.cell.b));
157  block.set_pair("_cell_length_c", to_str(st.cell.c));
158  block.set_pair("_cell_angle_alpha", to_str(st.cell.alpha));
159  block.set_pair("_cell_angle_beta", to_str(st.cell.beta));
160  block.set_pair("_cell_angle_gamma", to_str(st.cell.gamma));
161  }
162  if (!st.spacegroup_hm.empty())
163  block.set_pair("_symmetry_space_group_name_H-M", cif::quote(st.spacegroup_hm));
164  if (st.wavelength != 0)
165  block.set_pair("_diffrn_radiation_wavelength", to_str(st.wavelength));
166 
167  cif::Loop& atom_loop = block.init_loop("_atom_site_", {"label",
168  "type_symbol",
169  "fract_x",
170  "fract_y",
171  "fract_z",
172  "U_iso_or_equiv",
173  "occupancy",
174  "disorder_group"});
175  for (const SmallStructure::Site& site: st.sites) {
176  atom_loop.add_row({
177  cif::quote(site.label),
178  site.type_symbol.empty() ? site.element_and_charge_symbol()
179  : cif::quote(site.type_symbol),
180  to_str(site.fract.x),
181  to_str(site.fract.y),
182  to_str(site.fract.z),
183  to_str(site.u_iso),
184  to_str(site.occ),
185  site.disorder_group == 0 ? "." : std::to_string(site.disorder_group)
186  });
187  }
188 
189  bool has_aniso = false;
190  for (const SmallStructure::Site& site: st.sites)
191  if (site.aniso.nonzero()) {
192  has_aniso = true;
193  break;
194  }
195  if (has_aniso) {
196  cif::Loop& aniso_loop = block.init_loop("_atom_site_aniso_", {
197  "label", "U_11", "U_22", "U_33", "U_12", "U_13", "U_23"});
198  for (const SmallStructure::Site& site: st.sites)
199  if (site.aniso.nonzero())
200  aniso_loop.add_row({
201  cif::quote(site.label),
202  to_str(site.aniso.u11), to_str(site.aniso.u22), to_str(site.aniso.u33),
203  to_str(site.aniso.u12), to_str(site.aniso.u13), to_str(site.aniso.u23)
204  });
205  }
206 
207  return block;
208 }
