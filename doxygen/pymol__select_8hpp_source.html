148 enum class CompareOp {

150  LE,
151  GT,
152  GE,

157 struct BfactorNode : Node {
158  CompareOp op;
159  float value;
160  BfactorNode(CompareOp o, float v) : op(o), value(v) {}
161  bool match(const gemmi::FlatAtom& a) const override {
162  switch (op) {
163  case CompareOp::LT: return a.b_iso < value;
164  case CompareOp::LE: return a.b_iso <= value;
165  case CompareOp::GT: return a.b_iso > value;
166  case CompareOp::GE: return a.b_iso >= value;
167  case CompareOp::EQ: return a.b_iso == value;
168  case CompareOp::NE: return a.b_iso != value;
169  }
170  return false;
171  }
172 };
173 
174 struct OccupancyNode : Node {
175  CompareOp op;
176  float value;
177  OccupancyNode(CompareOp o, float v) : op(o), value(v) {}
178  bool match(const gemmi::FlatAtom& a) const override {
179  switch (op) {
180  case CompareOp::LT: return a.occ < value;
181  case CompareOp::LE: return a.occ <= value;
182  case CompareOp::GT: return a.occ > value;
183  case CompareOp::GE: return a.occ >= value;
184  case CompareOp::EQ: return a.occ == value;
185  case CompareOp::NE: return a.occ != value;
186  }
187  return false;
188  }
189 };

201 struct SidechainNode : Node {

582 // AND
583 template<> struct action<and_rest> {
584  static void apply0(State& s) {
585  auto rhs = std::move(s.stack.back()); s.stack.pop_back();
586  auto lhs = std::move(s.stack.back()); s.stack.pop_back();
587 
588  auto node = std::make_unique<psimpl::AndNode>();
589  node->left = std::move(lhs);
590  node->right = std::move(rhs);
591  s.stack.push_back(std::move(node));
592  }

600 
601  auto node = std::make_unique<psimpl::OrNode>();
602  node->left = std::move(lhs);
603  node->right = std::move(rhs);

621 inline std::unique_ptr<psimpl::Node> compile_pymol_selection(const std::string& selector) {
622  psimpl::State state;
623  tao::pegtl::memory_input<> in(selector, "");
624  try {
