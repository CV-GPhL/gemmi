130 inline Correlation combine_two_correlations(const Correlation& a, const Correlation& b) {
131  auto sq = [](double x) { return x * x; };
132  Correlation r;
133  r.n = a.n + b.n;
134  r.mean_x = (a.n * a.mean_x + b.n * b.mean_x) / r.n;
135  r.mean_y = (a.n * a.mean_y + b.n * b.mean_y) / r.n;
136  r.sum_xx = a.sum_xx + a.n * sq(a.mean_x - r.mean_x)
137  + b.sum_xx + b.n * sq(b.mean_x - r.mean_x);
138  r.sum_yy = a.sum_yy + a.n * sq(a.mean_y - r.mean_y)
139  + b.sum_yy + b.n * sq(b.mean_y - r.mean_y);
140  r.sum_xy = a.sum_xy + a.n * (a.mean_x - r.mean_x) * (a.mean_y - r.mean_y)
141  + b.sum_xy + b.n * (b.mean_x - r.mean_x) * (b.mean_y - r.mean_y);
142  return r;
143 }

173  DataStats stats;
174  double sum = 0;
175  double sq_sum = 0;
176  stats.dmin = INFINITY;
177  stats.dmax = -INFINITY;
178  for (double d : data) {
179  if (std::isnan(d)) {
180  stats.nan_count++;
181  continue;
182  }
183  sum += d;
184  sq_sum += d * d;
185  if (d < stats.dmin)
186  stats.dmin = d;
187  if (d > stats.dmax)
188  stats.dmax = d;
189  }
190  if (stats.nan_count != data.size()) {
191  stats.dmean = sum / (data.size() - stats.nan_count);
192  stats.rms = std::sqrt(sq_sum / (data.size() - stats.nan_count) - stats.dmean * stats.dmean);
193  } else {
194  stats.dmin = NAN;
195  stats.dmax = NAN;
196  }
197  return stats;
