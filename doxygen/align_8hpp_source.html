24 inline std::vector<int> prepare_target_gapo(const ConstResidueSpan& polymer,
25  PolymerType polymer_type,
26  const AlignmentScoring* scoring=nullptr) {
27  if (!scoring)
28  scoring = AlignmentScoring::partial_model();
29  std::vector<int> gaps;
30  gaps.reserve(polymer.size());
31  gaps.push_back(0); // free gap opening at the beginning of sequence
32  if (is_polypeptide(polymer_type) || is_polynucleotide(polymer_type)) {
33  auto first_conformer = polymer.first_conformer();
34  auto res = first_conformer.begin();
35  for (auto next_res = res; ++next_res != first_conformer.end(); res = next_res) {
36  bool connected = are_connected3(*res, *next_res, polymer_type);
37  gaps.push_back(connected ? scoring->bad_gapo : scoring->good_gapo);
38  }
39  gaps.push_back(0); // free gap after the end of chain
40  }
41  return gaps;
42 }

50 inline AlignmentResult align_sequence_to_polymer(
51  const std::vector<std::string>& full_seq,
52  const ConstResidueSpan& polymer,
53  PolymerType polymer_type,
54  const AlignmentScoring* scoring=nullptr) {
55  if (!polymer)
56  return AlignmentResult();
57  std::map<std::string, std::uint8_t> encoding;
58  if (!scoring)
59  scoring = AlignmentScoring::partial_model();
60  for (const std::string& res_name : scoring->matrix_encoding)
61  encoding.emplace(res_name, (std::uint8_t)encoding.size());
62  for (const Residue& res : polymer)
63  encoding.emplace(res.name, (std::uint8_t)encoding.size());
64  for (const std::string& mon_list : full_seq)
65  encoding.emplace(Entity::first_mon(mon_list), (std::uint8_t)encoding.size());
66  if (encoding.size() > 255)
67  return AlignmentResult();
68 
69  std::vector<std::uint8_t> encoded_full_seq(full_seq.size());
70  for (size_t i = 0; i != full_seq.size(); ++i)
71  encoded_full_seq[i] = encoding.at(Entity::first_mon(full_seq[i]));
72 
73  std::vector<std::uint8_t> encoded_model_seq;
74  encoded_model_seq.reserve(polymer.size());
75  for (const Residue& res : polymer.first_conformer())
76  encoded_model_seq.push_back(encoding.at(res.name));
77 
78  return align_sequences(encoded_full_seq, encoded_model_seq,
79  prepare_target_gapo(polymer, polymer_type, scoring),
80  (std::uint8_t)encoding.size(), *scoring);
81 }

120 inline void assign_label_seq_to_polymer(ResidueSpan& polymer,
121  const Entity* ent, bool force) {
122  AlignmentResult result;
123 
124  // sequence not known
125  if (!ent || ent->full_sequence.empty()) {
126  if (!force)
127  return;
128  PolymerType ptype = get_or_check_polymer_type(ent, polymer);
129  const Residue* prev = nullptr;
130  for (const Residue& res : polymer.first_conformer()) {
131  if (prev && !are_connected3(*prev, res, ptype))
132  result.push_cigar(1, 1); // assume a single insertion
133  result.push_cigar(0, 1);
134  prev = &res;
135  }
136 
137  // exact match - common case that doesn't require alignment
138  } else if (seqid_matches_seqres(polymer, *ent)) {
139  result.push_cigar(0, (int)ent->full_sequence.size());
140 
141  // sequence alignment
142  } else {
143  PolymerType ptype = get_or_check_polymer_type(ent, polymer);
144  result = align_sequence_to_polymer(ent->full_sequence, polymer, ptype);
145  }
146 
147  auto res_group = polymer.first_conformer().begin();
148  int id = 1;
149  for (AlignmentResult::Item item : result.cigar) {
150  switch (item.op()) {
151  case 'I':
152  id += item.len();
153  break;
154  case 'D': // leaving label_seq as it is
155  for (uint32_t i = 0; i < item.len(); ++i)
156  res_group++;
157  break;
158  case 'M': // not checking for mismatches
159  for (uint32_t i = 0; i < item.len(); ++i, ++id)
160  for (Residue* res = &*res_group++; res != &*res_group; ++res)
161  res->label_seq = id;
162  break;
163  }
164  }
165 }

213 inline void prepare_positions_for_superposition(std::vector<Position>& pos1,
214  std::vector<Position>& pos2,
215  ConstResidueSpan fixed,
216  ConstResidueSpan movable,
217  PolymerType ptype,
218  SupSelect sel,
219  char altloc='\0',
220  std::vector<int>* ca_offsets=nullptr) {
221  AlignmentResult result = align_sequence_to_polymer(fixed.extract_sequence(),
222  movable, ptype,
223  AlignmentScoring::blosum62());
224  auto it1 = fixed.first_conformer().begin();
225  auto it2 = movable.first_conformer().begin();
226  std::vector<AtomNameElement> used_atoms;
227  bool is_na = is_polynucleotide(ptype);
228  const AtomNameElement* ca_p = nullptr;
229  if (sel == SupSelect::CaP) {
230  used_atoms.push_back({is_na ? "P" : "CA", is_na ? El::P : El::C});
231  } else if (sel == SupSelect::MainChain) {
232  used_atoms = get_mainchain_atoms(ptype);
233  ca_p = &used_atoms[is_na ? 0 : 1];
234  }
235  for (AlignmentResult::Item item : result.cigar) {
236  char op = item.op();
237  for (uint32_t i = 0; i < item.len(); ++i) {
238  int ca_offset = -1;
239  if (op == 'M' && it1->name == it2->name) {
240  if (!used_atoms.empty()) {
241  for (const AtomNameElement& ane : used_atoms) {
242  const Atom* a1 = it1->find_atom(ane.atom_name, altloc, ane.el);
243  const Atom* a2 = it2->find_atom(ane.atom_name, altloc, ane.el);
244  if (a1 && a2) {
245  if (&ane == ca_p)
246  ca_offset = (int)pos1.size();
247  pos1.push_back(a1->pos);
248  pos2.push_back(a2->pos);
249  }
250  }
251  } else {
252  for (const Atom& a1 : it1->atoms)
253  if (a1.altloc_matches(altloc))
254  if (const Atom* a2 = it2->find_atom(a1.name, altloc, a1.element)) {
255  pos1.push_back(a1.pos);
256  pos2.push_back(a2->pos);
257  }
258  }
259  }
260  if (op == 'M' || op == 'I') {
261  ++it1;
262  if (ca_offsets)
263  ca_offsets->push_back(ca_offset);
264  }
265  if (op == 'M' || op == 'D')
266  ++it2;
267  }
268  }
269 }

278 inline SupResult calculate_current_rmsd(ConstResidueSpan fixed,
279  ConstResidueSpan movable,
280  PolymerType ptype,
281  SupSelect sel,
282  char altloc='\0') {
283  std::vector<Position> pos1, pos2;
284  prepare_positions_for_superposition(pos1, pos2, fixed, movable, ptype, sel, altloc);
285  SupResult r;
286  r.count = pos1.size();
287  double sd = 0;
288  for (size_t i = 0; i != pos1.size(); ++i)
289  sd += pos1[i].dist_sq(pos2[i]);
290  r.rmsd = std::sqrt(sd / r.count);
291  return r;
292 }

303 inline SupResult calculate_superposition(ConstResidueSpan fixed,
304  ConstResidueSpan movable,
305  PolymerType ptype,
306  SupSelect sel,
307  int trim_cycles=0,
308  double trim_cutoff=2.0,
309  char altloc='\0') {
310  std::vector<Position> pos1, pos2;
311  prepare_positions_for_superposition(pos1, pos2, fixed, movable, ptype, sel, altloc);
312  const double* weights = nullptr;
313  size_t len = pos1.size();
314  SupResult sr = superpose_positions(pos1.data(), pos2.data(), len, weights);
315 
316  for (int n = 0; n < trim_cycles; ++n) {
317  double max_dist_sq = sq(trim_cutoff * sr.rmsd);
318  size_t p = 0;
319  for (size_t i = 0; i != len; ++i) {
320  Vec3 m2 = sr.transform.apply(pos2[i]);
321  if (m2.dist_sq(pos1[i]) <= max_dist_sq) {
322  if (i != p) {
323  pos1[p] = pos1[i];
324  pos2[p] = pos2[i];
325  }
326  ++p;
327  }
328  }
329  if (p == len)
330  break;
331  len = p;
332  if (len < 3)
333  fail("in calculate_superposition(): only ", std::to_string(len),
334  " atoms after trimming");
335  sr = superpose_positions(pos1.data(), pos2.data(), len, weights);
336  }
337 
338  return sr;
339 }

348 inline std::vector<SupResult> calculate_superpositions_in_moving_window(
349  ConstResidueSpan fixed,
350  ConstResidueSpan movable,
351  PolymerType ptype,
352  double radius=10.0) {
353  const double radius_sq = radius * radius;
354  std::vector<Position> pos1, pos2;
355  char altloc = '\0';
356  SupSelect sel = SupSelect::MainChain;
357  std::vector<int> ca_offsets;
358  prepare_positions_for_superposition(pos1, pos2, fixed, movable, ptype,
359  sel, altloc, &ca_offsets);
360  const double* weights = nullptr;
361  std::vector<SupResult> result;
362  for (int offset : ca_offsets) {
363  if (offset == -1) {
364  result.push_back(SupResult{NAN, 0, {}, {}, {}});
365  continue;
366  }
367  const Position& ca_pos = pos1[offset];
368  int a = offset;
369  while (a > 0 && ca_pos.dist_sq(pos1[a-1]) < radius_sq)
370  --a;
371  int b = offset;
372  while (b+1 < (int)pos1.size() && ca_pos.dist_sq(pos1[b+1]) < radius_sq)
373  ++b;
374  result.push_back(superpose_positions(&pos1[a], &pos2[a], b-a+1, weights));
375  }
376  return result;
377 }
