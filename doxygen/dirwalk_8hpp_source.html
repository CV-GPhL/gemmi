4 // in alphabetical order. Wraps the tinydir library (as we cannot yet

7 // DirWalk<> iterates through all files and directories.
8 // CifWalk yields only cif files (either files that end with .cif or .cif.gz,
9 // or files that look like SF mmCIF files from wwPDB, e.g. r3aaasf.ent.gz).
10 // It's good for traversing a local copy of the wwPDB archive.
11 // PdbWalk: .pdb or .ent (optionally with .gz) except r????sf.ent
12 // CoorFileWalk: .cif, .pdb or .ent (optionally with .gz)
13 // except r????sf.ent and *-sf.cif
14 //
15 // Usage:
16 // for (const std::string& file : gemmi::DirWalk<>(top_dir))
17 // do_something(file);
18 // or
19 // for (const std::string& file : gemmi::CifWalk(top_dir))
20 // do_something(file);
21 // You should also catch std::runtime_error.

114 template<bool FileOnly=true, typename Filter=impl::IsAnyFile>
115 class DirWalk {
116 public:
121  explicit DirWalk(const char* path, char try_pdbid='\0') {
122  if (impl::utf8_tinydir_file_open(&top_, path) != -1)
123  return;
124  if (try_pdbid != '\0' && is_pdb_code(path)) {
125  std::string epath = expand_pdb_code_to_path(path, try_pdbid, true);
126  if (impl::utf8_tinydir_file_open(&top_, epath.c_str()) != -1)
127  return;
128  sys_fail("Cannot open " + epath);
129  }
130  sys_fail("Cannot open " + std::string(path));
131  }
136  explicit DirWalk(const std::string& path, char try_pdbid='\0')
137  : DirWalk(path.c_str(), try_pdbid) {}
140  ~DirWalk() {
141  for (auto& d : dirs_)
142  tinydir_close(&d.second);
143  }
148  void push_dir(size_t cur_pos, const _tinydir_char_t* path) {
149  dirs_.emplace_back();
150  dirs_.back().first = cur_pos;
151  if (tinydir_open_sorted(&dirs_.back().second, path) == -1)
152  sys_fail("Cannot open directory " + as_utf8(path));
153  }
157  size_t pop_dir() {
158  assert(!dirs_.empty());
159  size_t old_pos = dirs_.back().first;
160  tinydir_close(&dirs_.back().second);
161  dirs_.pop_back();
162  return old_pos;
163  }

167  struct Iter {
168  DirWalk& walk;
169  size_t cur;
170 
174  const tinydir_dir& get_dir() const { return walk.dirs_.back().second; }
175 
179  const tinydir_file& get() const {
180  if (walk.dirs_.empty())
181  return walk.top_;
182  assert(cur < get_dir().n_files);
183  return get_dir()._files[cur];
184  }
185 
189  std::string operator*() const { return as_utf8(get().path); }
190 
195  bool is_special(const _tinydir_char_t* name) const {
196  return name[0] == '.' && (name[1] == '\0' ||
197  (name[1] == '.' && name[2] == '\0'));
198  }
199 
203  size_t depth() const { return walk.dirs_.size(); }
204 
207  void next() { // depth first
208  const tinydir_file& tf = get();
209  if (tf.is_dir) {
210  walk.push_dir(cur, tf.path);
211  cur = 0;
212  } else {
213  cur++;
214  }
215  while (!walk.dirs_.empty()) {
216  if (cur == get_dir().n_files)
217  cur = walk.pop_dir() + 1;
218  else if (is_special(get_dir()._files[cur].name))
219  cur++;
220  else
221  break;
222  }
223  }
224 
227  void operator++() {
228  for (;;) {
229  next();
230  const tinydir_file& f = get();
231  if ((!FileOnly && f.is_dir)
232  || (!f.is_dir && walk.filter.check(as_utf8(f.name)))
233  || walk.is_single_file()
234  || (depth() == 0 && cur == 1))
235  break;
236  }
237  }
238 
243  bool operator==(const Iter& o) const { return depth()==0 && cur == o.cur; }
248  bool operator!=(const Iter& o) const { return !operator==(o); }
249  };
250 
254  Iter begin() {
255  Iter it{*this, 0};
256  if (FileOnly && !is_single_file()) // i.e. the top item is a directory
257  ++it;
258  return it;
259  }
260 
264  Iter end() { return Iter{*this, 1}; }
268  bool is_single_file() { return !top_.is_dir; }
269 
270 private:
271  friend struct Iter;
272  tinydir_file top_;
273  std::vector<std::pair<size_t, tinydir_dir>> dirs_;
274 protected:
275  Filter filter;
276 };
