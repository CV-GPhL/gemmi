150  std::tuple<const ChemLink*, bool, const ChemComp::Aliasing*, const ChemComp::Aliasing*>
151  match_link(const Residue& res1, const std::string& atom1, char alt1,
152  const Residue& res2, const std::string& atom2, char alt2,
153  double min_bond_sq=0) const {
154  const ChemLink* best_link = nullptr;
155  bool inverted = false;
156  const ChemComp::Aliasing* aliasing1 = nullptr;
157  const ChemComp::Aliasing* aliasing2 = nullptr;
158  const ChemComp::Aliasing* aliasing1_final = nullptr;
159  const ChemComp::Aliasing* aliasing2_final = nullptr;
160  int best_score = -1000;
161  for (const auto& ml : links) {
162  const ChemLink& link = ml.second;
163  if (link.rt.bonds.empty() || starts_with(link.name, "auto-"))
164  continue;
165  // for now we don't have link definitions with >1 bonds
166  const Restraints::Bond& bond = link.rt.bonds[0];
167  if (sq(bond.value) < min_bond_sq)
168  continue;
169  if (link_side_matches_residue(link.side1, res1.name, &aliasing1) &&
170  link_side_matches_residue(link.side2, res2.name, &aliasing2) &&
171  atom_match_with_alias(bond.id1.atom, atom1, aliasing1) &&
172  atom_match_with_alias(bond.id2.atom, atom2, aliasing2)) {
173  int score = link.calculate_score(res1, &res2, alt1, alt2, aliasing1, aliasing2);
174  if (score > best_score) {
175  best_link = &link;
176  best_score = score;
177  aliasing1_final = aliasing1;
178  aliasing2_final = aliasing2;
179  inverted = false;
180  }
181  }
182  if (link_side_matches_residue(link.side1, res2.name, &aliasing2) &&
183  link_side_matches_residue(link.side2, res1.name, &aliasing1) &&
184  atom_match_with_alias(bond.id1.atom, atom2, aliasing2) &&
185  atom_match_with_alias(bond.id2.atom, atom1, aliasing1)) {
186  // NOLINTNEXTLINE(readability-suspicious-call-argument)
187  int score = link.calculate_score(res2, &res1, alt2, alt1, aliasing2, aliasing1);
188  if (score > best_score) {
189  best_link = &link;
190  best_score = score;
191  aliasing1_final = aliasing1;
192  aliasing2_final = aliasing2;
193  inverted = true;
194  }
195  }
196  }
197  return std::make_tuple(best_link, inverted, aliasing1_final, aliasing2_final);
198  }

202  void add_monomer_if_present(const cif::Block& block) {
203  if (block.has_tag("_chem_comp_atom.atom_id")) {
204  ChemComp cc = make_chemcomp_from_block(block);

212  }

220  bool link_side_matches_residue(const ChemLink::Side& side,
221  const std::string& res_name,
222  ChemComp::Aliasing const** aliasing) const {
223  assert(aliasing);
224  *aliasing = nullptr;
225  if (!side.comp.empty())
226  return side.comp == res_name;
227  auto it = monomers.find(res_name);
228  if (it != monomers.end()) {
229  if (side.matches_group(it->second.group))
230  return true;
231  for (const ChemComp::Aliasing& a : it->second.aliases)
232  if (side.matches_group(a.group)) {
233  *aliasing = &a;
234  return true;
235  }
236  }
237  return false;
238  }

296 inline MonLib read_monomer_lib(const std::string& monomer_dir,
297  const std::vector<std::string>& resnames,
298  const std::string& libin="",
299  bool ignore_missing=false) {
300  MonLib monlib;
301  if (!libin.empty())
302  monlib.read_monomer_cif(libin);
303  std::string error;
304  Logger logger;
305  if (!ignore_missing)
306  logger.callback = [&error](const std::string& s) { cat_to(error, s, '\n'); };
307  bool ok = monlib.read_monomer_lib(monomer_dir, resnames, logger);
308  if (!ignore_missing && !ok) {

310  fail(error);
311  }
312  return monlib;
