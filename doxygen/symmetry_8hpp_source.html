147  int det_rot() const {
148  return rot[0][0] * (rot[1][1] * rot[2][2] - rot[1][2] * rot[2][1])

160  int rot_type() const {
161  int det = det_rot();
162  int tr_den = rot[0][0] + rot[1][1] + rot[2][2];
163  int tr = tr_den / DEN;
164  const int table[] = {0, 0, 2, 3, 4, 6, 1};
165  if (std::abs(det) == DEN * DEN * DEN && tr * DEN == tr_den && std::abs(tr) <= 3)
166  return det > 0 ? table[3 + tr] : -table[3 - tr];
167  return 0;

174  Op combine(const Op& b) const {
175  if (is_hkl() != b.is_hkl())
176  fail("can't combine real- and reciprocal-space Op");
177  Op r;
178  for (int i = 0; i != 3; ++i) {
179  r.tran[i] = tran[i] * Op::DEN;
180  for (int j = 0; j != 3; ++j) {
181  r.rot[i][j] = (rot[i][0] * b.rot[0][j] +
182  rot[i][1] * b.rot[1][j] +
183  rot[i][2] * b.rot[2][j]) / Op::DEN;
184  r.tran[i] += rot[i][j] * b.tran[j];
185  }
186  r.tran[i] /= Op::DEN;
187  }
188  r.notation = notation;
189  return r;
190  }

298 inline Op Op::inverse() const {
299  int detr = det_rot();
300  if (detr == 0)
301  fail("cannot invert matrix: " + Op{rot, {0,0,0}, notation}.triplet());
302  int d2 = Op::DEN * Op::DEN;
303  Op inv;
304  inv.rot[0][0] = d2 * (rot[1][1] * rot[2][2] - rot[2][1] * rot[1][2]) / detr;
305  inv.rot[0][1] = d2 * (rot[0][2] * rot[2][1] - rot[0][1] * rot[2][2]) / detr;
306  inv.rot[0][2] = d2 * (rot[0][1] * rot[1][2] - rot[0][2] * rot[1][1]) / detr;
307  inv.rot[1][0] = d2 * (rot[1][2] * rot[2][0] - rot[1][0] * rot[2][2]) / detr;
308  inv.rot[1][1] = d2 * (rot[0][0] * rot[2][2] - rot[0][2] * rot[2][0]) / detr;
309  inv.rot[1][2] = d2 * (rot[1][0] * rot[0][2] - rot[0][0] * rot[1][2]) / detr;
310  inv.rot[2][0] = d2 * (rot[1][0] * rot[2][1] - rot[2][0] * rot[1][1]) / detr;
311  inv.rot[2][1] = d2 * (rot[2][0] * rot[0][1] - rot[0][0] * rot[2][1]) / detr;
312  inv.rot[2][2] = d2 * (rot[0][0] * rot[1][1] - rot[1][0] * rot[0][1]) / detr;
313  for (int i = 0; i != 3; ++i)
314  inv.tran[i] = (-tran[0] * inv.rot[i][0]
315  -tran[1] * inv.rot[i][1]
316  -tran[2] * inv.rot[i][2]) / Op::DEN;
317  inv.notation = notation;
318  return inv;
319 }

349 inline std::vector<Op::Tran> centring_vectors(char centring_type) {
350  constexpr int h = Op::DEN / 2;
351  constexpr int t = Op::DEN / 3;
352  constexpr int d = 2 * t;
353  // note: find_centering() depends on the order of operations in vector
354  switch (centring_type & ~0x20) {
355  case 'P': return {{0, 0, 0}};
356  case 'A': return {{0, 0, 0}, {0, h, h}};
357  case 'B': return {{0, 0, 0}, {h, 0, h}};
358  case 'C': return {{0, 0, 0}, {h, h, 0}};
359  case 'I': return {{0, 0, 0}, {h, h, h}};
360  case 'R': return {{0, 0, 0}, {d, t, t}, {t, d, d}};
361  // hall_symbols.html has no H, ITfC 2010 has no S and T
362  case 'H': return {{0, 0, 0}, {d, t, 0}, {t, d, 0}};
363  case 'S': return {{0, 0, 0}, {t, t, d}, {d, t, d}};
364  case 'T': return {{0, 0, 0}, {t, d, t}, {d, t, d}};
365  case 'F': return {{0, 0, 0}, {0, h, h}, {h, 0, h}, {h, h, 0}};
366  default: fail("not a centring type: ", centring_type);
367  }
368 }

412  char find_centering() const {
413  if (cen_ops.size() == 1 && cen_ops[0] == Op::Tran{0, 0, 0})
414  return 'P';

506  void change_basis_impl(const Op& cob, const Op& inv) {
507  if (sym_ops.empty() || cen_ops.empty())
508  return;
509 
510  // Apply change-of-basis to sym_ops.
511  // Ignore the first item in sym_ops -- it's identity.
512  for (auto op = sym_ops.begin() + 1; op != sym_ops.end(); ++op)
513  *op = cob.combine(*op).combine(inv).wrap();
514 
515  // The number of centering vectors may be different.
516  // As an ad-hoc method (not proved to be robust) add lattice points
517  // from a super-cell.
518  int idet = inv.det_rot() / (Op::DEN * Op::DEN * Op::DEN);
519  if (idet > 1) {
520  std::vector<Op::Tran> new_cen_ops;
521  new_cen_ops.reserve(cen_ops.size() * idet * idet * idet);
522  for (int i = 0; i < idet; ++i)
523  for (int j = 0; j < idet; ++j)
524  for (int k = 0; k < idet; ++k)
525  for (Op::Tran& cen : cen_ops)
526  new_cen_ops.push_back({i * Op::DEN + cen[0],
527  j * Op::DEN + cen[1],
528  k * Op::DEN + cen[2]});
529  cen_ops.swap(new_cen_ops);
530  }
531 
532  // Apply change-of-basis to centering vectors
533  Op cvec = Op::identity();
534  for (auto tr = cen_ops.begin() + 1; tr != cen_ops.end(); ++tr) {
535  cvec.tran = *tr;
536  *tr = cob.combine(cvec).combine(inv).wrap().tran;
537  }
538 
539  // Remove redundant centering vectors.
540  for (int i = static_cast<int>(cen_ops.size()) - 1; i > 0; --i)
541  for (int j = i - 1; j >= 0; --j)
542  if (cen_ops[i] == cen_ops[j]) {
543  cen_ops.erase(cen_ops.begin() + i);
544  break;
545  }
546  }

680 inline void GroupOps::add_missing_elements() {
681  // We always keep identity as sym_ops[0].
682  if (sym_ops.empty() || sym_ops[0] != Op::identity())
683  fail("oops");
684  if (sym_ops.size() == 1)
685  return;
686  constexpr size_t max_size = 1024;
687  // Below we assume that all centring vectors are already known (in cen_ops)
688  // so when checking for a new element we compare only the 3x3 matrix.
689  // Dimino's algorithm. https://physics.stackexchange.com/a/351400/95713
690  std::vector<Op> gen(sym_ops.begin() + 1, sym_ops.end());
691  sym_ops.resize(2);
692  const Op::Rot idrot = Op::identity().rot;
693  for (Op g = sym_ops[1] * sym_ops[1]; g.rot != idrot; g *= sym_ops[1]) {
694  sym_ops.push_back(g);
695  if (sym_ops.size() > max_size)
696  fail("Too many elements in the group - bad generators");
697  }
698  // the rest is in separate function b/c it's reused in twin.hpp
699  add_missing_elements_part2(gen, max_size, false);
700 }

702 inline void GroupOps::add_missing_elements_part2(const std::vector<Op>& gen,
703  size_t max_size, bool ignore_bad_gen) {
704  for (size_t i = 1; i < gen.size(); ++i) {
705  std::vector<Op> coset_repr(1, Op::identity());
706  size_t init_size = sym_ops.size();
707  for (;;) {
708  size_t len = coset_repr.size();
709  for (size_t j = 0; j != len; ++j) {
710  for (size_t n = 0; n != i + 1; ++n) {
711  Op sg = gen[n] * coset_repr[j];
712  if (find_by_rotation(sg.rot) == nullptr) {
713  sym_ops.push_back(sg);
714  for (size_t k = 1; k != init_size; ++k)
715  sym_ops.push_back(sg * sym_ops[k]);
716  coset_repr.push_back(sg);
717  }
718  }
719  }
720  if (len == coset_repr.size())
721  break;
722  if (sym_ops.size() > max_size) {
723  if (!ignore_bad_gen)
724  fail("Too many elements in the group - bad generators");
725  // ignore this generator and continue with the next one
726  sym_ops.resize(init_size);
727  break;
728  }
729  }
730  }

747  }
