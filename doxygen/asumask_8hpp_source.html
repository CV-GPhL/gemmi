188  if (grid.data[idx] == 0) {
189  grid.data[idx] = 1;
190  for (const GridOp& op : ops) {
191  Point t = op.apply(u, v, w);

195  }

212  std::vector<AsuBrick> possible_bricks;
213  for (int a : allowed_sizes)
214  for (int b : allowed_sizes)
215  for (int c : allowed_sizes) {

294 std::vector<V> get_asu_mask(const GridMeta& grid) {
295  std::vector<V> mask(grid.point_count(), 2);
296  std::vector<GridOp> ops = grid.get_scaled_ops_except_id();
297  auto end = find_asu_brick(grid.spacegroup).uvw_end(grid);
298  for (int w = 0; w < end[2]; ++w)
299  for (int v = 0; v < end[1]; ++v)
300  for (int u = 0; u < end[0]; ++u) {
301  size_t idx = grid.index_q(u, v, w);
302  if (mask[idx] == 2) {
303  mask[idx] = 0;
304  for (const GridOp& op : ops) {
305  std::array<int, 3> t = op.apply(u, v, w);
306  size_t mate_idx = grid.index_n(t[0], t[1], t[2]);
307  // grid point can be on special position
308  if (mate_idx != idx)
309  mask[mate_idx] = 1;
310  }
311  }
312  }
313  if (std::find(mask.begin(), mask.end(), 2) != mask.end())
314  fail("get_asu_mask(): internal error");
315  return mask;
316 }

335 inline std::pair<int, int> trim_false_values(const std::vector<bool>& vec) {
336  const int n = (int) vec.size();
337  assert(n != 0);
338  std::pair<int, int> span{n, n}; // return value for all-true vector
339  int max_trim = 0;
340  // first calculate the wrapped span (initial + final non-zero values)
341  if (!vec[0] || !vec[n-1]) {
342  // determine trailing-false length and store it in span.first
343  while (span.first != 0 && !vec[span.first-1])
344  --span.first;
345  if (span.first == 0) // all-false vector
346  return span; // i.e. {0,n}
347  // determine leading-false length and store it in span.second
348  span.second = 0;
349  while (span.second != n && !vec[span.second])
350  ++span.second;
351  max_trim = span.second + (n - span.first);
352  }
353  for (int start = 0; ;) {
354  for (;;) {
355  if (start == n)
356  return span;
357  if (!vec[start])
358  break;
359  ++start;
360  }
361  int end = start + 1;
362  while (end != n && !vec[end])
363  ++end;
364  if (end - start > max_trim) {
365  max_trim = end - start;
366  span.first = start;
367  span.second = end;
368  }
369  start = end;
370  }
371  unreachable();
372 }

381 template<typename T>
382 Box<Fractional> get_nonzero_extent(const GridBase<T>& grid) {
383  grid.check_not_empty();
384  std::vector<bool> nonzero[3];
385  nonzero[0].resize(grid.nu, false);
386  nonzero[1].resize(grid.nv, false);
387  nonzero[2].resize(grid.nw, false);
388  size_t idx = 0;
389  for (int w = 0; w != grid.nw; ++w)
390  for (int v = 0; v != grid.nv; ++v)
391  for (int u = 0; u != grid.nu; ++u, ++idx) {
392  T val = grid.data[idx];
393  if (!(impl::is_nan(val) || val == 0)) {
394  nonzero[0][u] = true;
395  nonzero[1][v] = true;
396  nonzero[2][w] = true;
397  }
398  }
399  Box<Fractional> box;
400  for (int i = 0; i < 3; ++i) {
401  auto span = impl::trim_false_values(nonzero[i]);
402  double inv_n = 1.0 / nonzero[i].size();
403  box.minimum.at(i) = (span.second - 0.5) * inv_n - int(span.second >= span.first);
404  box.maximum.at(i) = (span.first - 0.5) * inv_n;
405  }
406  return box;
407 }
