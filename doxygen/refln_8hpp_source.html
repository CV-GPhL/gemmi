179  std::vector<double> make_1_d2_vector() const {
180  if (!cell.is_crystal() || cell.a <= 0)
181  fail("Unit cell is not known");
182  auto hkl_idx = get_hkl_column_indices();
183  std::vector<double> r(default_loop->length());
184  for (size_t j = 0, n = 0; j != r.size(); j++, n += default_loop->width()) {
185  Miller hkl;
186  for (int i = 0; i != 3; ++i)
187  hkl[i] = cif::as_int(default_loop->values[n + hkl_idx[i]]);
188  r[j] = cell.calculate_1_d2(hkl);
189  }
190  return r;
191  }

210 std::vector<ReflnBlock> as_refln_blocks(std::vector<cif::Block>&& blocks) {
211  std::vector<ReflnBlock> rvec;
212  rvec.reserve(blocks.size());
213  for (cif::Block& block : blocks)
214  rvec.emplace_back(std::move(block));
215  blocks.clear();
216  // Some blocks miss space group or unit cell, try to fill it in.
217  const SpaceGroup* first_sg = nullptr;
218  const UnitCell* first_cell = nullptr;
219  for (ReflnBlock& rblock : rvec) {
220  if (!first_sg)
221  first_sg = rblock.spacegroup;
222  else if (!rblock.spacegroup)
223  rblock.spacegroup = first_sg;
224  if (rblock.cell.is_crystal()) {
225  if (!first_cell)
226  first_cell = &rblock.cell;
227  } else if (first_cell) {
228  rblock.cell = *first_cell;
229  }
230  }
231  return rvec;
232 }

243 inline ReflnBlock get_refln_block(std::vector<cif::Block>&& blocks,
244  const std::vector<std::string>& labels,
245  const char* block_name=nullptr) {
246  const SpaceGroup* first_sg = nullptr;
247  bool has_block = false;
248  for (cif::Block& block : blocks) {
249  if (!first_sg)
250  if (const std::string* hm = impl::find_spacegroup_hm_value(block)) {
251  first_sg = find_spacegroup_by_name(cif::as_string(*hm));
252  if (first_sg && first_sg->ext == 'H') {
253  UnitCell cell;
254  impl::set_cell_from_mmcif(block, cell);
255  first_sg = find_spacegroup_by_name(cif::as_string(*hm), cell.alpha, cell.gamma);
256  }
257  }
258  if (!block_name || block.name == block_name) {
259  has_block = true;
260  if (cif::Loop* loop = block.find_loop("_refln.index_h").get_loop())
261  if (std::all_of(labels.begin(), labels.end(),
262  [&](const std::string& s) { return s.empty() || loop->has_tag("_refln."+s); })) {
263  ReflnBlock rblock(std::move(block));
264  if (!rblock.spacegroup && first_sg)
265  rblock.spacegroup = first_sg;
266  return rblock;
267  }
268  }
269  }
270  if (block_name && !has_block)
271  fail("Required block not found in SF-mmCIF file.");
272  fail("Tags not found in SF-mmCIF file: _refln.", join_str(labels, ", _refln."));
273 }
