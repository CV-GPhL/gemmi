305  void calculate_properties() {
306  // ensure exact values for right angles
307  double cos_alpha = alpha == 90. ? 0. : std::cos(rad(alpha));
308  double cos_beta = beta == 90. ? 0. : std::cos(rad(beta));
309  double cos_gamma = gamma == 90. ? 0. : std::cos(rad(gamma));
310  double sin_alpha = alpha == 90. ? 1. : std::sin(rad(alpha));
311  double sin_beta = beta == 90. ? 1. : std::sin(rad(beta));
312  double sin_gamma = gamma == 90. ? 1. : std::sin(rad(gamma));
313  if (sin_alpha == 0 || sin_beta == 0 || sin_gamma == 0)
314  fail("Impossible angle - N*180deg.");
315 
316  // volume - formula from Giacovazzo p.62
317  volume = a * b * c * std::sqrt(1 - cos_alpha * cos_alpha
318  - cos_beta * cos_beta - cos_gamma * cos_gamma
319  + 2 * cos_alpha * cos_beta * cos_gamma);
320 
321  // reciprocal parameters a*, b*, ... (Giacovazzo, p. 64)
322  ar = b * c * sin_alpha / volume;
323  br = a * c * sin_beta / volume;
324  cr = a * b * sin_gamma / volume;
325  double cos_alphar_sin_beta = (cos_beta * cos_gamma - cos_alpha) / sin_gamma;
326  cos_alphar = cos_alphar_sin_beta / sin_beta;
327  //cos_alphar = (cos_beta * cos_gamma - cos_alpha) / (sin_beta * sin_gamma);
328  cos_betar = (cos_alpha * cos_gamma - cos_beta) / (sin_alpha * sin_gamma);
329  cos_gammar = (cos_alpha * cos_beta - cos_gamma) / (sin_alpha * sin_beta);
330 
331  if (explicit_matrices)
332  return;
333 
334  // The orthogonalization matrix we use is described in ITfC B p.262:
335  // "An alternative mode of orthogonalization, used by the Protein
336  // Data Bank and most programs, is to align the a1 axis of the unit
337  // cell with the Cartesian X_1 axis, and to align the a*_3 axis with the
338  // Cartesian X_3 axis."
339  double sin_alphar = std::sqrt(1.0 - cos_alphar * cos_alphar);
340  orth.mat = {a, b * cos_gamma, c * cos_beta,
341  0., b * sin_gamma, -c * cos_alphar_sin_beta,
342  0., 0. , c * sin_beta * sin_alphar};
343  orth.vec = {0., 0., 0.};
344 
345  double o12 = -cos_gamma / (sin_gamma * a);
346  double o13 = -(cos_gamma * cos_alphar_sin_beta + cos_beta * sin_gamma)
347  / (sin_alphar * sin_beta * sin_gamma * a);
348  double o23 = cos_alphar / (sin_alphar * sin_gamma * b);
349  frac.mat = {1 / a, o12, o13,
350  0., 1 / orth.mat[1][1], o23,
351  0., 0., 1 / orth.mat[2][2]};
352  frac.vec = {0., 0., 0.};
353  }

385  double calculate_u_eq(const SMat33<double>& ani) const {
386  double aar = a * ar;
387  double bbr = b * br;
388  double ccr = c * cr;
389  // it could be optimized using orth.mat[0][1] and orth.mat[0][2]
390  double cos_beta = beta == 90. ? 0. : std::cos(rad(beta));
391  double cos_gamma = gamma == 90. ? 0. : std::cos(rad(gamma));
392  return 1/3. * (sq(aar) * ani.u11 + sq(bbr) * ani.u22 + sq(ccr) * ani.u33 +
393  2 * (aar * bbr * cos_gamma * ani.u12 +
394  aar * ccr * cos_beta * ani.u13 +
395  bbr * ccr * cos_alpha() * ani.u23));
396  }

694  NearestImage find_nearest_image(const Position& ref, const Position& pos, Asu asu) const {
695  NearestImage image;
696  if (asu == Asu::Different)
697  image.dist_sq = INFINITY;
698  else
699  image.dist_sq = ref.dist_sq(pos);
700  if (asu == Asu::Same)
701  return image;
702  Fractional fpos = fractionalize(pos);
703  Fractional fref = fractionalize(ref);
704  search_pbc_images(fpos - fref, image);
705  if (asu == Asu::Different &&
706  image.pbc_shift[0] == 0 && image.pbc_shift[1] == 0 && image.pbc_shift[2] == 0)
707  image.dist_sq = INFINITY;
708  for (int n = 0; n != static_cast<int>(images.size()); ++n)
709  if (search_pbc_images(images[n].apply(fpos) - fref, image))
710  image.sym_idx = n + 1;
711  return image;
712  }
