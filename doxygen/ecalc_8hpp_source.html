24 std::vector<double> calculate_amplitude_normalizers(const DataProxy& data, int fcol_idx,
25  const Binner& binner) {
26  struct CountAndSum {
27  int n = 0;
28  double sum = 0.;
29  };
30  int nreflections = data.size() / data.stride();
31  std::vector<double> multipliers(nreflections, NAN);
32  if (data.spacegroup() == nullptr)
33  gemmi::fail("unknown space group in the data file");
34  GroupOps gops = data.spacegroup()->operations();
35  std::vector<double> inv_d2(multipliers.size());
36  for (size_t i = 0, n = 0; n < data.size(); n += data.stride(), ++i)
37  inv_d2[i] = data.unit_cell().calculate_1_d2(data.get_hkl(n));
38  std::vector<int> bin_index = binner.get_bins_from_1_d2(inv_d2);
39  std::vector<CountAndSum> stats(binner.size());
40  for (size_t i = 0, n = 0; n < data.size(); n += data.stride(), i++) {
41  Miller hkl = data.get_hkl(n);
42  double f = data.get_num(n + fcol_idx);
43  if (!std::isnan(f)) {
44  int epsilon = gops.epsilon_factor(hkl);
45  double inv_epsilon = 1.0 / epsilon;
46  double f2 = f * f * inv_epsilon;
47  multipliers[i] = std::sqrt(inv_epsilon);
48  CountAndSum& cs = stats[bin_index[i]];
49  cs.n++;
50  cs.sum += f2;
51  }
52  }
53 
54  // simple smoothing with kernel [0.75 1 0.75]
55  std::vector<double> smoothed(stats.size());
56  {
57  const double k = 0.75;
58  smoothed[0] = (stats[0].sum + k * stats[1].sum) / (stats[0].n + k * stats[1].n);
59  size_t n = stats.size() - 1;
60  for (size_t i = 1; i < n; ++i)
61  smoothed[i] = (stats[i].sum + k * (stats[i-1].sum + stats[i+1].sum))
62  / (stats[i].n + k * (stats[i-1].n + stats[i+1].n));
63  smoothed[n] = (stats[n].sum + k * stats[n-1].sum) / (stats[n].n + k * stats[n-1].n);
64  }

90  if (x > binner.mids.front() && x < binner.mids.back()) {
91  // linear interpolation in 1/d^2
92  if (x > binner.mids[bin])
93  ++bin;
94  double x0 = binner.mids[bin - 1];
95  double x1 = binner.mids[bin];
96  double y0 = smoothed[bin - 1];
97  double y1 = smoothed[bin];
98  assert(x0 <= x && x <= x1);
99  rms = y0 + (x - x0) * (y1 - y0) / (x1 - x0);
100  }

102  }
