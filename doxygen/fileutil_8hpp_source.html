28 inline std::string path_basename(const std::string& path,
29  std::initializer_list<const char*> exts) {
30  size_t pos = path.find_last_of("\\/");
31  std::string basename = pos == std::string::npos ? path : path.substr(pos + 1);
32  for (const char* ext : exts) {
33  size_t len = std::strlen(ext);
34  if (basename.size() > len &&
35  basename.compare(basename.length() - len, len, ext, len) == 0)
36  basename.resize(basename.length() - len);
37  }
38  return basename;
39 }
