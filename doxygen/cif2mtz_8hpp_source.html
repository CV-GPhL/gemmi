39 inline bool possible_old_style(const ReflnBlock& rb, DataType data_type) {
40  if (rb.refln_loop == nullptr)
41  return false;
42  for (const std::string& tag : rb.refln_loop->tags) {
43  if (tag.size() < 7 + 6)
44  return false;
45  int tag_id = ialpha4_id(tag.c_str() + 7);
46  if (tag_id != ialpha4_id("inde") && // index_[hkl]
47  tag_id != ialpha4_id("wave") && // wavelength_id
48  tag_id != ialpha4_id("crys") && // crystal_id
49  tag_id != ialpha4_id("scal") && // scale_group_code
50  tag_id != ialpha4_id("stat") && // status
51  tag_id != ialpha4_id("inte") && // intensity_meas, intensity_sigma
52  (data_type == DataType::Unmerged ||
53  (tag_id != ialpha4_id("F_me") && // F_meas_au, F_meas_sigma_au
54  tag != "_refln.pdbx_r_free_flag")))
55  return false;
56  }
57  return true;
58 }

76 inline cif::Loop transcript_old_anomalous_to_standard(const cif::Loop& loop,
77  const SpaceGroup* sg) {
78  std::vector<int> positions;
79  positions.reserve(13); // usually less, but it doesn't matter
80  for (const char* tag : {"_refln.index_h", "_refln.index_k", "_refln.index_l"}) {
81  int pos = loop.find_tag_lc(tag);
82  if (pos == -1)
83  fail("while reading old anomalous: _refln.index_{h,k,l} not found");
84  positions.push_back(pos);
85  }
86  for (const char* tag : {"_refln.status", "_refln.pdbx_r_free_flag"}) {
87  int pos = loop.find_tag_lc(tag);
88  if (pos != -1)
89  positions.push_back(pos);
90  }
91 
92  cif::Loop ret;
93  ret.tags.reserve(positions.size());
94  for (int p : positions)
95  ret.tags.push_back(loop.tags[p]);
96  const char* old_labels[2][2] = {
97  {"_refln.F_meas_au", "_refln.F_meas_sigma_au"},
98  {"_refln.intensity_meas", "_refln.intensity_sigma"}
99  };
100  const char* new_labels[2][4] = {
101  {"_refln.pdbx_F_plus", "_refln.pdbx_F_plus_sigma",
102  "_refln.pdbx_F_minus", "_refln.pdbx_F_minus_sigma"},
103  {"_refln.pdbx_I_plus", "_refln.pdbx_I_plus_sigma",
104  "_refln.pdbx_I_minus", "_refln.pdbx_I_minus_sigma"}
105  };
106  size_t common_tags = positions.size();
107  for (int n = 0; n < 2; ++n) {
108  int idx = loop.find_tag(old_labels[n][0]);
109  if (idx >= 0 && idx+1 < (int)loop.width() && loop.tags[idx+1] == old_labels[n][1]) {
110  positions.insert(positions.end(), {idx, idx+1, idx, idx+1});
111  ret.tags.insert(ret.tags.end(), new_labels[n], new_labels[n] + 4);
112  }
113  }
114  if (common_tags == positions.size())
115  fail("while reading old anomalous: _refln has neither F_meas_au nor intensity_meas");
116 
117  ret.values.reserve(loop.length() * ret.width()); // upper bound
118  std::unordered_map<Miller, std::string*, MillerHash> seen;
119  if (!sg)
120  sg = &get_spacegroup_p1();
121  ReciprocalAsu asu(sg);
122  GroupOps gops = sg->operations();
123  for (size_t i = 0; i < loop.values.size(); i += loop.width()) {
124  const std::string* row = &loop.values[i];
125  Miller hkl;
126  for (size_t j = 0; j < 3; ++j)
127  hkl[j] = cif::as_int(row[positions[j]]);
128  auto hkl_sign = asu.to_asu_sign(hkl, gops);
129  // pointers don't change, .reserve() above prevents re-allocations
130  std::string* new_row = ret.values.data() + ret.values.size();
131  auto r = seen.emplace(hkl_sign.first, new_row);
132  bool sign = hkl_sign.second;
133  if (r.second) { // adding a new row
134  for (int p : positions)
135  ret.values.push_back(row[p]);
136  // Don't move hkl to asu here, only change the sign if F- is before F+.
137  if (!sign) // negative sign
138  for (int j = 0; j < 3; ++j)
139  new_row[j] = std::to_string(-hkl[j]);
140  size_t first_absent = common_tags + (sign ? 2 : 0);
141  for (size_t j = first_absent; j < ret.width(); j += 4) {
142  new_row[j] = ".";
143  new_row[j+1] = ".";
144  }
145  } else { // modifying existing row
146  std::string* modified_row = r.first->second;
147  if (sign) // positive sign - this hkl might be better
148  for (int j = 0; j < 3; ++j)
149  modified_row[j] = row[positions[j]];
150  // if a status or free flag value differs, set it to null
151  for (size_t j = 3; j < common_tags; ++j)
152  if (modified_row[j] != row[positions[j]])
153  modified_row[j] = ".";
154  for (size_t j = common_tags + (sign ? 0 : 2); j < ret.width(); j += 4) {
155  modified_row[j] = row[positions[j]];
156  modified_row[j+1] = row[positions[j+1]];
157  }
158  }
159  }
160 
161  return ret;
162 }

257  Entry(const std::string& line) {
258  std::vector<std::string> tokens;
259  tokens.reserve(4);
260  split_str_into_multi(line, " \t\r\n", tokens);
261  if (tokens.size() != 4 && tokens.size() != 5)
262  fail("line should have 4 or 5 words: " + line);
263  if (tokens[2].size() != 1 || tokens[3].size() != 1 ||
264  (tokens[3][0] != '0' && tokens[3][0] != '1'))
265  fail("incorrect line: " + line);
266  refln_tag = tokens[0];
267  col_label = tokens[1];
268  col_type = tokens[2][0];
269  dataset_id = tokens[3][0] - '0';
270  // for compatibility with older spec
271  if (col_type == 's' && tokens.size() == 4) {
272  col_type = 'I';
273  tokens.push_back("o=1,f=0");
274  }
275  if (tokens.size() == 5) {
276  std::vector<std::string> items = split_str(tokens[4], ',');
277  code_to_number.reserve(items.size());
278  for (const std::string& item : items) {
279  size_t pos = item.find('=');
280  if (pos == std::string::npos)
281  fail("wrong mapping (", item, ") in: ", line);
282  float f;
283  auto result = fast_float::from_chars(item.c_str() + pos + 1,
284  item.c_str() + item.size(), f);
285  if (result.ec != std::errc())
286  fail("failed to parse value in ", item, " in: ", line);
287  code_to_number.emplace_back(item.substr(0, pos), f);
288  }
289  }
290  }

299  float translate_code_to_number(const std::string& v) const {
300  if (v.size() == 1) {
301  for (const auto& c2n : code_to_number)
302  if (c2n.first.size() == 1 && c2n.first[0] == v[0])
303  return c2n.second;
304  } else {
305  std::string s = cif::as_string(v);
306  for (const auto& c2n : code_to_number)
307  if (c2n.first == s)
308  return c2n.second;
309  }
310  return NAN;
311  }

339  Mtz convert_block_to_mtz(const ReflnBlock& rb, Logger& logger) const {
340  Mtz mtz;
341  mtz.title = title.empty() ? "Converted from mmCIF block " + rb.block.name : title;
342  if (!history.empty()) {
343  mtz.history.reserve(mtz.history.size() + history.size());
344  mtz.history.insert(mtz.history.end(), history.begin(), history.end());
345  }
346  mtz.cell = rb.cell;
347  mtz.spacegroup = rb.spacegroup;
348  mtz.add_dataset("HKL_base");
349 
350  const cif::Loop* loop = rb.refln_loop ? rb.refln_loop : rb.diffrn_refln_loop;
351  if (!loop)
352  fail("_refln category not found in mmCIF block: " + rb.block.name);
353  bool unmerged = force_unmerged || !rb.refln_loop;
354 
355  if (!unmerged) {
356  Mtz::Dataset& ds = mtz.add_dataset("unknown");
357  if (!std::isnan(wavelength))
358  ds.wavelength = wavelength;
359  else if (rb.wavelength_count > 1)
360  logger.note("ignoring wavelengths, ", rb.wavelength_count,
361  " are present in block ", rb.block.name, '.');
362  else
363  ds.wavelength = rb.wavelength;
364  }
365 
366  logger.level<7>("Searching tags with known MTZ equivalents ...");
367  std::vector<int> indices;
368  std::vector<const Entry*> entries; // used for code_to_number only
369  std::string tag = loop->tags[0];
370  const size_t tag_offset = rb.tag_offset();
371  // NOLINTNEXTLINE(readability-suspicious-call-argument): clang-tidy finds npos suspicious
372  auto set_tag = [&](const std::string& t) { tag.replace(tag_offset, std::string::npos, t); };
373 
374  std::vector<Entry> spec_entries;
375  if (!spec_lines.empty()) {
376  spec_entries.reserve(spec_lines.size());
377  for (const std::string& line : spec_lines)
378  spec_entries.emplace_back(line);
379  } else {
380  const char** line = default_spec(!unmerged);
381  for (; *line != nullptr; ++line)
382  spec_entries.emplace_back(*line);
383  }
384 
385  // always start with H, K, L
386  set_tag("index_h");
387  for (char c : {'h', 'k', 'l'}) {
388  tag.back() = c;
389  int index = loop->find_tag(tag);
390  if (index == -1)
391  fail("Miller index tag not found: " + tag);
392  indices.push_back(index);
393  entries.push_back(nullptr);
394  auto col = mtz.columns.emplace(mtz.columns.end());
395  col->dataset_id = 0;
396  col->type = 'H';
397  col->label = alpha_up(c);
398  }
399 
400  // M/ISYM and BATCH
401  if (unmerged) {
402  auto col = mtz.columns.emplace(mtz.columns.end());
403  col->dataset_id = 0;
404  col->type = 'Y';
405  col->label = "M/ISYM";
406 
407  col = mtz.columns.emplace(mtz.columns.end());
408  col->dataset_id = 0;
409  col->type = 'B';
410  col->label = "BATCH";
411  }
412 
413  // other columns according to the spec
414  bool column_added = false;
415  for (const Entry& entry : spec_entries) {
416  set_tag(entry.refln_tag);
417  int index = loop->find_tag(tag);
418  if (index == -1)
419  continue;
420  if (mtz.column_with_label(entry.col_label))
421  continue;
422  column_added = true;
423  indices.push_back(index);
424  entries.push_back(entry.code_to_number.empty() ? nullptr : &entry);
425  auto col = mtz.columns.emplace(mtz.columns.end());
426  // dataset_id is meaningless in unmerged MTZ files
427  col->dataset_id = unmerged ? 0 : entry.dataset_id;
428  col->type = entry.col_type;
429  col->label = entry.col_label;
430  logger.level<7>(" ", tag, " -> ", col->label);
431  }
432  if (!column_added)
433  fail(force_unmerged ? "Unmerged d" : "D", "ata not found in block ", rb.block.name);
434 
435  for (size_t i = 0; i != mtz.columns.size(); ++i) {
436  mtz.columns[i].parent = &mtz;
437  mtz.columns[i].idx = i;
438  }
439  mtz.nreflections = (int) loop->length();
440 
441  std::unique_ptr<UnmergedHklMover> hkl_mover;
442  struct BatchInfo {
443  int sweep_id;
444  int frame_id;
445  };
446  std::vector<BatchInfo> batch_nums;
447  if (unmerged) {
448  hkl_mover.reset(new UnmergedHklMover(mtz.spacegroup));
449  set_tag("diffrn_id");
450  int sweep_id_index = loop->find_tag(tag);
451  if (sweep_id_index == -1)
452  logger.level<7>("No diffrn_id. Assuming a single sweep.");
453  set_tag("pdbx_image_id");
454  int image_id_index = loop->find_tag(tag);
455  if (logger.threshold >= 7) {
456  if (image_id_index == -1)
457  logger.mesg("No pdbx_image_id, setting BATCH to a dummy value.");
458  else
459  logger.mesg(" ", tag, " & diffrn_id -> BATCH");
460  }
461  struct SweepInfo {
462  std::set<int> frame_ids;
463  int offset;
464  float wavelength = 0.f;
465  std::string crystal_id = "unknown";
466  int dataset_id;
467  };
468  std::map<int, SweepInfo> sweeps;
469  cif::Block& block = const_cast<cif::Block&>(rb.block);
470  cif::Table tab_w0 = block.find("_diffrn.", {"id", "crystal_id"});
471  cif::Table tab_w1 = block.find("_diffrn_radiation.",
472  {"diffrn_id", "wavelength_id"});
473  cif::Table tab_w2 = block.find("_diffrn_radiation_wavelength.",
474  {"id", "wavelength"});
475  // store sweep and frame numbers corresponding to reflections
476  batch_nums.reserve(loop->length());
477  for (size_t i = 0; i < loop->values.size(); i += loop->tags.size()) {
478  int sweep_id = 1;
479  if (sweep_id_index >= 0)
480  sweep_id = cif::as_int(loop->values[i + sweep_id_index], 1);
481  int frame = 1;
482  if (image_id_index >= 0) {
483  double d = cif::as_number(loop->values[i + image_id_index], 1.);
484  frame = (int) std::ceil(d);
485  }
486  batch_nums.push_back({sweep_id, frame});
487  if (frame >= 0) {
488  SweepInfo& sweep = sweeps[sweep_id];
489  // if new sweep was added - try to set crystal_id and wavelength
490  if (sweep.frame_ids.empty() && sweep_id_index >= 0) {
491  const std::string& sweep_str = loop->values[i + sweep_id_index];
492  try {
493  sweep.crystal_id = tab_w0.find_row(sweep_str).str(1);
494  } catch(std::exception&) {}
495  try {
496  const std::string& wave_id = tab_w1.find_row(sweep_str)[1];
497  const std::string& wavelen = tab_w2.find_row(wave_id)[1];
498  sweep.wavelength = (float) cif::as_number(wavelen, 0.);
499  } catch(std::exception&) {}
500  }
501  sweep.frame_ids.insert(frame);
502  }
503  }
504 
505  // add datasets and set SweepInfo::dataset_id
506  for (auto& sweep_pair : sweeps) {
507  SweepInfo& si = sweep_pair.second;
508  Mtz::Dataset* ds = mtz.dataset_with_name(si.crystal_id);
509  if (ds == nullptr) {
510  ds = &mtz.add_dataset(si.crystal_id);
511  // both crystal name and dataset name are set to crystal_id
512  ds->project_name = "unknown";
513  ds->wavelength = si.wavelength;
514  }
515  si.dataset_id = ds->id;
516  }
517 
518  // determine offset that makes frame numbers unique
519  int cap = 0;
520  for (auto& it : sweeps) {
521  it.second.offset = cap;
522  cap += *--it.second.frame_ids.end() + 1100;
523  cap -= cap % 1000;
524  }
525  // add offset to BatchInfo::frame_id
526  for (BatchInfo& p : batch_nums)
527  if (p.sweep_id >= 0 && p.frame_id >= 0)
528  p.frame_id += sweeps.at(p.sweep_id).offset;
529 
530  // add MTZ batches
531  for (const auto& sweep_pair : sweeps) {
532  const SweepInfo& sweep_info = sweep_pair.second;
533  Mtz::Batch batch;
534  batch.set_dataset_id(sweep_info.dataset_id);
535  batch.set_cell(mtz.cell);
536  batch.set_wavelength(sweep_info.wavelength);
537  // FIXME should we set more properties in BATCH header?
538  int min_frame = *sweep_info.frame_ids.begin();
539  int max_frame = *--sweep_info.frame_ids.end();
540  if (2 * sweep_info.frame_ids.size() > size_t(max_frame - min_frame)) {
541  // probably consecutive range, even if some frames are missing
542  for (int n = min_frame; n <= max_frame; ++n) {
543  batch.number = n + sweep_info.offset;
544  mtz.batches.push_back(batch);
545  }
546  } else {
547  for (int n : sweep_info.frame_ids) {
548  batch.number = n + sweep_info.offset;
549  mtz.batches.push_back(batch);
550  }
551  }
552  }
553  } // - if (unmerged)
554 
555  // fill in the data
556  mtz.data.resize(mtz.columns.size() * mtz.nreflections);
557  size_t k = 0, row = 0;
558  for (size_t i = 0; i < loop->values.size(); i += loop->tags.size()) {
559  if (unmerged) {
560  std::array<int, 3> hkl;
561  for (size_t ii = 0; ii != 3; ++ii)
562  hkl[ii] = cif::as_int(loop->values[i + indices[ii]]);
563  int isym = hkl_mover->move_to_asu(hkl);
564  for (size_t j = 0; j != 3; ++j)
565  mtz.data[k++] = (float) hkl[j];
566  mtz.data[k++] = (float) isym;
567  mtz.data[k++] = batch_nums.empty() ? 1.f : (float) batch_nums[row++].frame_id;
568  } else {
569  for (size_t j = 0; j != 3; ++j)
570  mtz.data[k++] = (float) cif::as_int(loop->values[i + indices[j]]);
571  }
572  for (size_t j = 3; j != indices.size(); ++j) {
573  const std::string& v = loop->values[i + indices[j]];
574  if (cif::is_null(v)) {
575  mtz.data[k] = (float) NAN;
576  } else if (entries[j] != nullptr) {
577  mtz.data[k] = entries[j]->translate_code_to_number(v);
578  } else {
579  mtz.data[k] = (float) cif::as_number(v);
580  if (std::isnan(mtz.data[k]))
581  logger.mesg("Value #", i + indices[j], " in the loop is not a number: ", v);
582  }
583  ++k;
584  }
585  }
586  return mtz;
587  }

606  Mtz auto_convert_block_to_mtz(ReflnBlock& rb, Logger& logger, char mode) const {
607  if (mode == 'f' && possible_old_style(rb, DataType::Anomalous))
608  *rb.refln_loop = transcript_old_anomalous_to_standard(*rb.refln_loop, rb.spacegroup);
609  Mtz mtz = convert_block_to_mtz(rb, logger);
610  if (mtz.is_merged() && mode == 'a') {
611  auto type_unique = check_data_type_under_symmetry(MtzDataProxy{mtz});
612  if (type_unique.first == DataType::Anomalous) {
613  if (possible_old_style(rb, DataType::Anomalous)) {
614  logger.note("data in ", rb.block.name,
615  " is read as \"old-style\" anomalous (", rb.refln_loop->length(),
616  " -> ", type_unique.second, " rows).");
617  *rb.refln_loop = transcript_old_anomalous_to_standard(*rb.refln_loop,
618  rb.spacegroup);
619  // this is rare, so it's OK to run the conversion twice
620  mtz = convert_block_to_mtz(rb, logger);
621  } else {
622  logger.err("in ", rb.block.name, ", out of ",
623  rb.refln_loop->length(), " HKLs, only ", type_unique.second,
624  " are unique under symmetry; the rest are equivalent to Friedel mates");
625  }
626  } else if (type_unique.first == DataType::Unmerged) {
627  logger.err("in ", rb.block.name, ", out of ",
628  rb.refln_loop->length(), " HKLs, only ", type_unique.second,
629  " are unique under symmetry");
630  if (possible_old_style(rb, gemmi::DataType::Unmerged))
631  logger.mesg("Possibly unmerged data - you may use option --refln-to=unmerged");
