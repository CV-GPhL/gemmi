32 void for_all_reflections(Func func,
33  const UnitCell& cell, const SpaceGroup* spacegroup,
34  double dmin, double dmax=0., bool unique=true) {
35  Miller lim = cell.get_hkl_limits(dmin);
36  double inv_dmin2 = 1. / sq(dmin);
37  double inv_dmax2 = 0.;
38  if (dmax > 0)
39  inv_dmax2 = dmax == INFINITY ? -1 : 1. / sq(dmax);
40  ReciprocalAsu asu(spacegroup);
41  GroupOps gops = spacegroup->operations();
42  Miller hkl;
43  for (hkl[0] = -lim[0]; hkl[0] <= lim[0]; ++hkl[0])
44  for (hkl[1] = -lim[1]; hkl[1] <= lim[1]; ++hkl[1])
45  for (hkl[2] = -lim[2]; hkl[2] <= lim[2]; ++hkl[2])
46  if (!unique || asu.is_in(hkl)) {
47  double inv_d2 = cell.calculate_1_d2(hkl);
48  if (inv_d2 <= inv_dmin2 && inv_d2 > inv_dmax2 &&
49  !gops.is_systematically_absent(hkl))
50  func(hkl);
51  }
52 }
