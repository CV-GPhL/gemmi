25 template<> inline void remove_alternative_conformations(Chain& chain) {
26  std::set<SeqId> seqids;
27  for (size_t i = 0; i < chain.residues.size(); ) {
28  if (seqids.insert(chain.residues[i].seqid).second)
29  ++i;
30  else
31  chain.residues.erase(chain.residues.begin() + i);
32  }
33  for (Residue& residue : chain.residues) {
34  std::set<std::string> names;
35  for (size_t i = 0; i < residue.atoms.size(); ) {
36  Atom& atom = residue.atoms[i];
37  atom.altloc = '\0';
38  if (names.insert(atom.name).second)
39  ++i;
40  else
41  residue.atoms.erase(residue.atoms.begin() + i);
42  }
43  }
44 }

223 inline void rename_residues(Structure& st, const std::string& old_name,
224  const std::string& new_name) {
225  auto update = [&](ResidueId& rid) {
226  if (rid.name == old_name)
227  rid.name = new_name;
228  };
229  process_addresses(st, [&](AtomAddress& aa) { update(aa.res_id); });
230  for (StructSite& site : st.sites)
231  for (StructSite::Member& member : site.members)
232  if (member.label_comp_id == old_name)
233  member.label_comp_id = new_name;
234  for (ModRes& modres : st.mod_residues)
235  update(modres.res_id);
236  for (Entity& ent : st.entities)
237  for (std::string& mon_ids : ent.full_sequence)
238  for (size_t start = 0;;) {
239  size_t end = mon_ids.find(',', start);
240  if (mon_ids.compare(start, end-start, old_name) == 0) {
241  mon_ids.replace(start, end-start, new_name);
242  if (end != std::string::npos)
243  end = start + new_name.size();
244  }
245  if (end == std::string::npos)
246  break;
247  start = end + 1;
248  }
249  for (Model& model : st.models)
250  for (Chain& chain : model.chains)
251  for (Residue& res : chain.residues)
252  update(res);
253 }

287 inline void replace_d_fraction_with_altlocs(Residue& res) {
288  for (size_t i = res.atoms.size(); i-- != 0; ) {
289  Atom& atom = res.atoms[i];
290  float d_fraction = atom.fraction;
291  if (atom.element == El::H && d_fraction > 0) {
292  if (d_fraction >= 1) {
293  atom.element = El::D;
294  if (atom.name[0] == 'H')
295  atom.name[0] = 'D';
296  } else {
297  int alt_offset = atom.altloc;
298  if (alt_offset) {
299  alt_offset -= 'A';
300  // we don't expect 4+ altlocs - ignore such cases
301  if (alt_offset < 0 || alt_offset >= 3)
302  continue;
303  }
304  atom.altloc = 'A' + alt_offset;
305  float d_occ = atom.occ * d_fraction;
306  atom.occ *= (1 - d_fraction);
307  auto deut = res.atoms.insert(res.atoms.begin() + i + 1, atom);
308  deut->altloc = 'D' + alt_offset;
309  deut->element = El::D;
310  deut->occ = d_occ;
311  if (deut->name[0] == 'H')
312  deut->name[0] = 'D';
313  }
314  }
315  }
316 }

323 inline bool replace_deuterium_with_fraction(Residue& res) {
324  bool found = false;
325  for (auto d = res.atoms.end(); d-- != res.atoms.begin(); )
326  if (d->element == El::D) {
327  found = true;
328  auto h = res.atoms.begin();
329  for (; h != res.atoms.end(); ++h)
330  if (h->element == El::H && h->pos.approx(d->pos, 1e-9))
331  break;
332  if (h != res.atoms.end()) {
333  h->occ += d->occ;
334  h->fraction = h->occ > 0.f ? d->occ / h->occ : 0.f;
335  if (h->altloc) {
336  bool keep_altloc = false;
337  for (auto i = res.atoms.begin(); i != res.atoms.end(); ++i)
338  if (i != d && i != h && (i->name == h->name || i->name == d->name))
339  keep_altloc = true;
340  if (!keep_altloc)
341  h->altloc = '\0';
342  }
343  res.atoms.erase(d);
344  } else {
345  d->element = El::H;
346  d->fraction = 1;
347  // Atom name is left unchanged. prepare_topology() first calls this
348  // function and then conditionally changes the name (Dxx -> Hxx).
349  }
350  }
351  return found;
352 }

396 inline void standardize_crystal_frame(Structure& st) {
397  if (!st.cell.explicit_matrices || !st.cell.is_crystal())
398  return;
399  Transform orig_frac = st.cell.frac;
400  st.cell.explicit_matrices = false;
401  st.cell.calculate_properties();
402  Transform tr = st.cell.orth.combine(orig_frac);
403  Transform tr_inv = tr.inverse();
404  st.has_origx = true;
405  st.origx = tr_inv.combine(st.origx);
406  for (NcsOp& ncsop : st.ncs)
407  ncsop.tr = tr.combine(ncsop.tr).combine(tr_inv);
408  transform_pos_and_adp(st, tr);
409 }
