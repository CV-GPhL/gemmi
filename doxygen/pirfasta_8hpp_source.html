71 inline std::vector<FastaSeq> read_pir_or_fasta(const std::string& str) {
72  if (str[0] != '>')
73  fail("PIR/FASTA files start with '>'");
74  bool pir = is_pir_format(str);
75  std::vector<FastaSeq> r;
76  int blank_lines = 0;
77  int paren_level = 0;
78  bool ended = false;
79  for (size_t pos=0, end=0; end != std::string::npos; pos = end + 1) {
80  end = str.find('\n', pos);
81  if (str[pos] == '>') {
82  ended = false;
83  if (paren_level != 0)
84  break;
85  r.emplace_back();
86  if (pir && end != std::string::npos)
87  end = str.find('\n', end+1);
88  r.back().header = str.substr(pos+1, end-(pos+1));
89  } else {
90  std::string& seq = r.back().seq;
91  for (size_t i = pos; i < std::min(end, str.size()); ++i) {
92  char c = str[i];
93  if (std::isspace(c)) {
94  if (c == '\n')
95  ++blank_lines;
96  continue;
97  }
98  // handle non-blank characters
99  if (ended)
100  fail("'*' is interpreted as sequence terminator, here it is followed by: ", c);
101  if (blank_lines >= 2)
102  fail("blank lines can be followed only by a line starting with '>'");
103  blank_lines = 0;
104  if (('a' <= (c | 0x20) && (c | 0x20) <= 'z') || c == '-' ||
105  (paren_level != 0 && '0' <= c && c <= '9')) {
106  // good character, nothing to be done here
107  } else if (c == '*') {
108  ended = true;
109  continue;
110  } else if (c == '(') {
111  if (++paren_level > 1)
112  fail("nested parentheses are not allowed");
113  } else if (c == ')') {
114  if (--paren_level < 0)
115  fail("')' without matching '('");
116  } else {
117  fail("unexpected character in sequence: ", c);
118  }
119  seq += c;
120  }
121  }
122  }
123  if (paren_level != 0)
124  fail("unmatched '('");
125  return r;
126 }
