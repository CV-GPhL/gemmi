48  template<int TableVal> struct lookup_char {
49  using analyze_t = pegtl::analysis::generic<pegtl::analysis::rule_type::ANY>;
50  template<typename Input> static bool match(Input& in) {
51  if (!in.empty() && cif::char_table(in.peek_char()) == TableVal) {
52  if (TableVal == 2) // this set includes new-line
53  in.bump(1);
54  else
55  in.bump_in_this_line(1);
56  return true;
57  }
58  return false;
59  }
60  };

91  // (e) Character strings and text fields.
92  template<typename Q>
93  struct endq : pegtl::seq<Q, pegtl::at<pegtl::sor<
94  pegtl::one<' ','\n','\r','\t','#'>,
95  pegtl::eof>>> {};

103  struct singlequoted : quoted<pegtl::one<'\''>> {};
104  struct doublequoted : quoted<pegtl::one<'"'>> {};
105  struct field_sep : pegtl::seq<pegtl::bol, pegtl::one<';'>> {};

107  // until<endq<field_sep>> instead of until<field_sep>.
108  struct textfield : pegtl::if_must<field_sep, pegtl::until<field_sep>> {};
109  struct unquoted : pegtl::seq<pegtl::not_at<keyword>,
110  pegtl::not_at<pegtl::one<'_','$','#'>>,

113  // (a) Basic structure of CIF. (c) Tags and values.
114  // datablockname in STAR/CIF should not be empty, but we made an exception
115  // for RELION which writes blocks starting with bare data_
116  struct datablockname : pegtl::star<nonblank_ch> {};
117  struct datablockheading : pegtl::sor<pegtl::seq<str_data, datablockname>, str_global> {};

125  struct item_value : value {};

203 template<> struct Action<rules::datablockname> {
204  template<typename Input> static void apply(const Input& in, Document& out) {
205  out.blocks.emplace_back(in.string());
206  Block& block = out.blocks.back();
207  // Empty block name (just data_ ) is not STAR/CIF conformant,
208  // but it's written by RELION and buccaneer; we must support it.
209  if (block.name.empty())
210  block.name += ' ';
211  out.items_ = &block.items;
212  }

294 template<typename Input> Document read_input(Input&& in, int check_level=1) {
295  Document doc;
296  doc.source = in.source();
297  parse_input(doc, in);
298  if (check_level > 0) {
299  check_for_missing_values(doc);
300  check_for_duplicates(doc);
301  if (check_level > 1) {
302  for (const cif::Block& block : doc.blocks) {
303  if (block.name == " ")
304  fail(doc.source + ": missing block name (bare data_)");
305  check_empty_loops(block, doc.source);
306  }
307  }
308  }
309  return doc;
310 }
