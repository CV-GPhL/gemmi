54 inline Blob make_blob_of_points(const std::vector<GridConstPoint>& points,
55  const GridMeta& grid,
56  const BlobCriteria& criteria) {
57  Blob blob;
58  if (points.size() < 3)
59  return blob;
60  double volume_per_point = grid.unit_cell.volume / grid.point_count();
61  double volume = points.size() * volume_per_point;
62  if (volume < criteria.min_volume)
63  return blob;
64  double sum[3] = {0., 0., 0.};
65  const GridConstPoint* peak_point = &points[0];
66  blob.peak_value = points[0].value;
67  double score = 0.;
68  for (const GridConstPoint& point : points) {
69  score += point.value;
70  if (point.value > blob.peak_value) {
71  blob.peak_value = point.value;
72  peak_point = &point;
73  }
74  sum[0] += double(point.u) * point.value;
75  sum[1] += double(point.v) * point.value;
76  sum[2] += double(point.w) * point.value;
77  }
78  if (blob.peak_value < criteria.min_peak)
79  return blob;
80  blob.score = score * volume_per_point;
81  if (blob.score < criteria.min_score)
82  return blob;
83  gemmi::Fractional fract(sum[0] / (score * grid.nu),
84  sum[1] / (score * grid.nv),
85  sum[2] / (score * grid.nw));
86  blob.centroid = grid.unit_cell.orthogonalize(fract);
87  blob.peak_pos = grid.get_position(peak_point->u, peak_point->v, peak_point->w);
88  blob.volume = volume;
89  return blob;
90 }

105 inline std::vector<Blob> find_blobs_by_flood_fill(const gemmi::Grid<float>& grid,
106  const BlobCriteria& criteria,
107  bool negate=false) {
108  std::vector<Blob> blobs;
109  std::array<std::array<int, 3>, 6> moves = {{{{-1, 0, 0}}, {{1, 0, 0}},
110  {{0 ,-1, 0}}, {{0, 1, 0}},
111  {{0, 0, -1}}, {{0, 0, 1}}}};
112  // the mask will be used as follows:
113  // -1=in blob, 0=in asu, not in blob (so far), 1=in neither
114  std::vector<std::int8_t> mask = gemmi::get_asu_mask(grid);
115  std::vector<gemmi::GridOp> ops = grid.get_scaled_ops_except_id();
116  size_t idx = 0;
117  for (int w = 0; w != grid.nw; ++w)
118  for (int v = 0; v != grid.nv; ++v)
119  for (int u = 0; u != grid.nu; ++u, ++idx) {
120  assert(idx == grid.index_q(u, v, w));
121  if (mask[idx] != 0)
122  continue;
123  float value = grid.data[idx];
124  if (negate)
125  value = -value;
126  if (value < criteria.cutoff)
127  continue;
128  std::vector<impl::GridConstPoint> points;
129  points.push_back({u, v, w, value});
130  mask[idx] = -1;
131  for (size_t j = 0; j < points.size()/*increasing!*/; ++j)
132  for (const std::array<int, 3>& mv : moves) {
133  int nabe_u = points[j].u + mv[0];
134  int nabe_v = points[j].v + mv[1];
135  int nabe_w = points[j].w + mv[2];
136  size_t nabe_idx = grid.index_s(nabe_u, nabe_v, nabe_w);
137  if (mask[nabe_idx] == -1)
138  continue;
139  float nabe_value = grid.data[nabe_idx];
140  if (negate)
141  nabe_value = -nabe_value;
142  if (nabe_value > criteria.cutoff) {
143  if (mask[nabe_idx] != 0)
144  for (const gemmi::GridOp& op : ops) {
145  auto t = op.apply(nabe_u, nabe_v, nabe_w);
146  size_t mate_idx = grid.index_s(t[0], t[1], t[2]);
147  if (mask[mate_idx] == 0)
148  mask[mate_idx] = 1;
149  }
150  mask[nabe_idx] = -1;
151  points.push_back({nabe_u, nabe_v, nabe_w, nabe_value});
152  }
153  }
154  if (Blob blob = impl::make_blob_of_points(points, grid, criteria))
155  blobs.push_back(blob);
156  }
157  std::sort(blobs.begin(), blobs.end(),
158  [](const Blob& a, const Blob& b) { return a.score > b.score; });
159  return blobs;
160 }
