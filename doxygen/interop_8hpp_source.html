19 inline SmallStructure::Site atom_to_site(const Atom& atom, const UnitCell& cell) {
20  SmallStructure::Site site;
21  site.label = atom.name;
22  site.type_symbol = atom.element.name();
23  site.fract = cell.fractionalize(atom.pos);
24  site.occ = atom.occ;
25  // occupancy may need to be adjusted if the atom is on special position
26  if (atom.occ <= 0.5) {
27  int n_mates = cell.is_special_position(atom.pos);
28  if (n_mates > 0 && atom.occ * (n_mates + 1) <= 1.0)
29  site.occ = atom.occ * (n_mates + 1);
30  }
31  site.u_iso = atom.b_iso / u_to_b();
32  if (atom.aniso.nonzero()) {
33  if (cell.alpha == 90. || cell.beta == 90. || cell.gamma == 90.) {
34  site.aniso = atom.aniso.scaled(1.0);
35  } else {
36  SMat33<double> t = atom.aniso.transformed_by<>(cell.frac.mat);
37  Vec3 v = {1.0 / cell.ar, 1.0 / cell.br, 1.0 / cell.cr};
38  site.aniso = {t.u11 * v.x * v.x,
39  t.u22 * v.y * v.y,
40  t.u33 * v.z * v.z,
41  t.u12 * v.x * v.y,
42  t.u13 * v.x * v.z,
43  t.u23 * v.y * v.z};
44  }
45  }
46  site.element = atom.element;
47  site.charge = atom.charge;
48  return site;
49 }

56 inline SmallStructure mx_to_sx_structure(const Structure& st, int n=0) {
57  const Model& model = st.models.at(n);
58  SmallStructure small_st;
59  small_st.name = st.name;
60  small_st.cell = st.cell;
61  small_st.spacegroup_hm = st.spacegroup_hm;
62  for (const Chain& chain : model.chains)
63  for (const Residue& residue : chain.residues)
64  for (const Atom& atom : residue.atoms)
65  small_st.sites.push_back(atom_to_site(atom, st.cell));
66  return small_st;
67 }
