241  std::pair<Mark*, double>
242  find_nearest_atom_within_k(const Position& pos, int k, double radius) {
243  Mark* mark = nullptr;
244  double nearest_dist_sq = radius * radius;
245  for_each_cell(pos, [&](std::vector<Mark>& marks, const Fractional& fr) {
246  Position p = use_pbc ? grid.unit_cell.orthogonalize(fr) : pos;
247  for (Mark& m : marks) {
248  double dist_sq = m.pos.dist_sq(p);
249  if (dist_sq < nearest_dist_sq) {
250  mark = &m;
251  nearest_dist_sq = dist_sq;
252  }
253  }
254  }, k);

263  Mark* find_nearest_atom(const Position& pos, double radius=INFINITY) {
264  double r_spec = radius_specified;
265  if (radius == 0.f)
266  radius = r_spec;
267  int max_k = std::max(std::max(std::max(grid.nu, grid.nv), grid.nw), 2);
268  for (int k = 1; k < max_k; k *= 2) {
269  auto result = find_nearest_atom_within_k(pos, k, radius);
270  // if Mark was not found, result.second is set to radius^2.
271  if (result.second < sq(k * r_spec))
272  return result.first;
273  if (result.first != nullptr) {
274  // We found an atom, but because it was further away than k*r_spec,
275  // so now it's sufficient to find the nearest atom in dist:
276  double dist = std::sqrt(result.second);
277  return find_nearest_atom_within_k(pos, sufficient_k(dist), radius).first;
278  }
279  }
280  if (!use_pbc)
281  // pos can be outside of bounding box. In such case, although it's slow,
282  // search in all cells. Using large number that will be clipped.
283  return find_nearest_atom_within_k(pos, INT_MAX/4, radius).first;
284  return nullptr;
285  }

326  void set_bounding_cell(const UnitCell& cell) {
327  use_pbc = cell.is_crystal();
328  if (use_pbc) {
329  grid.unit_cell = cell;
330  } else {
331  // cf. calculate_box()
332  Box<Position> box;
333  for (CRA cra : model->all())
334  box.extend(cra.atom->pos);
335  // The box needs to include all NCS images (strict NCS from MTRIXn).
336  // To avoid additional function parameter that would pass Structure::ncs,
337  // here we obtain NCS transformations from UnitCell::images.
338  std::vector<FTransform> ncs = cell.get_ncs_transforms();
339  if (!ncs.empty()) {
340  for (CRA cra : model->all())
341  // images store fractional transforms, but for non-crystal
342  // it should be the same as Cartesian transform.
343  for (const Transform& tr : ncs)
344  box.extend(Position(tr.apply(cra.atom->pos)));
345  }
346  box.add_margin(0.01);
347  Position size = box.get_size();
348  grid.unit_cell.set(size.x, size.y, size.z, 90, 90, 90);
349  grid.unit_cell.frac.vec -= grid.unit_cell.fractionalize(box.minimum);
350  grid.unit_cell.orth.vec += box.minimum;
351  for (const Transform& tr : ncs) {
352  UnitCell& c = grid.unit_cell;
353  // cf. add_ncs_images_to_cs_images()
354  c.images.push_back(c.frac.combine(tr.combine(c.orth)));
355  }
356  }

401 inline void NeighborSearch::add_atom(const Atom& atom,
402  int n_ch, int n_res, int n_atom) {
403  const UnitCell& gcell = grid.unit_cell;
404  Fractional frac0 = gcell.fractionalize(atom.pos);
405  {
406  Fractional frac = frac0.wrap_to_unit();
407  // for non-crystals, frac==frac0 => pos = atom.pos
408  Position pos = use_pbc ? gcell.orthogonalize(frac) : atom.pos;
409  get_subcell(frac).emplace_back(pos, atom.altloc, atom.element.elem,
410  0, n_ch, n_res, n_atom);
411  }
412  for (int n_im = 0; n_im != (int) gcell.images.size(); ++n_im) {
413  Fractional frac = gcell.images[n_im].apply(frac0).wrap_to_unit();
414  Position pos = gcell.orthogonalize(frac);
415  get_subcell(frac).emplace_back(pos, atom.altloc, atom.element.elem,
416  short(n_im + 1), n_ch, n_res, n_atom);
417  }
418 }

420 // We exclude special position images of atoms here, but not in add_atom.
421 // This choice is somewhat arbitrary, but it also reflects the fact that
422 // in MX files occupances of atoms on special positions are (almost always)
423 // fractional and all images are to be taken into account.
424 inline void NeighborSearch::add_site(const SmallStructure::Site& site, int n) {
425  const double SPECIAL_POS_TOL = 0.4;
426  const UnitCell& gcell = grid.unit_cell;
427  std::vector<Fractional> others;
428  others.reserve(gcell.images.size());
429  Fractional frac0 = site.fract.wrap_to_unit();
430  {
431  Position pos = gcell.orthogonalize(frac0);
432  get_subcell(frac0).emplace_back(pos, '\0', site.element.elem, 0, -1, -1, n);
433  }
434  for (int n_im = 0; n_im != (int) gcell.images.size(); ++n_im) {
435  Fractional frac = gcell.images[n_im].apply(site.fract).wrap_to_unit();
436  if (gcell.distance_sq(frac, frac0) < sq(SPECIAL_POS_TOL) ||
437  std::any_of(others.begin(), others.end(), [&](const Fractional& f) {
438  return gcell.distance_sq(frac, f) < sq(SPECIAL_POS_TOL);
439  }))
440  continue;
441  Position pos = gcell.orthogonalize(frac);
442  get_subcell(frac).emplace_back(pos, '\0', site.element.elem,
443  short(n_im + 1), -1, -1, n);
444  others.push_back(frac);
445  }

448 template<typename Func>
449 void NeighborSearch::for_each_cell(const Position& pos, const Func& func, int k) {
450  Fractional fr = grid.unit_cell.fractionalize(pos);
451  if (use_pbc)
452  fr = fr.wrap_to_unit();
453  int u0 = int(fr.x * grid.nu) - k;
454  int v0 = int(fr.y * grid.nv) - k;
455  int w0 = int(fr.z * grid.nw) - k;
456  int uend = u0 + 2 * k + 1;
457  int vend = v0 + 2 * k + 1;
458  int wend = w0 + 2 * k + 1;
459  if (use_pbc) {
460  auto shift = [](int j, int n) {
461  if (j < 0)
462  return (j + 1) / n - 1;
463  if (j >= n)
464  return j / n;
465  return 0;
466  };
467  for (int w = w0; w < wend; ++w) {
468  int dw = shift(w, grid.nw);
469  for (int v = v0; v < vend; ++v) {
470  int dv = shift(v, grid.nv);
471  size_t idx0 = grid.index_q(0, v - dv * grid.nv, w - dw * grid.nw);
472  for (int u = u0; u < uend; ++u) {
473  int du = shift(u, grid.nu);
474  size_t idx = idx0 + (u - du * grid.nu);
475  func(grid.data[idx], Fractional(fr.x - du, fr.y - dv, fr.z - dw));
476  }
477  }
478  }
479  } else {
480  u0 = std::max(0, u0);
481  v0 = std::max(0, v0);
482  w0 = std::max(0, w0);
483  uend = std::min(uend, grid.nu);
484  vend = std::min(vend, grid.nv);
485  wend = std::min(wend, grid.nw);
486  for (int w = w0; w < wend; ++w)
487  for (int v = v0; v < vend; ++v)
488  for (int u = u0; u < uend; ++u) {
489  size_t idx = grid.index_q(u, v, w);
490  func(grid.data[idx], fr);
491  }
492  }
493 }
