63 inline double log_cosh(double x) {
64  // cosh(x) would overflow for x > 710.5, so we calculate:
65  // ln(cosh(x)) = ln(e^x + e^-x) - ln(2) = ln(e^x * (1 + e^-2x)) - ln(2)
66  x = std::abs(x);
67  return x - std::log(2) + std::log1p(std::exp(-2 * x));
68 }

641  std::array<double, 3> calculate_eigenvalues() const {
642  double p1 = u12*u12 + u13*u13 + u23*u23;
643  if (p1 == 0)
644  return {{u11, u22, u33}};
645  double q = (1./3.) * trace();
646  SMat33<double> b{u11 - q, u22 - q, u33 - q, u12, u13, u23};
647  double p2 = sq(b.u11) + sq(b.u22) + sq(b.u33) + 2 * p1;
648  double p = std::sqrt((1./6.) * p2);
649  double r = b.determinant() / ((1./3.) * p2 * p);
650  double phi = 0;
651  if (r <= -1)
652  phi = (1./3.) * pi();
653  else if (r < 1)
654  phi = (1./3.) * std::acos(r);
655  double eig1 = q + 2 * p * std::cos(phi);
656  double eig3 = q + 2 * p * std::cos(phi + 2./3.*pi());
657  return {{eig1, 3 * q - eig1 - eig3, eig3}};
658  }
